629  def _add_items(self, source_field_name, target_field_name, *objs):
630  # source_field_name: the PK fieldname in join table for the source object
631  # target_field_name: the PK fieldname in join table for the target object
632  # *objs - objects to add. Either object instances, or primary keys of object instances.
633 
634  # If there aren't any objects, there is nothing to do.
635  from django.db.models import Model
636  if objs:
637  new_ids = set()
638  for obj in objs:
639  if isinstance(obj, self.model):
640  if not router.allow_relation(obj, self.instance):
641  raise ValueError('Cannot add "%r": instance is on database "%s", value is on database "%s"' %
642  (obj, self.instance._state.db, obj._state.db))

644  if fk_val is None:
645  raise ValueError('Cannot add "%r": the value for field "%s" is None' %
646  (obj, target_field_name))
647  new_ids.add(self._get_fk_val(obj, target_field_name))
648  elif isinstance(obj, Model):
649  raise TypeError("'%s' instance expected, got %r" % (self.model._meta.object_name, obj))
650  else:

652  db = router.db_for_write(self.through, instance=self.instance)

654  vals = vals.filter(**{
655  source_field_name: self._fk_val,
656  '%s__in' % target_field_name: new_ids,
657  })
658  new_ids = new_ids - set(vals)

660  if self.reverse or source_field_name == self.source_field_name:
661  # Don't send the signal when we are inserting the
662  # duplicate data row for symmetrical reverse entries.
663  signals.m2m_changed.send(sender=self.through, action='pre_add',
664  instance=self.instance, reverse=self.reverse,
665  model=self.model, pk_set=new_ids, using=db)
666  # Add the ones that aren't there already
667  self.through._default_manager.using(db).bulk_create([
668  self.through(**{
669  '%s_id' % source_field_name: self._fk_val,
670  '%s_id' % target_field_name: obj_id,
671  })
672  for obj_id in new_ids
673  ])
674 
675  if self.reverse or source_field_name == self.source_field_name:
676  # Don't send the signal when we are inserting the
677  # duplicate data row for symmetrical reverse entries.
678  signals.m2m_changed.send(sender=self.through, action='post_add',
679  instance=self.instance, reverse=self.reverse,
680  model=self.model, pk_set=new_ids, using=db)

683  # source_field_name: the PK colname in join table for the source object
684  # target_field_name: the PK colname in join table for the target object
685  # *objs - objects to remove

688  if objs:

697  db = router.db_for_write(self.through, instance=self.instance)
