57 def code_changed():
58  global _mtimes, _win
59  filenames = []
60  for m in list(sys.modules.values()):
61  try:
62  filenames.append(m.__file__)
63  except AttributeError:
64  pass
65  for filename in filenames + _error_files:
66  if not filename:
67  continue
68  if filename.endswith(".pyc") or filename.endswith(".pyo"):
69  filename = filename[:-1]
70  if filename.endswith("$py.class"):
71  filename = filename[:-9] + ".py"
72  if not os.path.exists(filename):
73  continue # File might be in an egg, so it can't be reloaded.
74  stat = os.stat(filename)
75  mtime = stat.st_mtime
76  if _win:
77  mtime -= stat.st_ctime
78  if filename not in _mtimes:
79  _mtimes[filename] = mtime
80  continue
81  if mtime != _mtimes[filename]:
82  _mtimes = {}
83  try:
84  del _error_files[_error_files.index(filename)]
85  except ValueError:
86  pass
87  return True
88  return False
