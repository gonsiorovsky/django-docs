30 class WhereNode(tree.Node):
31  """
32  Used to represent the SQL where-clause.
33 
34  The class is tied to the Query class that created it (in order to create
35  the correct SQL).
36 
37  A child is usually a tuple of:
38  (Constraint(alias, targetcol, field), lookup_type, value)
39  where value can be either raw Python value, or Query, ExpressionNode or
40  something else knowing how to turn itself into SQL.
41 
42  However, a child could also be any class with as_sql() and either
43  relabeled_clone() method or relabel_aliases() and clone() methods. The
44  second alternative should be used if the alias is not the only mutable
45  variable.
46  """

49  def _prepare_data(self, data):
50  """
51  Prepare data for addition to the tree. If the data is a list or tuple,
52  it is expected to be of the form (obj, lookup_type, value), where obj
53  is a Constraint object, and is then slightly munged before being
54  stored (to avoid storing any reference to field objects). Otherwise,
55  the 'data' is stored unchanged and can be any class with an 'as_sql()'
56  method.
57  """
58  if not isinstance(data, (list, tuple)):
59  return data
60  obj, lookup_type, value = data
61  if is_iterator(value):
62  # Consume any generators immediately, so that we can determine
63  # emptiness and transform any non-empty values correctly.
64  value = list(value)
65 
66  # The "value_annotation" parameter is used to pass auxilliary information
67  # about the value(s) to the query construction. Specifically, datetime
68  # and empty values need special handling. Other types could be used
69  # here in the future (using Python types is suggested for consistency).
70  if (isinstance(value, datetime.datetime)
71  or (isinstance(obj.field, DateTimeField) and lookup_type != 'isnull')):
72  value_annotation = datetime.datetime
73  elif hasattr(value, 'value_annotation'):
74  value_annotation = value.value_annotation
75  else:
76  value_annotation = bool(value)
77 
78  if hasattr(obj, "prepare"):
79  value = obj.prepare(lookup_type, value)
80  return (obj, lookup_type, value_annotation, value)

82  def as_sql(self, qn, connection):
83  """
84  Returns the SQL version of the where clause and the value to be
85  substituted in. Returns '', [] if this node matches everything,
86  None, [] if this node is empty, and raises EmptyResultSet if this
87  node can't match anything.
88  """
89  # Note that the logic here is made slightly more complex than
90  # necessary because there are two kind of empty nodes: Nodes
91  # containing 0 children, and nodes that are known to match everything.
92  # A match-everything node is different than empty node (which also
93  # technically matches everything) for backwards compatibility reasons.
94  # Refs #5261.
95  result = []
96  result_params = []
97  everything_childs, nothing_childs = 0, 0
98  non_empty_childs = len(self.children)
99 
100  for child in self.children:
101  try:
102  if hasattr(child, 'as_sql'):
103  sql, params = child.as_sql(qn=qn, connection=connection)
104  else:
105  # A leaf node in the tree.
106  sql, params = self.make_atom(child, qn, connection)
107  except EmptyResultSet:
108  nothing_childs += 1
109  else:
110  if sql:
111  result.append(sql)
112  result_params.extend(params)
113  else:
114  if sql is None:
115  # Skip empty childs totally.
116  non_empty_childs -= 1
117  continue
118  everything_childs += 1
119  # Check if this node matches nothing or everything.
120  # First check the amount of full nodes and empty nodes
121  # to make this node empty/full.
122  if self.connector == AND:
123  full_needed, empty_needed = non_empty_childs, 1
124  else:
125  full_needed, empty_needed = 1, non_empty_childs
126  # Now, check if this node is full/empty using the
127  # counts.
128  if empty_needed - nothing_childs <= 0:
129  if self.negated:
130  return '', []
131  else:
132  raise EmptyResultSet
133  if full_needed - everything_childs <= 0:
134  if self.negated:
135  raise EmptyResultSet
136  else:
137  return '', []
138 
139  if non_empty_childs == 0:
140  # All the child nodes were empty, so this one is empty, too.
141  return None, []
142  conn = ' %s ' % self.connector
143  sql_string = conn.join(result)
144  if sql_string:
145  if self.negated:
146  # Some backends (Oracle at least) need parentheses
147  # around the inner SQL in the negated case, even if the
148  # inner SQL contains just a single expression.
149  sql_string = 'NOT (%s)' % sql_string
150  elif len(result) > 1:
151  sql_string = '(%s)' % sql_string
152  return sql_string, result_params

166  def make_atom(self, child, qn, connection):
167  """
168  Turn a tuple (Constraint(table_alias, column_name, db_type),
169  lookup_type, value_annotation, params) into valid SQL.
170 
171  The first item of the tuple may also be an Aggregate.
172 
173  Returns the string for the SQL fragment and the parameters to use for
174  it.
175  """
176  lvalue, lookup_type, value_annotation, params_or_value = child
177  field_internal_type = lvalue.field.get_internal_type() if lvalue.field else None
178 
179  if isinstance(lvalue, Constraint):
180  try:
181  lvalue, params = lvalue.process(lookup_type, params_or_value, connection)
182  except EmptyShortCircuit:
183  raise EmptyResultSet
184  elif isinstance(lvalue, Aggregate):
185  params = lvalue.field.get_db_prep_lookup(lookup_type, params_or_value, connection)
186  else:
187  raise TypeError("'make_atom' expects a Constraint or an Aggregate "
188  "as the first item of its 'child' argument.")
189 
190  if isinstance(lvalue, tuple):
191  # A direct database column lookup.
192  field_sql, field_params = self.sql_for_columns(lvalue, qn, connection, field_internal_type), []
193  else:
194  # A smart object with an as_sql() method.
195  field_sql, field_params = lvalue.as_sql(qn, connection)
196 
197  is_datetime_field = value_annotation is datetime.datetime
198  cast_sql = connection.ops.datetime_cast_sql() if is_datetime_field else '%s'
199 
200  if hasattr(params, 'as_sql'):
201  extra, params = params.as_sql(qn, connection)
202  cast_sql = ''
203  else:
204  extra = ''
205 
206  params = field_params + params
207 
208  if (len(params) == 1 and params[0] == '' and lookup_type == 'exact'
209  and connection.features.interprets_empty_strings_as_nulls):
210  lookup_type = 'isnull'
211  value_annotation = True
212 
213  if lookup_type in connection.operators:
214  format = "%s %%s %%s" % (connection.ops.lookup_cast(lookup_type),)
215  return (format % (field_sql,
216  connection.operators[lookup_type] % cast_sql,
217  extra), params)
218 
219  if lookup_type == 'in':
220  if not value_annotation:
221  raise EmptyResultSet
222  if extra:
223  return ('%s IN %s' % (field_sql, extra), params)
224  max_in_list_size = connection.ops.max_in_list_size()
225  if max_in_list_size and len(params) > max_in_list_size:
226  # Break up the params list into an OR of manageable chunks.
227  in_clause_elements = ['(']
228  for offset in xrange(0, len(params), max_in_list_size):
229  if offset > 0:
230  in_clause_elements.append(' OR ')
231  in_clause_elements.append('%s IN (' % field_sql)
232  group_size = min(len(params) - offset, max_in_list_size)
233  param_group = ', '.join(repeat('%s', group_size))
234  in_clause_elements.append(param_group)
235  in_clause_elements.append(')')
236  in_clause_elements.append(')')
237  return ''.join(in_clause_elements), params
238  else:
239  return ('%s IN (%s)' % (field_sql,
240  ', '.join(repeat('%s', len(params)))),
241  params)
242  elif lookup_type in ('range', 'year'):
243  return ('%s BETWEEN %%s and %%s' % field_sql, params)
244  elif is_datetime_field and lookup_type in ('month', 'day', 'week_day',
245  'hour', 'minute', 'second'):
246  tzname = timezone.get_current_timezone_name() if settings.USE_TZ else None
247  sql, tz_params = connection.ops.datetime_extract_sql(lookup_type, field_sql, tzname)
248  return ('%s = %%s' % sql, tz_params + params)
249  elif lookup_type in ('month', 'day', 'week_day'):
250  return ('%s = %%s'
251  % connection.ops.date_extract_sql(lookup_type, field_sql), params)
252  elif lookup_type == 'isnull':
253  assert value_annotation in (True, False), "Invalid value_annotation for isnull"
254  return ('%s IS %sNULL' % (field_sql, ('' if value_annotation else 'NOT ')), ())
255  elif lookup_type == 'search':
256  return (connection.ops.fulltext_search_sql(field_sql), params)
257  elif lookup_type in ('regex', 'iregex'):
258  return connection.ops.regex_lookup(lookup_type) % (field_sql, cast_sql), params
259 
260  raise TypeError('Invalid lookup_type: %r' % lookup_type)

275  def relabel_aliases(self, change_map):
276  """
277  Relabels the alias values of any children. 'change_map' is a dictionary
278  mapping old (current) alias values to the new values.
279  """
280  for pos, child in enumerate(self.children):

293  """

296  value) tuples, or objects supporting .clone().

356  """
357  Returns a tuple of data suitable for inclusion in a WhereNode

359  """

362  try:
