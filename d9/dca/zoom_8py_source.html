64  "Converts a longitude, latitude coordinate pair for the given zoom level."

66  # number of pixels for the given zoom level.
67  lon, lat = self.get_lon_lat(lonlat)
68  npix = self._npix[zoom]

70  # Calculating the pixel x coordinate by multiplying the longitude value
71  # with with the number of degrees/pixel at the given zoom level.
72  px_x = round(npix + (lon * self._degpp[zoom]))

74  # Creating the factor, and ensuring that 1 or -1 is not passed in as the
75  # base to the logarithm. Here's why:
76  # if fac = -1, we'll get log(0) which is undefined;
77  # if fac = 1, our logarithm base will be divided by 0, also undefined.
78  fac = min(max(sin(DTOR * lat), -0.9999), 0.9999)
