199 class Atomic(object):
200  """
201  This class guarantees the atomic execution of a given block.
202 
203  An instance can be used either as a decorator or as a context manager.
204 
205  When it's used as a decorator, __call__ wraps the execution of the
206  decorated function in the instance itself, used as a context manager.
207 
208  When it's used as a context manager, __enter__ creates a transaction or a
209  savepoint, depending on whether a transaction is already in progress, and
210  __exit__ commits the transaction or releases the savepoint on normal exit,
211  and rolls back the transaction or to the savepoint on exceptions.
212 
213  It's possible to disable the creation of savepoints if the goal is to
214  ensure that some code runs within a transaction without creating overhead.
215 
216  A stack of savepoints identifiers is maintained as an attribute of the
217  connection. None denotes the absence of a savepoint.
218 
219  This allows reentrancy even if the same AtomicWrapper is reused. For
220  example, it's possible to define `oa = @atomic('other')` and use `@oa` or
221  `with oa:` multiple times.
222 
223  Since database connections are thread-local, this is thread-safe.
224  """

434 def _transaction_func(entering, exiting, using):
435  """
436  Takes 3 things, an entering function (what to do to start this block of
437  transaction management), an exiting function (what to do to end it, on both
438  success and failure, and using which can be: None, indiciating using is
439  DEFAULT_DB_ALIAS, a callable, indicating that using is DEFAULT_DB_ALIAS and
440  to return the function already wrapped.
441 
442  Returns either a Transaction objects, which is both a decorator and a
443  context manager, or a wrapped function, if using is a callable.
444  """
445  # Note that although the first argument is *called* `using`, it
446  # may actually be a function; @autocommit and @autocommit('foo')
447  # are both allowed forms.
448  if using is None:
449  using = DEFAULT_DB_ALIAS
450  if callable(using):
451  return Transaction(entering, exiting, DEFAULT_DB_ALIAS)(using)
452  return Transaction(entering, exiting, using)
