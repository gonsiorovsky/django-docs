40  def sql_create_model(self, model, style, known_models=set()):
41  """
42  Returns the SQL required to create a single model, as a tuple of:
43  (list_of_sql, pending_references_dict)
44  """
45  opts = model._meta
46  if not opts.managed or opts.proxy or opts.swapped:
47  return [], {}
48  final_output = []
49  table_output = []
50  pending_references = {}
51  qn = self.connection.ops.quote_name
52  for f in opts.local_fields:
53  col_type = f.db_type(connection=self.connection)
54  tablespace = f.db_tablespace or opts.db_tablespace
55  if col_type is None:
56  # Skip ManyToManyFields, because they're not represented as
57  # database columns in this table.
58  continue
59  # Make the definition (e.g. 'foo VARCHAR(30)') for this field.
60  field_output = [style.SQL_FIELD(qn(f.column)),
61  style.SQL_COLTYPE(col_type)]
62  # Oracle treats the empty string ('') as null, so coerce the null
63  # option whenever '' is a possible value.
64  null = f.null
65  if (f.empty_strings_allowed and not f.primary_key and
66  self.connection.features.interprets_empty_strings_as_nulls):
67  null = True
68  if not null:
69  field_output.append(style.SQL_KEYWORD('NOT NULL'))
70  if f.primary_key:
71  field_output.append(style.SQL_KEYWORD('PRIMARY KEY'))
72  elif f.unique:
73  field_output.append(style.SQL_KEYWORD('UNIQUE'))
74  if tablespace and f.unique:
75  # We must specify the index tablespace inline, because we
76  # won't be generating a CREATE INDEX statement for this field.
77  tablespace_sql = self.connection.ops.tablespace_sql(
78  tablespace, inline=True)
79  if tablespace_sql:
80  field_output.append(tablespace_sql)
81  if f.rel and f.db_constraint:
82  ref_output, pending = self.sql_for_inline_foreign_key_references(
83  model, f, known_models, style)
84  if pending:
85  pending_references.setdefault(f.rel.to, []).append(
86  (model, f))
87  else:
88  field_output.extend(ref_output)
89  table_output.append(' '.join(field_output))
90  for field_constraints in opts.unique_together:
91  table_output.append(style.SQL_KEYWORD('UNIQUE') + ' (%s)' %
92  ", ".join(
93  [style.SQL_FIELD(qn(opts.get_field(f).column))
94  for f in field_constraints]))
95 
96  full_statement = [style.SQL_KEYWORD('CREATE TABLE') + ' ' +
97  style.SQL_TABLE(qn(opts.db_table)) + ' (']
98  for i, line in enumerate(table_output): # Combine and add commas.
99  full_statement.append(
100  ' %s%s' % (line, ',' if i < len(table_output) - 1 else ''))
101  full_statement.append(')')
102  if opts.db_tablespace:
103  tablespace_sql = self.connection.ops.tablespace_sql(
104  opts.db_tablespace)
105  if tablespace_sql:
106  full_statement.append(tablespace_sql)
107  full_statement.append(';')
108  final_output.append('\n'.join(full_statement))
109 
110  if opts.has_auto_field:
111  # Add any extra SQL needed to support auto-incrementing primary
112  # keys.
113  auto_column = opts.auto_field.db_column or opts.auto_field.name
114  autoinc_sql = self.connection.ops.autoinc_sql(opts.db_table,
115  auto_column)
116  if autoinc_sql:
117  for stmt in autoinc_sql:
118  final_output.append(stmt)
119 
120  return final_output, pending_references

144  def sql_for_pending_references(self, model, style, pending_references):
145  """
146  Returns any ALTER TABLE statements to add constraints after the fact.
147  """
148  opts = model._meta
149  if not opts.managed or opts.swapped:
150  return []
151  qn = self.connection.ops.quote_name
152  final_output = []
153  if model in pending_references:
154  for rel_class, f in pending_references[model]:
155  rel_opts = rel_class._meta
156  r_table = rel_opts.db_table
157  r_col = f.column
158  table = opts.db_table
159  col = opts.get_field(f.rel.field_name).column
160  # For MySQL, r_name must be unique in the first 64 characters.
161  # So we are careful with character usage here.
162  r_name = '%s_refs_%s_%s' % (
163  r_col, col, self._digest(r_table, table))
164  final_output.append(style.SQL_KEYWORD('ALTER TABLE') +
165  ' %s ADD CONSTRAINT %s FOREIGN KEY (%s) REFERENCES %s (%s)%s;' %
166  (qn(r_table), qn(truncate_name(
167  r_name, self.connection.ops.max_name_length())),
168  qn(r_col), qn(table), qn(col),
169  self.connection.ops.deferrable_sql()))
170  del pending_references[model]
171  return final_output
