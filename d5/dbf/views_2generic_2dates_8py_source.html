687 def _get_next_prev(generic_view, date, is_previous, period):
688  """
689  Helper: Get the next or the previous valid date. The idea is to allow
690  links on month/day views to never be 404s by never providing a date
691  that'll be invalid for the given view.
692 
693  This is a bit complicated since it handles different intervals of time,
694  hence the coupling to generic_view.
695 
696  However in essence the logic comes down to:
697 
698  * If allow_empty and allow_future are both true, this is easy: just
699  return the naive result (just the next/previous day/week/month,
700  reguardless of object existence.)
701 
702  * If allow_empty is true, allow_future is false, and the naive result
703  isn't in the future, then return it; otherwise return None.
704 
705  * If allow_empty is false and allow_future is true, return the next
706  date *that contains a valid object*, even if it's in the future. If
707  there are no next objects, return None.
708 
709  * If allow_empty is false and allow_future is false, return the next
710  date that contains a valid object. If that date is in the future, or
711  if there are no next objects, return None.
712 
713  """
714  date_field = generic_view.get_date_field()
715  allow_empty = generic_view.get_allow_empty()
716  allow_future = generic_view.get_allow_future()
717 
718  get_current = getattr(generic_view, '_get_current_%s' % period)
719  get_next = getattr(generic_view, '_get_next_%s' % period)
720 
721  # Bounds of the current interval
722  start, end = get_current(date), get_next(date)
723 
724  # If allow_empty is True, the naive result will be valid

726  if is_previous:
