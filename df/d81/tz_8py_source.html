42 @register.filter('timezone')
43 def do_timezone(value, arg):
44  """
45  Converts a datetime to local time in a given time zone.
46 
47  The argument must be an instance of a tzinfo subclass or a time zone name.
48  If it is a time zone name, pytz is required.
49 
50  Naive datetimes are assumed to be in local time in the default time zone.
51  """
52  if not isinstance(value, datetime):
53  return ''
54 
55  # Obtain a timezone-aware datetime
56  try:
57  if timezone.is_naive(value):
58  default_timezone = timezone.get_default_timezone()
59  value = timezone.make_aware(value, default_timezone)
60  # Filters must never raise exceptions, and pytz' exceptions inherit
61  # Exception directly, not a specific subclass. So catch everything.
62  except Exception:
63  return ''
64 
65  # Obtain a tzinfo instance
66  if isinstance(arg, tzinfo):
67  tz = arg
68  elif isinstance(arg, six.string_types) and pytz is not None:
69  try:
70  tz = pytz.timezone(arg)
71  except pytz.UnknownTimeZoneError:
72  return ''
73  else:
74  return ''
75

79  # automatic conversion of the value to local time.

81  result.hour, result.minute, result.second,
