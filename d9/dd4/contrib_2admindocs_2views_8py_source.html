368 def simplify_regex(pattern):
369  """
370  Clean up urlpattern regexes into something somewhat readable by Mere Humans:
371  turns something like "^(?P<sport_slug>\w+)/athletes/(?P<athlete_slug>\w+)/$"
372  into "<sport_slug>/athletes/<athlete_slug>/"
373  """
374  # handle named groups first
375  pattern = named_group_matcher.sub(lambda m: m.group(1), pattern)
376 
377  # handle non-named groups
378  pattern = non_named_group_matcher.sub("<var>", pattern)
379 
380  # clean up any outstanding regex-y characters.
381  pattern = pattern.replace('^', '').replace('$', '').replace('?', '').replace('//', '/').replace('\\', '')
382  if not pattern.startswith('/'):
383  pattern = '/' + pattern
384  return pattern
