9 def ensure_default_manager(sender, **kwargs):
10  """
11  Ensures that a Model subclass contains a default manager and sets the
12  _default_manager attribute on the class. Also sets up the _base_manager
13  points to a plain Manager instance (which could be the same as
14  _default_manager if it's not a subclass of Manager).
15  """
16  cls = sender
17  if cls._meta.abstract:
18  setattr(cls, 'objects', AbstractManagerDescriptor(cls))
19  return
20  elif cls._meta.swapped:
21  setattr(cls, 'objects', SwappedManagerDescriptor(cls))
22  return
23  if not getattr(cls, '_default_manager', None):
24  # Create the default manager, if needed.
25  try:
26  cls._meta.get_field('objects')
27  raise ValueError("Model %s must specify a custom Manager, because it has a field named 'objects'" % cls.__name__)
28  except FieldDoesNotExist:
29  pass
30  cls.add_to_class('objects', Manager())
31  cls._base_manager = cls.objects
32  elif not getattr(cls, '_base_manager', None):
33  default_mgr = cls._default_manager.__class__
34  if (default_mgr is Manager or
35  getattr(default_mgr, "use_for_related_fields", False)):
36  cls._base_manager = cls._default_manager
37  else:
38  # Default manager isn't a plain Manager class, or a suitable
39  # replacement, so we walk up the base class hierarchy until we hit
40  # something appropriate.
41  for base_class in default_mgr.mro()[1:]:
42  if (base_class is Manager or
43  getattr(base_class, "use_for_related_fields", False)):
44  cls.add_to_class('_base_manager', base_class())
45  return
46  raise AssertionError("Should never get here. Please report a bug, including your model and model manager setup.")
