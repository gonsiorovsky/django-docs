139  """
140  The descriptor for the file attribute on the model instance. Returns a

155  raise AttributeError(
156  "The '%s' attribute can only be accessed from %s instances."
157  % (self.field.name, owner.__name__))
158 
159  # This is slightly complicated, so worth an explanation.
160  # instance.file`needs to ultimately return some instance of `File`,
161  # probably a subclass. Additionally, this returned object needs to have
162  # the FieldFile API so that users can easily do things like
163  # instance.file.path and have that delegated to the file storage engine.
164  # Easy enough if we're strict about assignment in __set__, but if you
165  # peek below you can see that we're not. So depending on the current
166  # value of the field we have to dynamically construct some sort of
167  # "thing" to return.
168 
169  # The instance dict contains whatever was originally assigned
170  # in __set__.
171  file = instance.__dict__[self.field.name]
172 
173  # If this value is a string (instance.file = "path/to/file") or None
174  # then we simply wrap it with the appropriate attribute class according
175  # to the file field. [This is FieldFile for FileFields and
176  # ImageFieldFile for ImageFields; it's also conceivable that user
177  # subclasses might also want to subclass the attribute class]. This
178  # object understands how to convert a path to a file, and also how to
179  # handle None.
180  if isinstance(file, six.string_types) or file is None:
181  attr = self.field.attr_class(instance, self.field, file)
182  instance.__dict__[self.field.name] = attr
183 
184  # Other types of files may be assigned as well, but they need to have
185  # the FieldFile interface added to the. Thus, we wrap any other type of
186  # File inside a FieldFile (well, the field's attr_class, which is
187  # usually FieldFile).
188  elif isinstance(file, File) and not isinstance(file, FieldFile):
189  file_copy = self.field.attr_class(instance, self.field, file.name)
190  file_copy.file = file
191  file_copy._committed = False
192  instance.__dict__[self.field.name] = file_copy
193 
194  # Finally, because of the (some would say boneheaded) way pickle works,
195  # the underlying FieldFile might not actually itself have an associated
196  # file. So we need to reset the details of the FieldFile in those cases.
197  elif isinstance(file, FieldFile) and not hasattr(file, 'field'):
198  file.instance = instance
199  file.field = self.field
200  file.storage = self.field.storage
201 
202  # That was fun, wasn't it?
203  return instance.__dict__[self.field.name]
