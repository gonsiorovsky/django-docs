201  def create_token(self, token_string, in_tag):
202  """
203  Convert the given token string into a new Token object and return it.
204  If in_tag is True, we are processing something that matched a tag,
205  otherwise it should be treated as a literal string.
206  """
207  if in_tag and token_string.startswith(BLOCK_TAG_START):
208  # The [2:-2] ranges below strip off *_TAG_START and *_TAG_END.
209  # We could do len(BLOCK_TAG_START) to be more "correct", but we've
210  # hard-coded the 2s here for performance. And it's not like
211  # the TAG_START values are going to change anytime, anyway.
212  block_content = token_string[2:-2].strip()
213  if self.verbatim and block_content == self.verbatim:
214  self.verbatim = False
215  if in_tag and not self.verbatim:

217  token = Token(TOKEN_VAR, token_string[2:-2].strip())
218  elif token_string.startswith(BLOCK_TAG_START):
219  if block_content[:9] in ('verbatim', 'verbatim '):

222  elif token_string.startswith(COMMENT_TAG_START):
223  content = ''

225  content = token_string[2:-2].strip()

902 def token_kwargs(bits, parser, support_legacy=False):
903  """
904  A utility method for parsing token keyword arguments.
905 
906  :param bits: A list containing remainder of the token (split by spaces)
907  that is to be checked for arguments. Valid arguments will be removed
908  from this list.
909 
910  :param support_legacy: If set to true ``True``, the legacy format
911  ``1 as foo`` will be accepted. Otherwise, only the standard ``foo=1``
912  format is allowed.
913 
914  :returns: A dictionary of the arguments retrieved from the ``bits`` token
915  list.
916 
917  There is no requirement for all remaining token ``bits`` to be keyword
918  arguments, so the dictionary will be returned as soon as an invalid
919  argument format is reached.
920  """
921  if not bits:
922  return {}
923  match = kwarg_re.match(bits[0])
924  kwarg_format = match and match.group(1)
925  if not kwarg_format:
926  if not support_legacy:
927  return {}
928  if len(bits) < 3 or bits[1] != 'as':
929  return {}
930 
931  kwargs = {}
932  while bits:
933  if kwarg_format:
934  match = kwarg_re.match(bits[0])
935  if not match or not match.group(1):
936  return kwargs
937  key, value = match.groups()
938  del bits[:1]
939  else:
940  if len(bits) < 3 or bits[1] != 'as':
941  return kwargs
942  key, value = bits[2], bits[0]
943  del bits[:3]
944  kwargs[key] = parser.compile_filter(value)
945  if bits and not kwarg_format:
946  if bits[0] != 'and':
947  return kwargs
948  del bits[:1]
949  return kwargs

951 def parse_bits(parser, bits, params, varargs, varkw, defaults,
952  takes_context, name):
953  """
954  Parses bits for template tag helpers (simple_tag, include_tag and
955  assignment_tag), in particular by detecting syntax errors and by
956  extracting positional and keyword arguments.
957  """
958  if takes_context:
959  if params[0] == 'context':
960  params = params[1:]
961  else:
962  raise TemplateSyntaxError(
963  "'%s' is decorated with takes_context=True so it must "
964  "have a first argument of 'context'" % name)
965  args = []
966  kwargs = {}
967  unhandled_params = list(params)
968  for bit in bits:
969  # First we try to extract a potential kwarg from the bit
970  kwarg = token_kwargs([bit], parser)
971  if kwarg:
972  # The kwarg was successfully extracted
973  param, value = list(six.iteritems(kwarg))[0]
974  if param not in params and varkw is None:
975  # An unexpected keyword argument was supplied
976  raise TemplateSyntaxError(
977  "'%s' received unexpected keyword argument '%s'" %
978  (name, param))
979  elif param in kwargs:
980  # The keyword argument has already been supplied once
981  raise TemplateSyntaxError(
982  "'%s' received multiple values for keyword argument '%s'" %
983  (name, param))
984  else:
985  # All good, record the keyword argument
986  kwargs[str(param)] = value
987  if param in unhandled_params:
988  # If using the keyword syntax for a positional arg, then
989  # consume it.
990  unhandled_params.remove(param)
991  else:
992  if kwargs:
993  raise TemplateSyntaxError(
994  "'%s' received some positional argument(s) after some "
995  "keyword argument(s)" % name)
996  else:
997  # Record the positional argument
998  args.append(parser.compile_filter(bit))
999  try:
1000  # Consume from the list of expected positional arguments
1001  unhandled_params.pop(0)
1002  except IndexError:
1003  if varargs is None:
1004  raise TemplateSyntaxError(
1005  "'%s' received too many positional arguments" %
1006  name)
1007  if defaults is not None:
1008  # Consider the last n params handled, where n is the
1009  # number of defaults.
1010  unhandled_params = unhandled_params[:-len(defaults)]
1011  if unhandled_params:
1012  # Some positional arguments were not supplied
1013  raise TemplateSyntaxError(
1014  "'%s' did not receive value(s) for the argument(s): %s" %
1015  (name, ", ".join(["'%s'" % p for p in unhandled_params])))
1016  return args, kwargs
