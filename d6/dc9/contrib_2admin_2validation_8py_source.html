270  """
271  Validate that list_filter is a sequence of one of three options:
272  1: 'field' - a basic field filter, possibly w/ relationships (eg, 'field__rel')
273  2: ('field', SomeFieldListFilter) - a field-based list filter class
274  3: SomeListFilter - a non-field list filter class
275  """

278  check_isseq(cls, 'list_filter', cls.list_filter)

283  raise ImproperlyConfigured("'%s.list_filter[%d]' is '%s'"

291  % (cls.__name__, idx, item.__name__))
292  else:
