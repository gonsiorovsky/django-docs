140  def convert_extent(self, clob):
141  if clob:
142  # Generally, Oracle returns a polygon for the extent -- however,
143  # it can return a single point if there's only one Point in the
144  # table.
145  ext_geom = Geometry(clob.read())
146  gtype = str(ext_geom.geom_type)
147  if gtype == 'Polygon':
148  # Construct the 4-tuple from the coordinates in the polygon.
149  shell = ext_geom.shell
150  ll, ur = shell[0][:2], shell[2][:2]
151  elif gtype == 'Point':
152  ll = ext_geom.coords[:2]
153  ur = ll
154  else:
155  raise Exception('Unexpected geometry type returned for extent: %s' % gtype)
156  xmin, ymin = ll
157  xmax, ymax = ur
158  return (xmin, ymin, xmax, ymax)
159  else:
160  return None

227  def spatial_lookup_sql(self, lvalue, lookup_type, value, field, qn):
228  "Returns the SQL WHERE clause for use in Oracle spatial SQL construction."
229  alias, col, db_type = lvalue
230 
231  # Getting the quoted table name as `geo_col`.
232  geo_col = '%s.%s' % (qn(alias), qn(col))
233 
234  # See if a Oracle Geometry function matches the lookup type next
235  lookup_info = self.geometry_functions.get(lookup_type, False)
236  if lookup_info:
237  # Lookup types that are tuples take tuple arguments, e.g., 'relate' and
238  # 'dwithin' lookup types.
239  if isinstance(lookup_info, tuple):
240  # First element of tuple is lookup type, second element is the type
241  # of the expected argument (e.g., str, float)
242  sdo_op, arg_type = lookup_info
243  geom = value[0]
244 
245  # Ensuring that a tuple _value_ was passed in from the user
246  if not isinstance(value, tuple):
247  raise ValueError('Tuple required for `%s` lookup type.' % lookup_type)
248  if len(value) != 2:
249  raise ValueError('2-element tuple required for %s lookup type.' % lookup_type)
250 
251  # Ensuring the argument type matches what we expect.
252  if not isinstance(value[1], arg_type):
253  raise ValueError('Argument type should be %s, got %s instead.' % (arg_type, type(value[1])))
254 
255  if lookup_type == 'relate':
256  # The SDORelate class handles construction for these queries,
257  # and verifies the mask argument.
258  return sdo_op(value[1]).as_sql(geo_col, self.get_geom_placeholder(field, geom))
259  else:
260  # Otherwise, just call the `as_sql` method on the SDOOperation instance.
261  return sdo_op.as_sql(geo_col, self.get_geom_placeholder(field, geom))
262  else:
263  # Lookup info is a SDOOperation instance, whose `as_sql` method returns
264  # the SQL necessary for the geometry function call. For example:
265  # SDO_CONTAINS("geoapp_country"."poly", SDO_GEOMTRY('POINT(5 23)', 4326)) = 'TRUE'
266  return lookup_info.as_sql(geo_col, self.get_geom_placeholder(field, value))
267  elif lookup_type == 'isnull':
268  # Handling 'isnull' lookup type
269  return "%s IS %sNULL" % (geo_col, ('' if value else 'NOT ')), []
270 
271  raise TypeError("Got invalid lookup_type: %s" % repr(lookup_type))
