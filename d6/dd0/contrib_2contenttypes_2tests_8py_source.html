60  def test_lookup_cache(self):

62  Make sure that the content type cache (see ContentTypeManager)
63  works correctly. Lookups for a particular content type -- by model, ID
64  or natural key -- should hit the database only on the first lookup.
65  """
66 
67  # At this point, a lookup for a ContentType should hit the DB
68  with self.assertNumQueries(1):
