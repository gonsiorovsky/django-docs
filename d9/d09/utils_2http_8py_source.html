112 def parse_http_date(date):
113  """
114  Parses a date format as specified by HTTP RFC2616 section 3.3.1.
115 
116  The three formats allowed by the RFC are accepted, even if only the first
117  one is still in widespread use.
118 
119  Returns an integer expressed in seconds since the epoch, in UTC.
120  """
121  # emails.Util.parsedate does the job for RFC1123 dates; unfortunately
122  # RFC2616 makes it mandatory to support RFC850 dates too. So we roll
123  # our own RFC-compliant parsing.
124  for regex in RFC1123_DATE, RFC850_DATE, ASCTIME_DATE:
125  m = regex.match(date)
126  if m is not None:
127  break
128  else:
129  raise ValueError("%r is not in a valid HTTP date format" % date)
130  try:
131  year = int(m.group('year'))
132  if year < 100:
133  if year < 70:
134  year += 2000
135  else:
136  year += 1900
137  month = MONTHS.index(m.group('mon').lower()) + 1
138  day = int(m.group('day'))
139  hour = int(m.group('hour'))
140  min = int(m.group('min'))
141  sec = int(m.group('sec'))
142  result = datetime.datetime(year, month, day, hour, min, sec)
143  return calendar.timegm(result.utctimetuple())
144  except Exception:
145  six.reraise(ValueError, ValueError("%r is not a valid date" % date), sys.exc_info()[2])

158 def base36_to_int(s):
159  """
160  Converts a base 36 string to an ``int``. Raises ``ValueError` if the
161  input won't fit into an int.
162  """
163  # To prevent overconsumption of server resources, reject any
164  # base36 string that is long than 13 base36 digits (13 digits
165  # is sufficient to base36-encode any 64-bit integer)
166  if len(s) > 13:
167  raise ValueError("Base36 input too large")
168  value = int(s, 36)
169  # ... then do a final check that the value will fit into an int to avoid
170  # returning a long (#15067). The long type was removed in Python 3.
171  if six.PY2 and value > sys.maxint:
172  raise ValueError("Base36 input too large")
173  return value

175 def int_to_base36(i):
176  """
177  Converts an integer to a base36 string
178  """
179  digits = "0123456789abcdefghijklmnopqrstuvwxyz"
180  factor = 0
181  if i < 0:
182  raise ValueError("Negative base36 conversion input.")
183  if six.PY2:
184  if not isinstance(i, six.integer_types):
185  raise TypeError("Non-integer base36 conversion input.")
186  if i > sys.maxint:
187  raise ValueError("Base36 conversion input too large.")
188  # Find starting factor
189  while True:
190  factor += 1
191  if i < 36 ** factor:
192  factor -= 1
193  break
194  base36 = []
195  # Construct base36 representation
196  while factor >= 0:
197  j = 36 ** factor
198  base36.append(digits[i // j])
199  i = i % j
200  factor -= 1
201  return ''.join(base36)

253  a different host and uses a safe scheme).
254

256  """
