274  def enter_transaction_management(self, managed=True, forced=False):
275  """
276  Enters transaction management for a running thread. It must be balanced with
277  the appropriate leave_transaction_management call, since the actual state is
278  managed as a stack.
279 
280  The state and dirty flag are carried over from the surrounding block or
281  from the settings, if there is no surrounding block (dirty is always false
282  when no current block is running).
283 
284  If you switch off transaction management and there is a pending
285  commit/rollback, the data will be commited, unless "forced" is True.
286  """
287  self.validate_no_atomic_block()
288 
289  self.transaction_state.append(managed)
290 
291  if not managed and self.is_dirty() and not forced:
292  self.commit()
293  self.set_clean()
294 
295  if managed == self.get_autocommit():
296  self.set_autocommit(not managed)
297 
298  def leave_transaction_management(self):
299  """
300  Leaves transaction management for a running thread. A dirty flag is carried
301  over to the surrounding block, as a commit will commit all changes, even
302  those from outside. (Commits are on connection level.)
303  """
304  self.validate_no_atomic_block()
305 
306  if self.transaction_state:
307  del self.transaction_state[-1]
308  else:
309  raise TransactionManagementError(
310  "This code isn't under transaction management")
311 
312  if self.transaction_state:
313  managed = self.transaction_state[-1]
314  else:
315  managed = not self.settings_dict['AUTOCOMMIT']
316 
317  if self._dirty:
318  self.rollback()
319  if managed == self.get_autocommit():
320  self.set_autocommit(not managed)
321  raise TransactionManagementError(
322  "Transaction managed block ended with pending COMMIT/ROLLBACK")
323 
324  if managed == self.get_autocommit():
325  self.set_autocommit(not managed)
