464  def spatial_lookup_sql(self, lvalue, lookup_type, value, field, qn):
465  """
466  Constructs spatial SQL from the given lookup value tuple a
467  (alias, col, db_type), the lookup type string, lookup value, and
468  the geometry field.
469  """
470  alias, col, db_type = lvalue
471 
472  # Getting the quoted geometry column.
473  geo_col = '%s.%s' % (qn(alias), qn(col))
474 
475  if lookup_type in self.geometry_operators:
476  if field.geography and not lookup_type in self.geography_operators:
477  raise ValueError('PostGIS geography does not support the '
478  '"%s" lookup.' % lookup_type)
479  # Handling a PostGIS operator.
480  op = self.geometry_operators[lookup_type]
481  return op.as_sql(geo_col, self.get_geom_placeholder(field, value))
482  elif lookup_type in self.geometry_functions:
483  if field.geography and not lookup_type in self.geography_functions:
484  raise ValueError('PostGIS geography type does not support the '
485  '"%s" lookup.' % lookup_type)
486 
487  # See if a PostGIS geometry function matches the lookup type.
488  tmp = self.geometry_functions[lookup_type]
489 
490  # Lookup types that are tuples take tuple arguments, e.g., 'relate' and
491  # distance lookups.
492  if isinstance(tmp, tuple):
493  # First element of tuple is the PostGISOperation instance, and the
494  # second element is either the type or a tuple of acceptable types
495  # that may passed in as further parameters for the lookup type.
496  op, arg_type = tmp
497 
498  # Ensuring that a tuple _value_ was passed in from the user
499  if not isinstance(value, (tuple, list)):
500  raise ValueError('Tuple required for `%s` lookup type.' % lookup_type)
501 
502  # Geometry is first element of lookup tuple.
503  geom = value[0]
504 
505  # Number of valid tuple parameters depends on the lookup type.
506  nparams = len(value)
507  if not self.num_params(lookup_type, nparams):
508  raise ValueError('Incorrect number of parameters given for `%s` lookup type.' % lookup_type)
509 
510  # Ensuring the argument type matches what we expect.
511  if not isinstance(value[1], arg_type):
512  raise ValueError('Argument type should be %s, got %s instead.' % (arg_type, type(value[1])))
513 
514  # For lookup type `relate`, the op instance is not yet created (has
515  # to be instantiated here to check the pattern parameter).
516  if lookup_type == 'relate':
517  op = op(self.geom_func_prefix, value[1])
518  elif lookup_type in self.distance_functions and lookup_type != 'dwithin':
519  if not field.geography and field.geodetic(self.connection):
520  # Geodetic distances are only available from Points to
521  # PointFields on PostGIS 1.4 and below.
522  if not self.connection.ops.geography:
523  if field.geom_type != 'POINT':
524  raise ValueError('PostGIS spherical operations are only valid on PointFields.')
525 
526  if str(geom.geom_type) != 'Point':
527  raise ValueError('PostGIS geometry distance parameter is required to be of type Point.')
528 
529  # Setting up the geodetic operation appropriately.
530  if nparams == 3 and value[2] == 'spheroid':
531  op = op['spheroid']
532  else:
533  op = op['sphere']
534  else:
535  op = op['cartesian']
536  else:
537  op = tmp
538  geom = value
539 
540  # Calling the `as_sql` function on the operation instance.
541  return op.as_sql(geo_col, self.get_geom_placeholder(field, geom))
542 
543  elif lookup_type == 'isnull':
544  # Handling 'isnull' lookup type
545  return "%s IS %sNULL" % (geo_col, ('' if value else 'NOT ')), []
546 
547  raise TypeError("Got invalid lookup_type: %s" % repr(lookup_type))
