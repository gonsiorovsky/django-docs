33 def construct_instance(form, instance, fields=None, exclude=None):
34  """
35  Constructs and returns a model instance from the bound ``form``'s
36  ``cleaned_data``, but does not save the returned instance to the
37  database.
38  """
39  from django.db import models
40  opts = instance._meta
41 
42  cleaned_data = form.cleaned_data
43  file_field_list = []
44  for f in opts.fields:
45  if not f.editable or isinstance(f, models.AutoField) \
46  or not f.name in cleaned_data:
47  continue
48  if fields is not None and f.name not in fields:
49  continue
50  if exclude and f.name in exclude:
51  continue
52  # Defer saving file-type fields until after the other fields, so a
53  # callable upload_to can use the values from other fields.
54  if isinstance(f, models.FileField):
55  file_field_list.append(f)
56  else:
57  f.save_form_data(instance, cleaned_data[f.name])
58 
59  for f in file_field_list:
60  f.save_form_data(instance, cleaned_data[f.name])
61 
62  return instance

110 def model_to_dict(instance, fields=None, exclude=None):
111  """
112  Returns a dict containing the data in ``instance`` suitable for passing as
113  a Form's ``initial`` keyword argument.
114 
115  ``fields`` is an optional list of field names. If provided, only the named
116  fields will be included in the returned dict.
117 
118  ``exclude`` is an optional list of field names. If provided, the named
119  fields will be excluded from the returned dict, even if they are listed in
120  the ``fields`` argument.
121  """
122  # avoid a circular import
123  from django.db.models.fields.related import ManyToManyField
124  opts = instance._meta
125  data = {}
126  for f in opts.concrete_fields + opts.virtual_fields + opts.many_to_many:
127  if not getattr(f, 'editable', False):
128  continue
129  if fields and not f.name in fields:
130  continue
131  if exclude and f.name in exclude:
132  continue
133  if isinstance(f, ManyToManyField):
134  # If the object doesn't have a primary key yet, just use an empty
135  # list for its m2m fields. Calling f.value_from_object will raise
136  # an exception.
137  if instance.pk is None:
138  data[f.name] = []
139  else:
140  # MultipleChoiceWidget needs a list of pks, not object instances.
141  data[f.name] = list(f.value_from_object(instance).values_list('pk', flat=True))
142  else:
143  data[f.name] = f.value_from_object(instance)
144  return data

184  continue

195  if help_texts and f.name in help_texts:
196  kwargs['help_text'] = help_texts[f.name]
197  if error_messages and f.name in error_messages:
198  kwargs['error_messages'] = error_messages[f.name]

213  field_dict = SortedDict(

347  def _get_validation_exclusions(self):
348  """
349  For backwards-compatibility, several types of fields need to be
350  excluded from model validation. See the following tickets for
351  details: #12507, #12521, #12553
352  """
353  exclude = []
354  # Build up a list of fields that should be excluded from model field
355  # validation and unique checks.
356  for f in self.instance._meta.fields:
357  field = f.name
358  # Exclude fields that aren't on the form. The developer may be
359  # adding these values to the model after form validation.
360  if field not in self.fields:
361  exclude.append(f.name)
362 
363  # Don't perform model validation on fields that were defined
364  # manually on the form and excluded via the ModelForm's Meta
365  # class. See #12901.
366  elif self._meta.fields and field not in self._meta.fields:
367  exclude.append(f.name)
368  elif self._meta.exclude and field in self._meta.exclude:
369  exclude.append(f.name)
370 
371  # Exclude fields that failed form validation. There's no need for
372  # the model fields to validate them as well.
373  elif field in self._errors.keys():
374  exclude.append(f.name)
375 
376  # Exclude empty fields that are not required by the form, if the
377  # underlying model field is required. This keeps the model field
378  # from raising a required error. Note: don't exclude the field from
379  # validation if the model field allows blanks. If it does, the blank
380  # value may be included in a unique check, so cannot be excluded
381  # from validation.
382  else:
383  form_field = self.fields[field]
384  field_value = self.cleaned_data.get(field, None)
385  if not f.blank and not form_field.required and field_value in form_field.empty_values:
386  exclude.append(f.name)
387  return exclude

630  valid_forms = [form for form in self.forms if form.is_valid() and form not in forms_to_delete]
631  for form in valid_forms:
632  exclude = form._get_validation_exclusions()
633  unique_checks, date_checks = form.instance._get_unique_checks(exclude=exclude)
634  all_unique_checks = all_unique_checks.union(set(unique_checks))
635  all_date_checks = all_date_checks.union(set(date_checks))
636 
637  errors = []
638  # Do each of the unique checks (unique and unique_together)
639  for uclass, unique_check in all_unique_checks:
640  seen_data = set()
641  for form in valid_forms:
642  # get data for each field of each of unique_check
643  row_data = tuple([form.cleaned_data[field] for field in unique_check if field in form.cleaned_data])
644  if row_data and not None in row_data:
645  # if we've already seen it then we have a uniqueness failure
646  if row_data in seen_data:
647  # poke error messages into the right places and mark
648  # the form as invalid
649  errors.append(self.get_unique_error_message(unique_check))
650  form._errors[NON_FIELD_ERRORS] = self.error_class([self.get_form_error()])
651  # remove the data from the cleaned_data dict since it was invalid
652  for field in unique_check:
653  if field in form.cleaned_data:
654  del form.cleaned_data[field]
655  # mark the data as seen
656  seen_data.add(row_data)
657  # iterate over each of the date checks now
658  for date_check in all_date_checks:
659  seen_data = set()
660  uclass, lookup, field, unique_for = date_check
661  for form in valid_forms:
662  # see if we have data for both fields
663  if (form.cleaned_data and form.cleaned_data[field] is not None
664  and form.cleaned_data[unique_for] is not None):
665  # if it's a date lookup we need to get the data for all the fields
666  if lookup == 'date':
667  date = form.cleaned_data[unique_for]
668  date_data = (date.year, date.month, date.day)
669  # otherwise it's just the attribute on the date/datetime
670  # object
671  else:
672  date_data = (getattr(form.cleaned_data[unique_for], lookup),)
673  data = (form.cleaned_data[field],) + date_data
674  # if we've already seen it then we have a uniqueness failure
675  if data in seen_data:
676  # poke error messages into the right places and mark
677  # the form as invalid
678  errors.append(self.get_date_error_message(date_check))
679  form._errors[NON_FIELD_ERRORS] = self.error_class([self.get_form_error()])
680  # remove the data from the cleaned_data dict since it was invalid
681  del form.cleaned_data[field]
682  # mark the data as seen
683  seen_data.add(data)
684  if errors:
685  raise ValidationError(errors)

912 def _get_foreign_key(parent_model, model, fk_name=None, can_fail=False):
913  """
914  Finds and returns the ForeignKey from model to parent if there is one
915  (returns None if can_fail is True and no such field exists). If fk_name is
916  provided, assume it is the name of the ForeignKey field. Unless can_fail is
917  True, an exception is raised if there is no ForeignKey from model to
918  parent_model.
919  """
920  # avoid circular import
921  from django.db.models import ForeignKey
922  opts = model._meta
923  if fk_name:
924  fks_to_parent = [f for f in opts.fields if f.name == fk_name]
925  if len(fks_to_parent) == 1:
926  fk = fks_to_parent[0]
927  if not isinstance(fk, ForeignKey) or \
928  (fk.rel.to != parent_model and
929  fk.rel.to not in parent_model._meta.get_parent_list()):
930  raise Exception("fk_name '%s' is not a ForeignKey to %s" % (fk_name, parent_model))
931  elif len(fks_to_parent) == 0:
932  raise Exception("%s has no field named '%s'" % (model, fk_name))
933  else:
934  # Try to discover what the ForeignKey from model to parent_model is
935  fks_to_parent = [
936  f for f in opts.fields
937  if isinstance(f, ForeignKey)
938  and (f.rel.to == parent_model
939  or f.rel.to in parent_model._meta.get_parent_list())
940  ]
941  if len(fks_to_parent) == 1:
942  fk = fks_to_parent[0]
943  elif len(fks_to_parent) == 0:
944  if can_fail:
945  return
946  raise Exception("%s has no ForeignKey to %s" % (model, parent_model))
947  else:
948  raise Exception("%s has more than 1 ForeignKey to %s" % (model, parent_model))
949  return fk
