336 def strip_quotes(want, got):
337  """
338  Strip quotes of doctests output values:
339 
340  >>> strip_quotes("'foo'")
341  "foo"
342  >>> strip_quotes('"foo"')
343  "foo"
344  """

346  s = s.strip()
347  return (len(s) >= 2
348  and s[0] == s[-1]
