103  start_alias=None, opts=None, as_pairs=False, from_parent=None):
104  """
105  Computes the default columns for selecting every field in the base
106  model. Will sometimes be called to pull in related models (e.g. via
107  select_related), in which case "opts" and "start_alias" will be given
108  to provide a starting point for the traversal.
109 
110  Returns a list of strings, quoted appropriately for use in SQL
111  directly, as well as a set of aliases used in the select statement (if
112  'as_pairs' is True, returns a list of (alias, col_name) pairs instead
113  of strings as the first component and None as the second component).
114 
115  This routine is overridden from Query to handle customized selection of
116  geometry columns.
117  """
118  result = []
119  if opts is None:
120  opts = self.query.get_meta()
121  aliases = set()
122  only_load = self.deferred_to_columns()
123  seen = self.query.included_inherited_models.copy()
124  if start_alias:
125  seen[None] = start_alias
126  for field, model in opts.get_concrete_fields_with_model():
127  if from_parent and model is not None and issubclass(from_parent, model):
128  # Avoid loading data for already loaded parents.
129  continue
130  alias = self.query.join_parent_model(opts, model, start_alias, seen)
131  table = self.query.alias_map[alias].table_name
132  if table in only_load and field.column not in only_load[table]:
133  continue
134  if as_pairs:
135  result.append((alias, field))
136  aliases.add(alias)
137  continue

139  # see if there was any custom selection specified in the
140  # dictionary, and set up the selection format appropriately.
141  field_sel = self.get_field_select(field, alias)
142  if with_aliases and field.column in col_aliases:
143  c_alias = 'Col%d' % len(col_aliases)
144  result.append('%s AS %s' % (field_sel, c_alias))
145  col_aliases.add(c_alias)
146  aliases.add(c_alias)
147  else:
148  r = field_sel
149  result.append(r)

155  def resolve_columns(self, row, fields=()):
156  """
157  This routine is necessary so that distances and geometries returned
158  from extra selection SQL get resolved appropriately into Python
159  objects.
160  """
161  values = []
162  aliases = list(self.query.extra_select)
163 
164  # Have to set a starting row number offset that is used for
165  # determining the correct starting row index -- needed for
166  # doing pagination with Oracle.
167  rn_offset = 0
168  if self.connection.ops.oracle:
169  if self.query.high_mark is not None or self.query.low_mark: rn_offset = 1
170  index_start = rn_offset + len(aliases)
171 
172  # Converting any extra selection values (e.g., geometries and
173  # distance objects added by GeoQuerySet methods).
174  values = [self.query.convert_values(v,
175  self.query.extra_select_fields.get(a, None),
176  self.connection)
177  for v, a in zip(row[rn_offset:index_start], aliases)]
178  if self.connection.ops.oracle or getattr(self.query, 'geo_values', False):
179  # We resolve the rest of the columns if we're on Oracle or if
180  # the `geo_values` attribute is defined.
181  for value, field in zip_longest(row[index_start:], fields):
182  values.append(self.query.convert_values(value, field, self.connection))
183  else:
184  values.extend(row[index_start:])
185  return tuple(values)

194  def get_field_select(self, field, alias=None, column=None):
195  """
196  Returns the SELECT SQL string for the given field. Figures out
197  if any custom selection SQL is needed for the column The `alias`
198  keyword may be used to manually specify the database table where
199  the column exists, if not in the model associated with this
200  `GeoQuery`. Similarly, `column` may be used to specify the exact
201  column name, rather than using the `column` attribute on `field`.
202  """
203  sel_fmt = self.get_select_format(field)
204  if field in self.query.custom_select:
205  field_sel = sel_fmt % self.query.custom_select[field]
206  else:

210  def get_select_format(self, fld):
211  """
212  Returns the selection format string, depending on the requirements
213  of the spatial backend. For example, Oracle and MySQL require custom
214  selection formats in order to retrieve geometries in OGC WKT. For all
215  other fields a simple '%s' format string is returned.
216  """
217  if self.connection.ops.select and hasattr(fld, 'geom_type'):
218  # This allows operations to be done on fields in the SELECT,
219  # overriding their values -- used by the Oracle and MySQL
220  # spatial backends to get database values as WKT, and by the
221  # `transform` method.
222  sel_fmt = self.connection.ops.select
223 
224  # Because WKT doesn't contain spatial reference information,
225  # the SRID is prefixed to the returned WKT to ensure that the
226  # transformed geometries have an SRID different than that of the
227  # field -- this is only used by `transform` for Oracle and
228  # SpatiaLite backends.
229  if self.query.transformed_srid and ( self.connection.ops.oracle or
230  self.connection.ops.spatialite ):
231  sel_fmt = "'SRID=%d;'||%s" % (self.query.transformed_srid, sel_fmt)
232  else:
233  sel_fmt = '%s'
234  return sel_fmt
