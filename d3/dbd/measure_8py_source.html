174  def default_units(self, kwargs):
175  """
176  Return the unit value and the default units specified
177  from the given keyword arguments dictionary.
178  """
179  val = 0.0
180  default_unit = self.STANDARD_UNIT
181  for unit, value in six.iteritems(kwargs):
182  if not isinstance(value, float): value = float(value)
183  if unit in self.UNITS:
184  val += self.UNITS[unit] * value
185  default_unit = unit
186  elif unit in self.ALIAS:
187  u = self.ALIAS[unit]
188  val += self.UNITS[u] * value
189  default_unit = u
190  else:
191  lower = unit.lower()
192  if lower in self.UNITS:
193  val += self.UNITS[lower] * value
194  default_unit = lower
195  elif lower in self.LALIAS:
196  u = self.LALIAS[lower]
197  val += self.UNITS[u] * value
198  default_unit = u
199  else:
200  raise AttributeError('Unknown unit type: %s' % unit)
201  return val, default_unit

204  def unit_attname(cls, unit_str):
205  """
206  Retrieves the unit attribute name for the given unit string.
207  For example, if the given unit string is 'metre', 'm' would be returned.
208  An exception is raised if an attribute cannot be found.
209  """
210  lower = unit_str.lower()
211  if unit_str in cls.UNITS:
212  return unit_str
213  elif lower in cls.UNITS:
214  return lower
215  elif lower in cls.LALIAS:
216  return cls.LALIAS[lower]
217  else:
218  raise Exception('Could not find a unit keyword associated with "%s"' % unit_str)
