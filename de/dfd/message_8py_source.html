40 def make_msgid(idstring=None):
41  """Returns a string suitable for RFC 2822 compliant Message-ID, e.g:
42 
43  <20020201195627.33539.96671@nightshade.la.mastaler.com>
44 
45  Optional idstring if given is a string used to strengthen the
46  uniqueness of the message id.
47  """
48  timeval = time.time()
49  utcdate = time.strftime('%Y%m%d%H%M%S', time.gmtime(timeval))
50  try:
51  pid = os.getpid()
52  except AttributeError:
53  # No getpid() in Jython, for example.
54  pid = 1
55  randint = random.randrange(100000)
56  if idstring is None:
57  idstring = ''
58  else:
59  idstring = '.' + idstring

61  msgid = '<%s.%s.%s%s@%s>' % (utcdate, pid, randint, idstring, idhost)

81 def forbid_multi_line_headers(name, val, encoding):
82  """Forbids multi-line headers, to prevent header injection."""
83  encoding = encoding or settings.DEFAULT_CHARSET
84  val = force_text(val)

101 def sanitize_address(addr, encoding):
102  if isinstance(addr, six.string_types):
103  addr = parseaddr(force_text(addr))
104  nm, addr = addr
105  # This try-except clause is needed on Python 3 < 3.2.4
106  # http://bugs.python.org/issue14291
107  try:
108  nm = Header(nm, encoding).encode()
109  except UnicodeEncodeError:
110  nm = Header(nm, 'utf-8').encode()
111  try:
112  addr.encode('ascii')
113  except UnicodeEncodeError: # IDN
114  if '@' in addr:
115  localpart, domain = addr.split('@', 1)
116  localpart = str(Header(localpart, encoding))
117  domain = domain.encode('idna').decode('ascii')
118  addr = '@'.join([localpart, domain])
119  else:
120  addr = Header(addr, encoding).encode()
121  return formataddr((nm, addr))
