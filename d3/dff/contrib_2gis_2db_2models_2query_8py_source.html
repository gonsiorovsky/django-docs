497  def _spatial_attribute(self, att, settings, field_name=None, model_att=None):
498  """
499  DRY routine for calling a spatial stored procedure on a geometry column
500  and attaching its output as an attribute of the model.
501 
502  Arguments:
503  att:
504  The name of the spatial attribute that holds the spatial
505  SQL function to call.
506 
507  settings:
508  Dictonary of internal settings to customize for the spatial procedure.
509 
510  Public Keyword Arguments:
511 
512  field_name:
513  The name of the geographic field to call the spatial
514  function on. May also be a lookup to a geometry field
515  as part of a foreign key relation.
516 
517  model_att:
518  The name of the model attribute to attach the output of
519  the spatial function to.
520  """
521  # Default settings.
522  settings.setdefault('desc', None)
523  settings.setdefault('geom_args', ())
524  settings.setdefault('geom_field', None)
525  settings.setdefault('procedure_args', {})
526  settings.setdefault('procedure_fmt', '%(geo_col)s')
527  settings.setdefault('select_params', [])
528 
529  connection = connections[self.db]
530  backend = connection.ops
531 
532  # Performing setup for the spatial column, unless told not to.
533  if settings.get('setup', True):
534  default_args, geo_field = self._spatial_setup(att, desc=settings['desc'], field_name=field_name,
535  geo_field_type=settings.get('geo_field_type', None))
536  for k, v in six.iteritems(default_args): settings['procedure_args'].setdefault(k, v)
537  else:
538  geo_field = settings['geo_field']
539 
540  # The attribute to attach to the model.
541  if not isinstance(model_att, six.string_types): model_att = att
542 
543  # Special handling for any argument that is a geometry.
544  for name in settings['geom_args']:
545  # Using the field's get_placeholder() routine to get any needed
546  # transformation SQL.
547  geom = geo_field.get_prep_value(settings['procedure_args'][name])
548  params = geo_field.get_db_prep_lookup('contains', geom, connection=connection)
549  geom_placeholder = geo_field.get_placeholder(geom, connection)
550 
551  # Replacing the procedure format with that of any needed
552  # transformation SQL.
553  old_fmt = '%%(%s)s' % name
554  new_fmt = geom_placeholder % '%%s'
555  settings['procedure_fmt'] = settings['procedure_fmt'].replace(old_fmt, new_fmt)
556  settings['select_params'].extend(params)
557 
558  # Getting the format for the stored procedure.
559  fmt = '%%(function)s(%s)' % settings['procedure_fmt']
560 
561  # If the result of this function needs to be converted.
562  if settings.get('select_field', False):
563  sel_fld = settings['select_field']
564  if isinstance(sel_fld, GeomField) and backend.select:
565  self.query.custom_select[model_att] = backend.select
566  if connection.ops.oracle:
567  sel_fld.empty_strings_allowed = False
568  self.query.extra_select_fields[model_att] = sel_fld
569 
570  # Finally, setting the extra selection attribute with
571  # the format string expanded with the stored procedure
572  # arguments.
573  return self.extra(select={model_att : fmt % settings['procedure_args']},
574  select_params=settings['select_params'])

691  elif length or perimeter:
