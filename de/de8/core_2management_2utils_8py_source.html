33 def handle_extensions(extensions=('html',), ignored=('py',)):
34  """
35  Organizes multiple extensions that are separated with commas or passed by
36  using --extension/-e multiple times. Note that the .py extension is ignored
37  here because of the way non-*.py files are handled in make_messages() (they
38  are copied to file.ext.py files to trick xgettext to parse them as Python
39  files).
40 
41  For example: running 'django-admin makemessages -e js,txt -e xhtml -a'
42  would result in an extension list: ['.js', '.txt', '.xhtml']
43 
44  >>> handle_extensions(['.html', 'html,js,py,py,py,.py', 'py,.py'])
45  set(['.html', '.js'])
46  >>> handle_extensions(['.html, txt,.tpl'])
47  set(['.html', '.tpl', '.txt'])
48  """
49  ext_list = []
50  for ext in extensions:
51  ext_list.extend(ext.replace(' ', '').split(','))
52  for i, ext in enumerate(ext_list):
53  if not ext.startswith('.'):
54  ext_list[i] = '.%s' % ext_list[i]
55  return set([x for x in ext_list if x.strip('.') not in ignored])
