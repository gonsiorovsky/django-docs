215  """

312  def spatial_lookup_sql(self, lvalue, lookup_type, value, field, qn):
313  """
314  Returns the SpatiaLite-specific SQL for the given lookup value
315  [a tuple of (alias, column, db_type)], lookup type, lookup
316  value, the model field, and the quoting function.
317  """
318  alias, col, db_type = lvalue
319 
320  # Getting the quoted field as `geo_col`.
321  geo_col = '%s.%s' % (qn(alias), qn(col))
322 
323  if lookup_type in self.geometry_functions:
324  # See if a SpatiaLite geometry function matches the lookup type.
325  tmp = self.geometry_functions[lookup_type]
326 
327  # Lookup types that are tuples take tuple arguments, e.g., 'relate' and
328  # distance lookups.
329  if isinstance(tmp, tuple):
330  # First element of tuple is the SpatiaLiteOperation instance, and the
331  # second element is either the type or a tuple of acceptable types
332  # that may passed in as further parameters for the lookup type.
333  op, arg_type = tmp
334 
335  # Ensuring that a tuple _value_ was passed in from the user
336  if not isinstance(value, (tuple, list)):
337  raise ValueError('Tuple required for `%s` lookup type.' % lookup_type)
338 
339  # Geometry is first element of lookup tuple.
340  geom = value[0]
341 
342  # Number of valid tuple parameters depends on the lookup type.
343  if len(value) != 2:
344  raise ValueError('Incorrect number of parameters given for `%s` lookup type.' % lookup_type)
345 
346  # Ensuring the argument type matches what we expect.
347  if not isinstance(value[1], arg_type):
348  raise ValueError('Argument type should be %s, got %s instead.' % (arg_type, type(value[1])))
349 
350  # For lookup type `relate`, the op instance is not yet created (has
351  # to be instantiated here to check the pattern parameter).
352  if lookup_type == 'relate':
353  op = op(value[1])
354  elif lookup_type in self.distance_functions:
355  op = op[0]
356  else:
357  op = tmp
358  geom = value
359  # Calling the `as_sql` function on the operation instance.
360  return op.as_sql(geo_col, self.get_geom_placeholder(field, geom))
361  elif lookup_type == 'isnull':
362  # Handling 'isnull' lookup type
363  return "%s IS %sNULL" % (geo_col, ('' if value else 'NOT ')), []
364 
365  raise TypeError("Got invalid lookup_type: %s" % repr(lookup_type))
