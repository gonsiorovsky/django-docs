12  def georss_coords(self, coords):
13  """
14  In GeoRSS coordinate pairs are ordered by lat/lon and separated by
15  a single white space. Given a tuple of coordinates, this will return
16  a unicode GeoRSS representation.
17  """
18  return ' '.join(['%f %f' % (coord[1], coord[0]) for coord in coords])

20  def add_georss_point(self, handler, coords, w3c_geo=False):
21  """
22  Adds a GeoRSS point with the given coords using the given handler.
23  Handles the differences between simple GeoRSS and the more pouplar
24  W3C Geo specification.
25  """

27  lon, lat = coords[:2]
28  handler.addQuickElement('geo:lat', '%f' % lat)
29  handler.addQuickElement('geo:lon', '%f' % lon)
30  else:
31  handler.addQuickElement('georss:point', self.georss_coords((coords,)))

33  def add_georss_element(self, handler, item, w3c_geo=False):
34  """
35  This routine adds a GeoRSS XML element using the given item and handler.
36  """
37  # Getting the Geometry object.
38  geom = item.get('geometry', None)
39  if not geom is None:
40  if isinstance(geom, (list, tuple)):
41  # Special case if a tuple/list was passed in. The tuple may be
42  # a point or a box
43  box_coords = None
44  if isinstance(geom[0], (list, tuple)):
45  # Box: ( (X0, Y0), (X1, Y1) )
46  if len(geom) == 2:
47  box_coords = geom
48  else:
49  raise ValueError('Only should be two sets of coordinates.')
50  else:
51  if len(geom) == 2:
52  # Point: (X, Y)
53  self.add_georss_point(handler, geom, w3c_geo=w3c_geo)
54  elif len(geom) == 4:
55  # Box: (X0, Y0, X1, Y1)
56  box_coords = (geom[:2], geom[2:])
57  else:
58  raise ValueError('Only should be 2 or 4 numeric elements.')
59  # If a GeoRSS box was given via tuple.
60  if not box_coords is None:
61  if w3c_geo: raise ValueError('Cannot use simple GeoRSS box in W3C Geo feeds.')
62  handler.addQuickElement('georss:box', self.georss_coords(box_coords))
63  else:
64  # Getting the lower-case geometry type.
65  gtype = str(geom.geom_type).lower()
66  if gtype == 'point':
67  self.add_georss_point(handler, geom.coords, w3c_geo=w3c_geo)
68  else:
69  if w3c_geo: raise ValueError('W3C Geo only supports Point geometries.')
70  # For formatting consistent w/the GeoRSS simple standard:
71  # http://georss.org/1.0#simple
72  if gtype in ('linestring', 'linearring'):
73  handler.addQuickElement('georss:line', self.georss_coords(geom.coords))
74  elif gtype in ('polygon',):
75  # Only support the exterior ring.
76  handler.addQuickElement('georss:polygon', self.georss_coords(geom[0].coords))
77  else:
78  raise ValueError('Geometry type "%s" not supported.' % geom.geom_type)
