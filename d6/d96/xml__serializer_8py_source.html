171  def _handle_object(self, node):
172  """
173  Convert an <object> node to a DeserializedObject.
174  """
175  # Look up the model using the model loading mechanism. If this fails,
176  # bail.
177  Model = self._get_model_from_node(node, "model")
178 
179  # Start building a data dictionary from the object.
180  # If the node is missing the pk set it to None
181  if node.hasAttribute("pk"):
182  pk = node.getAttribute("pk")
183  else:
184  pk = None
185 
186  data = {Model._meta.pk.attname : Model._meta.pk.to_python(pk)}
187 
188  # Also start building a dict of m2m data (this is saved as
189  # {m2m_accessor_attribute : [list_of_related_objects]})
190  m2m_data = {}
191 
192  model_fields = Model._meta.get_all_field_names()
193  # Deseralize each field.
194  for field_node in node.getElementsByTagName("field"):
195  # If the field is missing the name attribute, bail (are you
196  # sensing a pattern here?)
197  field_name = field_node.getAttribute("name")
198  if not field_name:
199  raise base.DeserializationError("<field> node is missing the 'name' attribute")
200 
201  # Get the field from the Model. This will raise a
202  # FieldDoesNotExist if, well, the field doesn't exist, which will
203  # be propagated correctly unless ignorenonexistent=True is used.
204  if self.ignore and field_name not in model_fields:
205  continue
206  field = Model._meta.get_field(field_name)
207 
208  # As is usually the case, relation fields get the special treatment.
209  if field.rel and isinstance(field.rel, models.ManyToManyRel):
210  m2m_data[field.name] = self._handle_m2m_field_node(field_node, field)
211  elif field.rel and isinstance(field.rel, models.ManyToOneRel):
212  data[field.attname] = self._handle_fk_field_node(field_node, field)
213  else:
214  if field_node.getElementsByTagName('None'):
215  value = None
216  else:
217  value = field.to_python(getInnerText(field_node).strip())
218  data[field.name] = value
219 
220  # Return a DeserializedObject so that the m2m data has a place to live.
221  return base.DeserializedObject(Model(**data), m2m_data)

223  def _handle_fk_field_node(self, node, field):
224  """
225  Handle a <field> node for a ForeignKey
226  """
227  # Check if there is a child node named 'None', returning None if so.
228  if node.getElementsByTagName('None'):
229  return None
230  else:
231  if hasattr(field.rel.to._default_manager, 'get_by_natural_key'):
232  keys = node.getElementsByTagName('natural')
233  if keys:
234  # If there are 'natural' subelements, it must be a natural key
235  field_value = [getInnerText(k).strip() for k in keys]
236  obj = field.rel.to._default_manager.db_manager(self.db).get_by_natural_key(*field_value)
237  obj_pk = getattr(obj, field.rel.field_name)
238  # If this is a natural foreign key to an object that
239  # has a FK/O2O as the foreign key, use the FK value
240  if field.rel.to._meta.pk.rel:
241  obj_pk = obj_pk.pk
242  else:
243  # Otherwise, treat like a normal PK
244  field_value = getInnerText(node).strip()
245  obj_pk = field.rel.to._meta.get_field(field.rel.field_name).to_python(field_value)
246  return obj_pk
247  else:
248  field_value = getInnerText(node).strip()
249  return field.rel.to._meta.get_field(field.rel.field_name).to_python(field_value)

251  def _handle_m2m_field_node(self, node, field):
252  """
253  Handle a <field> node for a ManyToManyField.
254  """
255  if hasattr(field.rel.to._default_manager, 'get_by_natural_key'):
256  def m2m_convert(n):
257  keys = n.getElementsByTagName('natural')
258  if keys:
259  # If there are 'natural' subelements, it must be a natural key
260  field_value = [getInnerText(k).strip() for k in keys]
261  obj_pk = field.rel.to._default_manager.db_manager(self.db).get_by_natural_key(*field_value).pk
262  else:
263  # Otherwise, treat like a normal PK value.
264  obj_pk = field.rel.to._meta.pk.to_python(n.getAttribute('pk'))
265  return obj_pk
266  else:
267  m2m_convert = lambda n: field.rel.to._meta.pk.to_python(n.getAttribute('pk'))
268  return [m2m_convert(c) for c in node.getElementsByTagName("object")]

270  def _get_model_from_node(self, node, attr):

291 def getInnerText(node):
292  """
293  Get all the inner text of a DOM node (recursively).
294  """
295  # inspired by http://mail.python.org/pipermail/xml-sig/2005-March/011022.html
296  inner_text = []
297  for child in node.childNodes:
298  if child.nodeType == child.TEXT_NODE or child.nodeType == child.CDATA_SECTION_NODE:
299  inner_text.append(child.data)
300  elif child.nodeType == child.ELEMENT_NODE:
301  inner_text.extend(getInnerText(child))
302  else:
303  pass
304  return "".join(inner_text)
