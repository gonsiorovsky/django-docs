14 def get_srid_info(srid, connection):
15  """
16  Returns the units, unit name, and spheroid WKT associated with the
17  given SRID from the `spatial_ref_sys` (or equivalent) spatial database
18  table for the given database connection. These results are cached.
19  """
20  global _srid_cache
21 
22  try:
23  # The SpatialRefSys model for the spatial backend.
24  SpatialRefSys = connection.ops.spatial_ref_sys()
25  except NotImplementedError:
26  # No `spatial_ref_sys` table in spatial backend (e.g., MySQL).
27  return None, None, None
28 
29  if not connection.alias in _srid_cache:
30  # Initialize SRID dictionary for database if it doesn't exist.
31  _srid_cache[connection.alias] = {}
32 
33  if not srid in _srid_cache[connection.alias]:
34  # Use `SpatialRefSys` model to query for spatial reference info.
35  sr = SpatialRefSys.objects.using(connection.alias).get(srid=srid)
36  units, units_name = sr.units
37  spheroid = SpatialRefSys.get_spheroid(sr.wkt)
38  _srid_cache[connection.alias][srid] = (units, units_name, spheroid)
39 
40  return _srid_cache[connection.alias][srid]

143  def get_prep_value(self, value):
144  """
145  Spatial lookup values are either a parameter that is (or may be
146  converted to) a geometry, or a sequence of lookup values that
147  begins with a geometry. This routine will setup the geometry
148  value properly, and preserve any other lookup parameters before
149  returning to the caller.
150  """
151  if isinstance(value, SQLEvaluator):
152  return value
153  elif isinstance(value, (tuple, list)):
154  geom = value[0]
155  seq_value = True
156  else:
157  geom = value
158  seq_value = False
159 
160  # When the input is not a GEOS geometry, attempt to construct one
161  # from the given string input.
162  if isinstance(geom, Geometry):
163  pass
164  elif isinstance(geom, (bytes, six.string_types)) or hasattr(geom, '__geo_interface__'):
165  try:
166  geom = Geometry(geom)
167  except GeometryException:
168  raise ValueError('Could not create geometry from lookup value.')
169  else:
170  raise ValueError('Cannot use object with type %s for a geometry lookup parameter.' % type(geom).__name__)
171 
172  # Assigning the SRID value.
173  geom.srid = self.get_srid(geom)
174 
175  if seq_value:
176  lookup_val = [geom]
177  lookup_val.extend(value[1:])
178  return tuple(lookup_val)

216  """
217  Prepare for the database lookup, and return any spatial parameters

221  field.
