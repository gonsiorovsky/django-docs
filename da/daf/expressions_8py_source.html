150 class DateModifierNode(ExpressionNode):
151  """
152  Node that implements the following syntax:
153  filter(end_date__gt=F('start_date') + datetime.timedelta(days=3, seconds=200))
154 
155  which translates into:
156  POSTGRES:
157  WHERE end_date > (start_date + INTERVAL '3 days 200 seconds')
158 
159  MYSQL:
160  WHERE end_date > (start_date + INTERVAL '3 0:0:200:0' DAY_MICROSECOND)
161 
162  ORACLE:
163  WHERE end_date > (start_date + INTERVAL '3 00:03:20.000000' DAY(1) TO SECOND(6))
164 
165  SQLITE:
166  WHERE end_date > django_format_dtdelta(start_date, "+" "3", "200", "0")
167  (A custom function is used in order to preserve six digits of fractional
168  second information on sqlite, and to format both date and datetime values.)
169 
170  Note that microsecond comparisons are not well supported with MySQL, since
171  MySQL does not store microsecond information.
172 
173  Only adding and subtracting timedeltas is supported, attempts to use other
174  operations raise a TypeError.
175  """
176  def __init__(self, children, connector, negated=False):
177  if len(children) != 2:
178  raise TypeError('Must specify a node and a timedelta.')
179  if not isinstance(children[1], datetime.timedelta):
180  raise TypeError('Second child must be a timedelta.')
181  if connector not in (self.ADD, self.SUB):
182  raise TypeError('Connector must be + or -, not %s' % connector)
183  super(DateModifierNode, self).__init__(children, connector, negated)
