74  def _make_feature(self, feat_id):
75  """
76  Helper routine for __getitem__ that constructs a Feature from the given
77  Feature ID. If the OGR Layer does not support random-access reading,
78  then each feature of the layer will be incremented through until the
79  a Feature is found matching the given feature ID.
80  """
81  if self._random_read:
82  # If the Layer supports random reading, return.
83  try:
84  return Feature(capi.get_feature(self.ptr, feat_id), self)
85  except OGRException:
86  pass
87  else:
88  # Random access isn't supported, have to increment through
89  # each feature until the given feature ID is encountered.
90  for feat in self:
91  if feat.fid == feat_id: return feat
92  # Should have returned a Feature, raise an OGRIndexError.
93  raise OGRIndexError('Invalid feature id: %s.' % feat_id)
