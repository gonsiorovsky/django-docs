225  Checks if the target file should be deleted if it already exists

239  except (OSError, NotImplementedError, AttributeError):
240  pass
241  else:
242  # The full path of the target file
243  if self.local:
244  full_path = self.storage.path(prefixed_path)
245  else:
246  full_path = None
247  # Skip the file if the source file is younger
248  # Avoid sub-second precision (see #14665, #19540)
249  if (target_last_modified.replace(microsecond=0)
250  >= source_last_modified.replace(microsecond=0)):

253  (not self.symlink and full_path

261  self.log("Pretending to delete '%s'" % path)
