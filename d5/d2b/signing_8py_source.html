1 """
2 Functions for creating and restoring url-safe signed JSON objects.
3 
4 The format used looks like this:
5 
6 >>> signing.dumps("hello")
7 'ImhlbGxvIg:1QaUZC:YIye-ze3TTx7gtSv422nZA4sgmk'
8 
9 There are two components here, separated by a ':'. The first component is a
10 URLsafe base64 encoded JSON of the object passed to dumps(). The second
11 component is a base64 encoded hmac/SHA1 hash of "$first_component:$secret"
12 
13 signing.loads(s) checks the signature and returns the deserialised object.
14 If the signature fails, a BadSignature exception is raised.
15 
16 >>> signing.loads("ImhlbGxvIg:1QaUZC:YIye-ze3TTx7gtSv422nZA4sgmk")
17 u'hello'
18 >>> signing.loads("ImhlbGxvIg:1QaUZC:YIye-ze3TTx7gtSv422nZA4sgmk-modified")
19 ...
20 BadSignature: Signature failed: ImhlbGxvIg:1QaUZC:YIye-ze3TTx7gtSv422nZA4sgmk-modified
21 
22 You can optionally compress the JSON prior to base64 encoding it to save
23 space, using the compress=True argument. This checks if compression actually
24 helps and only applies compression if the result is a shorter string:
25 
26 >>> signing.dumps(range(1, 20), compress=True)
27 '.eJwFwcERACAIwLCF-rCiILN47r-GyZVJsNgkxaFxoDgxcOHGxMKD_T7vhAml:1QaUaL:BA0thEZrp4FQVXIXuOvYJtLJSrQ'
28 
29 The fact that the string is compressed is signalled by the prefixed '.' at the
30 start of the base64 JSON.
31 
32 There are 65 url-safe characters: the 64 used by url-safe base64 and the ':'.
33 These functions make use of all of them.
34 """

94 def dumps(obj, key=None, salt='django.core.signing', serializer=JSONSerializer, compress=False):
95  """
96  Returns URL-safe, sha1 signed base64 compressed JSON string. If key is
97  None, settings.SECRET_KEY is used instead.
98 
99  If compress is True (not the default) checks if compressing using zlib can
100  save some space. Prepends a '.' to signify compression. This is included
101  in the signature, to protect against zip bombs.
102 
103  Salt can be used to namespace the hash, so that a signed string is
104  only valid for a given namespace. Leaving this at the default
105  value or re-using a salt value across different parts of your
106  application without good cause is a security risk.
107 
108  The serializer is expected to return a bytestring.
109  """
110  data = serializer().dumps(obj)
111 
112  # Flag for if it's been compressed or not
113  is_compressed = False
114 
115  if compress:
116  # Avoid zlib dependency unless compress is being used
117  compressed = zlib.compress(data)
118  if len(compressed) < (len(data) - 1):
119  data = compressed
120  is_compressed = True
121  base64d = b64_encode(data)
122  if is_compressed:
123  base64d = b'.' + base64d
124  return TimestampSigner(key, salt=salt).sign(base64d)

128  """
129  Reverse of dumps(), raises BadSignature if signature fails.
130 
131  The serializer is expected to accept a bytestring.
132  """
133  # TimestampSigner.unsign always returns unicode but base64 and zlib
134  # compression operate on bytes.
135  base64d = force_bytes(TimestampSigner(key, salt=salt).unsign(s, max_age=max_age))

147 class Signer(object):
148 
149  def __init__(self, key=None, sep=':', salt=None):
150  # Use of native strings in all versions of Python
151  self.sep = str(sep)
152  self.key = str(key or settings.SECRET_KEY)
153  self.salt = str(salt or
154  '%s.%s' % (self.__class__.__module__, self.__class__.__name__))
155 
156  def signature(self, value):
157  signature = base64_hmac(self.salt + 'signer', value, self.key)
158  # Convert the signature from bytes to str only on Python 3
159  return force_str(signature)
160 
161  def sign(self, value):
162  value = force_str(value)
163  return str('%s%s%s') % (value, self.sep, self.signature(value))
164 
165  def unsign(self, signed_value):
166  signed_value = force_str(signed_value)
167  if not self.sep in signed_value:
168  raise BadSignature('No "%s" found in value' % self.sep)
169  value, sig = signed_value.rsplit(self.sep, 1)
170  if constant_time_compare(sig, self.signature(value)):
171  return force_text(value)
172  raise BadSignature('Signature "%s" does not match' % sig)

185  def unsign(self, value, max_age=None):
186  """
187  Retrieve original value and check it wasn't signed more
188  than max_age seconds ago.
189  """
190  result = super(TimestampSigner, self).unsign(value)
191  value, timestamp = result.rsplit(self.sep, 1)
192  timestamp = baseconv.base62.decode(timestamp)
193  if max_age is not None:
