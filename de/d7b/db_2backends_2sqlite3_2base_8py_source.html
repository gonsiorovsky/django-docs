394  def check_constraints(self, table_names=None):
395  """
396  Checks each table name in `table_names` for rows with invalid foreign key references. This method is
397  intended to be used in conjunction with `disable_constraint_checking()` and `enable_constraint_checking()`, to
398  determine if rows with invalid references were entered while constraint checks were off.
399 
400  Raises an IntegrityError on the first invalid foreign key reference encountered (if any) and provides
401  detailed information about the invalid reference in the error message.
402 
403  Backends can override this method if they can more directly apply constraint checking (e.g. via "SET CONSTRAINTS
404  ALL IMMEDIATE")
405  """
406  cursor = self.cursor()
407  if table_names is None:
408  table_names = self.introspection.table_names(cursor)
409  for table_name in table_names:
410  primary_key_column_name = self.introspection.get_primary_key_column(cursor, table_name)
411  if not primary_key_column_name:
412  continue
413  key_columns = self.introspection.get_key_columns(cursor, table_name)
414  for column_name, referenced_table_name, referenced_column_name in key_columns:
415  cursor.execute("""
416  SELECT REFERRING.`%s`, REFERRING.`%s` FROM `%s` as REFERRING
417  LEFT JOIN `%s` as REFERRED
418  ON (REFERRING.`%s` = REFERRED.`%s`)
419  WHERE REFERRING.`%s` IS NOT NULL AND REFERRED.`%s` IS NULL"""
420  % (primary_key_column_name, column_name, table_name, referenced_table_name,
421  column_name, referenced_column_name, column_name, referenced_column_name))
422  for bad_row in cursor.fetchall():
423  raise utils.IntegrityError("The row in table '%s' with primary key '%s' has an invalid "
424  "foreign key: %s.%s contains a value '%s' that does not have a corresponding value in %s.%s."
425  % (table_name, bad_row[0], table_name, column_name, bad_row[1],
