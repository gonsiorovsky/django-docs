110 def url_params_from_lookup_dict(lookups):
111  """
112  Converts the type of lookups specified in a ForeignKey limit_choices_to
113  attribute to a dictionary of query parameters
114  """
115  params = {}
116  if lookups and hasattr(lookups, 'items'):
117  items = []
118  for k, v in lookups.items():
119  if callable(v):
120  v = v()
121  if isinstance(v, (tuple, list)):
122  v = ','.join([str(x) for x in v])
123  elif isinstance(v, bool):
124  # See django.db.fields.BooleanField.get_prep_lookup
125  v = ('0', '1')[v]
126  else:
127  v = six.text_type(v)
128  items.append((k, v))
129  params.update(dict(items))
130  return params
