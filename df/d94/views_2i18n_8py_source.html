16 def set_language(request):
17  """
18  Redirect to a given url while setting the chosen language in the
19  session or cookie. The url and the language code need to be
20  specified in the request parameters.
21 
22  Since this view changes how the user will see the rest of the site, it must
23  only be accessed as a POST request. If called as a GET request, it will
24  redirect to the page in the request (the 'next' parameter) without changing
25  any state.
26  """
27  next = request.REQUEST.get('next')
28  if not is_safe_url(url=next, host=request.get_host()):
29  next = request.META.get('HTTP_REFERER')
30  if not is_safe_url(url=next, host=request.get_host()):
31  next = '/'
32  response = http.HttpResponseRedirect(next)
33  if request.method == 'POST':
34  lang_code = request.POST.get('language', None)
35  if lang_code and check_for_language(lang_code):
36  if hasattr(request, 'session'):
37  request.session['django_language'] = lang_code
38  else:
39  response.set_cookie(settings.LANGUAGE_COOKIE_NAME, lang_code)
40  return response

189  packages = [p for p in packages if p == 'django.conf' or p in settings.INSTALLED_APPS]
190  t = {}
191  paths = []
192  en_selected = locale.startswith('en')
193  en_catalog_missing = True
194  # paths of requested packages
195  for package in packages:
196  p = importlib.import_module(package)
197  path = os.path.join(os.path.dirname(upath(p.__file__)), 'locale')
198  paths.append(path)

200  paths.extend(list(reversed(settings.LOCALE_PATHS)))
201  # first load all english languages files for defaults
202  for path in paths:
203  try:
204  catalog = gettext_module.translation(domain, path, ['en'])
205  t.update(catalog._catalog)
206  except IOError:
207  pass
208  else:
209  # 'en' is the selected language and at least one of the packages
210  # listed in `packages` has an 'en' catalog
211  if en_selected:
212  en_catalog_missing = False
213  # next load the settings.LANGUAGE_CODE translations if it isn't english
214  if default_locale != 'en':
215  for path in paths:
216  try:
217  catalog = gettext_module.translation(domain, path, [default_locale])
218  except IOError:
219  catalog = None

221  t.update(catalog._catalog)
222  # last load the currently selected language, if it isn't identical to the default.
223  if locale != default_locale:

225  # translation catalog (presumably due to being the language translated
226  # from) then a wrong language catalog might have been loaded in the
227  # previous step. It needs to be discarded.
228  if en_selected and en_catalog_missing:
229  t = {}
230  else:

232  for path in paths:

246  if plural is not None:

248  # Plural-Forms: nplurals=3; plural=n%10==1 && n%100!=11 ? 0 : n%10>=2 && n%10<=4 && (n%100<10 || n%100>=20) ? 1 : 2;
