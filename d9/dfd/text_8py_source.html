29 def wrap(text, width):
30  """
31  A word-wrap function that preserves existing line breaks and most spaces in
32  the text. Expects that existing line breaks are posix newlines.
33  """
34  text = force_text(text)
35  def _generator():
36  it = iter(text.split(' '))
37  word = next(it)
38  yield word
39  pos = len(word) - word.rfind('\n') - 1
40  for word in it:
41  if "\n" in word:
42  lines = word.split('\n')
43  else:
44  lines = (word,)
45  pos += len(lines[0]) + 1
46  if pos > width:
47  yield '\n'
48  pos = len(lines[-1])
49  else:
50  yield ' '
51  if len(lines) > 1:
52  pos = len(lines[-1])
53  yield word
54  return ''.join(_generator())

144  def _html_words(self, length, truncate):
145  """
146  Truncates HTML to a certain number of words (not counting tags and
147  comments). Closes opened tags if they were correctly closed in the
148  given HTML.
149 
150  Newlines in the HTML are preserved.
151  """
152  if length <= 0:
153  return ''
154  html4_singlets = (
155  'br', 'col', 'link', 'base', 'img',
156  'param', 'area', 'hr', 'input'
157  )
158  # Count non-HTML words and keep note of open tags
159  pos = 0
160  end_text_pos = 0
161  words = 0
162  open_tags = []
163  while words <= length:
164  m = re_words.search(self._wrapped, pos)
165  if not m:
166  # Checked through whole string
167  break
168  pos = m.end(0)
169  if m.group(1):
170  # It's an actual non-HTML word
171  words += 1
172  if words == length:
173  end_text_pos = pos
174  continue
175  # Check for tag
176  tag = re_tag.match(m.group(0))
177  if not tag or end_text_pos:
178  # Don't worry about non tags or tags after our truncate point
179  continue
180  closing_tag, tagname, self_closing = tag.groups()
181  # Element names are always case-insensitive
182  tagname = tagname.lower()
183  if self_closing or tagname in html4_singlets:
184  pass
185  elif closing_tag:
186  # Check for match in open tags list
187  try:
188  i = open_tags.index(tagname)
189  except ValueError:
190  pass
191  else:
192  # SGML: An end tag closes, back to the matching start tag,
193  # all unclosed intervening start tags with omitted end tags
194  open_tags = open_tags[i + 1:]
195  else:
196  # Add it to the start of the open tags list
197  open_tags.insert(0, tagname)
198  if words <= length:
199  # Don't try to close tags if we don't need to truncate
200  return self._wrapped
201  out = self._wrapped[:end_text_pos]
202  truncate_text = self.add_truncation_text('', truncate)
203  if truncate_text:
204  out += truncate_text
205  # Close any tags still open
206  for tag in open_tags:
207  out += '</%s>' % tag
208  # Return string
209  return out
