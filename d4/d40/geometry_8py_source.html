49  def __init__(self, geo_input, srid=None):
50  """
51  The base constructor for GEOS geometry objects, and may take the
52  following inputs:
53 
54  * strings:
55  - WKT
56  - HEXEWKB (a PostGIS-specific canonical form)
57  - GeoJSON (requires GDAL)
58  * buffer:
59  - WKB
60 
61  The `srid` keyword is used to specify the Source Reference Identifier
62  (SRID) number for this Geometry. If not set, the SRID will be None.
63  """
64  if isinstance(geo_input, bytes):
65  geo_input = force_text(geo_input)
66  if isinstance(geo_input, six.string_types):
67  wkt_m = wkt_regex.match(geo_input)
68  if wkt_m:
69  # Handling WKT input.
70  if wkt_m.group('srid'): srid = int(wkt_m.group('srid'))
71  g = wkt_r().read(force_bytes(wkt_m.group('wkt')))
72  elif hex_regex.match(geo_input):
73  # Handling HEXEWKB input.
74  g = wkb_r().read(force_bytes(geo_input))
75  elif gdal.HAS_GDAL and json_regex.match(geo_input):
76  # Handling GeoJSON input.
77  g = wkb_r().read(gdal.OGRGeometry(geo_input).wkb)
78  else:
79  raise ValueError('String or unicode input unrecognized as WKT EWKT, and HEXEWKB.')
80  elif isinstance(geo_input, GEOM_PTR):
81  # When the input is a pointer to a geomtry (GEOM_PTR).
82  g = geo_input
83  elif isinstance(geo_input, memoryview):
84  # When the input is a buffer (WKB).
85  g = wkb_r().read(geo_input)
86  elif isinstance(geo_input, GEOSGeometry):
87  g = capi.geom_clone(geo_input.ptr)
88  else:
89  # Invalid geometry type.
90  raise TypeError('Improper geometry input type: %s' % str(type(geo_input)))
91 
92  if bool(g):
93  # Setting the pointer object with a valid pointer.
94  self.ptr = g

491  def transform(self, ct, clone=False):
492  """
493  Requires GDAL. Transforms the geometry according to the given
494  transformation object, which may be an integer SRID, and WKT or
495  PROJ.4 string. By default, the geometry is transformed in-place and
496  nothing is returned. However if the `clone` keyword is set, then this
497  geometry will not be modified and a transformed clone will be returned
498  instead.
499  """
500  srid = self.srid
501 
502  if ct == srid:
503  # short-circuit where source & dest SRIDs match
504  if clone:
505  return self.clone()
506  else:
507  return
508 
509  if (srid is None) or (srid < 0):
510  raise GEOSException("Calling transform() with no SRID set is not supported")
511 
512  if not gdal.HAS_GDAL:
513  raise GEOSException("GDAL library is not available to transform() geometry.")
514 
515  # Creating an OGR Geometry, which is then transformed.
516  g = self.ogr
517  g.transform(ct)
518  # Getting a new GEOS pointer
519  ptr = wkb_r().read(g.wkb)
520  if clone:
521  # User wants a cloned transformed geometry returned.
522  return GEOSGeometry(ptr, srid=g.srid)
523  if ptr:
524  # Reassigning pointer, and performing post-initialization setup
525  # again due to the reassignment.
526  capi.destroy_geom(self.ptr)
527  self.ptr = ptr
528  self._post_init(g.srid)
529  else:
530  raise GEOSException('Transformed WKB was invalid.')
