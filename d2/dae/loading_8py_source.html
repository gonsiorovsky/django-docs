54  def _populate(self):
55  """
56  Fill in all the cache information. This method is threadsafe, in the
57  sense that every caller will see the same state upon return, and if the
58  cache is already initialised, it does no work.
59  """
60  if self.loaded:
61  return
62  # Note that we want to use the import lock here - the app loading is
63  # in many cases initiated implicitly by importing, and thus it is
64  # possible to end up in deadlock when one thread initiates loading
65  # without holding the importer lock and another thread then tries to
66  # import something which also launches the app loading. For details of
67  # this situation see #18251.
68  imp.acquire_lock()
69  try:
70  if self.loaded:
71  return
72  for app_name in settings.INSTALLED_APPS:
73  if app_name in self.handled:
74  continue
75  self.load_app(app_name, True)
76  if not self.nesting_level:
77  for app_name in self.postponed:
78  self.load_app(app_name)
79  self.loaded = True
80  finally:
81  imp.release_lock()

90  def load_app(self, app_name, can_postpone=False):
91  """
92  Loads the app with the provided fully qualified name, and returns the
93  model module.
94  """
95  self.handled.add(app_name)
96  self.nesting_level += 1
97  app_module = import_module(app_name)
98  try:
99  models = import_module('%s.models' % app_name)
100  except ImportError:
101  self.nesting_level -= 1
102  # If the app doesn't have a models module, we can just ignore the
103  # ImportError and return no models for it.
104  if not module_has_submodule(app_module, 'models'):
105  return None
106  # But if the app does have a models module, we need to figure out
107  # whether to suppress or propagate the error. If can_postpone is
108  # True then it may be that the package is still being imported by
109  # Python and the models module isn't available yet. So we add the
110  # app to the postponed list and we'll try it again after all the
111  # recursion has finished (in populate). If can_postpone is False
112  # then it's time to raise the ImportError.
113  else:
114  if can_postpone:
115  self.postponed.append(app_name)
116  return None
117  else:
118  raise
119 
120  self.nesting_level -= 1
121  if models not in self.app_store:
122  self.app_store[models] = len(self.app_store)
123  self.app_labels[self._label_for(models)] = models
124  return models

135  def get_apps(self):
136  """
137  Returns a list of all installed modules that contain models.
138  """
139  self._populate()
140 
141  apps = self.app_store.items()
142  if self.available_apps is not None:
143  apps = [elt for elt in apps
144  if self._label_for(elt[0]) in self.available_apps]

170  def get_app(self, app_label, emptyOK=False):
171  """
172  Returns the module containing the models for the given app_label.
173 
174  Returns None if the app has no models in it and emptyOK is True.
175 
176  Raises UnavailableApp when set_available_apps() in in effect and
177  doesn't include app_label.
178  """
179  self._populate()
180  imp.acquire_lock()
181  try:
182  for app_name in settings.INSTALLED_APPS:
183  if app_label == app_name.split('.')[-1]:
184  mod = self.load_app(app_name, False)
185  if mod is None and not emptyOK:
186  raise ImproperlyConfigured("App with label %s is missing a models.py module." % app_label)
187  if self.available_apps is not None and app_label not in self.available_apps:
188  raise UnavailableApp("App with label %s isn't available." % app_label)
189  return mod
190  raise ImproperlyConfigured("App with label %s could not be found" % app_label)
191  finally:
192  imp.release_lock()

199  def get_models(self, app_mod=None,
200  include_auto_created=False, include_deferred=False,
201  only_installed=True, include_swapped=False):
202  """
203  Given a module containing models, returns a list of the models.
204  Otherwise returns a list of all installed models.
205 
206  By default, auto-created models (i.e., m2m models without an
207  explicit intermediate table) are not included. However, if you
208  specify include_auto_created=True, they will be.
209 
210  By default, models created to satisfy deferred attribute
211  queries are *not* included in the list of models. However, if
212  you specify include_deferred, they will be.
213 
214  By default, models that aren't part of installed apps will *not*
215  be included in the list of models. However, if you specify
216  only_installed=False, they will be.
217 
218  By default, models that have been swapped out will *not* be
219  included in the list of models. However, if you specify
220  include_swapped, they will be.
221  """
222  cache_key = (app_mod, include_auto_created, include_deferred, only_installed, include_swapped)
223  model_list = None
224  try:
225  model_list = self._get_models_cache[cache_key]
226  if self.available_apps is not None and only_installed:
227  model_list = [m for m in model_list
228  if m._meta.app_label in self.available_apps]
229  return model_list
230  except KeyError:
231  pass
232  self._populate()
233  if app_mod:
234  if app_mod in self.app_store:
235  app_list = [self.app_models.get(self._label_for(app_mod),
236  SortedDict())]
237  else:
238  app_list = []
239  else:
240  if only_installed:
241  app_list = [self.app_models.get(app_label, SortedDict())
242  for app_label in six.iterkeys(self.app_labels)]
243  else:
244  app_list = six.itervalues(self.app_models)
245  model_list = []
246  for app in app_list:
247  model_list.extend(
248  model for model in app.values()
249  if ((not model._deferred or include_deferred) and
250  (not model._meta.auto_created or include_auto_created) and
251  (not model._meta.swapped or include_swapped))
252  )
253  self._get_models_cache[cache_key] = model_list
254  if self.available_apps is not None and only_installed:
255  model_list = [m for m in model_list
256  if m._meta.app_label in self.available_apps]
257  return model_list

282  def register_models(self, app_label, *models):
283  """
284  Register a set of models as belonging to an app.
285  """
286  for model in models:
287  # Store as 'name: model' pair in a dictionary
288  # in the app_models dictionary
289  model_name = model._meta.model_name
290  model_dict = self.app_models.setdefault(app_label, SortedDict())
291  if model_name in model_dict:
292  # The same model may be imported via different paths (e.g.
293  # appname.models and project.appname.models). We use the source
294  # filename as a means to detect identity.
295  fname1 = os.path.abspath(upath(sys.modules[model.__module__].__file__))
296  fname2 = os.path.abspath(upath(sys.modules[model_dict[model_name].__module__].__file__))
297  # Since the filename extension could be .py the first time and
298  # .pyc or .pyo the second time, ignore the extension when
299  # comparing.
300  if os.path.splitext(fname1)[0] == os.path.splitext(fname2)[0]:
301  continue
