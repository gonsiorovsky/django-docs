1 """
2 This module contains helper functions for controlling caching. It does so by
3 managing the "Vary" header of responses. It includes functions to patch the
4 header of response objects directly and decorators that change functions to do
5 that header-patching themselves.
6 
7 For information on the Vary header, see:
8 
9  http://www.w3.org/Protocols/rfc2616/rfc2616-sec14.html#sec14.44
10 
11 Essentially, the "Vary" HTTP header defines which headers a cache should take
12 into account when building its cache key. Requests with the same path but
13 different header content for headers named in "Vary" need to get different
14 cache keys to prevent delivery of wrong content.
15 
16 An example: i18n middleware would need to distinguish caches by the
17 "Accept-language" header.
18 """

34 def patch_cache_control(response, **kwargs):
35  """
36  This function patches the Cache-Control header by adding all
37  keyword arguments to it. The transformation is as follows:
38 
39  * All keyword parameter names are turned to lowercase, and underscores
40  are converted to hyphens.
41  * If the value of a parameter is True (exactly True, not just a
42  true value), only the parameter name is added to the header.
43  * All other parameters are added with their value, after applying
44  str() to it.
45  """
46  def dictitem(s):
47  t = s.split('=', 1)
48  if len(t) > 1:
49  return (t[0].lower(), t[1])
50  else:
51  return (t[0].lower(), True)
52 
53  def dictvalue(t):
54  if t[1] is True:
55  return t[0]
56  else:
57  return '%s=%s' % (t[0], t[1])
58 
59  if response.has_header('Cache-Control'):
60  cc = cc_delim_re.split(response['Cache-Control'])
61  cc = dict([dictitem(el) for el in cc])
62  else:
63  cc = {}
64 
65  # If there's already a max-age header but we're being asked to set a new
66  # max-age, use the minimum of the two ages. In practice this happens when
67  # a decorator and a piece of middleware both operate on a given view.
68  if 'max-age' in cc and 'max_age' in kwargs:
69  kwargs['max_age'] = min(int(cc['max-age']), kwargs['max_age'])
70 
71  # Allow overriding private caching and vice versa
72  if 'private' in cc and 'public' in kwargs:
73  del cc['private']
74  elif 'public' in cc and 'private' in kwargs:
75  del cc['public']
76 
77  for (k, v) in kwargs.items():
78  cc[k.replace('_', '-')] = v
79  cc = ', '.join([dictvalue(el) for el in cc.items()])
80  response['Cache-Control'] = cc

102 def patch_response_headers(response, cache_timeout=None):
103  """
104  Adds some useful headers to the given HttpResponse object:
105  ETag, Last-Modified, Expires and Cache-Control
106 
107  Each header is only added if it isn't already set.
108 
109  cache_timeout is in seconds. The CACHE_MIDDLEWARE_SECONDS setting is used
110  by default.
111  """
112  if cache_timeout is None:
113  cache_timeout = settings.CACHE_MIDDLEWARE_SECONDS
114  if cache_timeout < 0:
115  cache_timeout = 0 # Can't have max-age negative
116  if settings.USE_ETAGS and not response.has_header('ETag'):
117  if hasattr(response, 'render') and callable(response.render):
118  response.add_post_render_callback(_set_response_etag)
119  else:
120  response = _set_response_etag(response)
121  if not response.has_header('Last-Modified'):
122  response['Last-Modified'] = http_date()
123  if not response.has_header('Expires'):
124  response['Expires'] = http_date(time.time() + cache_timeout)
125  patch_cache_control(response, max_age=cache_timeout)

133 def patch_vary_headers(response, newheaders):
134  """
135  Adds (or updates) the "Vary" header in the given HttpResponse object.
136  newheaders is a list of header names that should be in "Vary". Existing
137  headers in "Vary" aren't removed.
138  """
139  # Note that we need to keep the original order intact, because cache
140  # implementations may rely on the order of the Vary contents in, say,
141  # computing an MD5 hash.
142  if response.has_header('Vary'):
143  vary_headers = cc_delim_re.split(response['Vary'])
144  else:
145  vary_headers = []
146  # Use .lower() here so we treat headers as case-insensitive.
147  existing_headers = set([header.lower() for header in vary_headers])
148  additional_headers = [newheader for newheader in newheaders
149  if newheader.lower() not in existing_headers]
150  response['Vary'] = ', '.join(vary_headers + additional_headers)

162 def _i18n_cache_key_suffix(request, cache_key):
163  """If necessary, adds the current locale or time zone to the cache key."""
164  if settings.USE_I18N or settings.USE_L10N:
165  # first check if LocaleMiddleware or another middleware added
166  # LANGUAGE_CODE to request, then fall back to the active language
167  # which in turn can also fall back to settings.LANGUAGE_CODE
168  cache_key += '.%s' % getattr(request, 'LANGUAGE_CODE', get_language())
169  if settings.USE_TZ:
170  # The datetime module doesn't restrict the output of tzname().
171  # Windows is known to use non-standard, locale-dependent names.
172  # User-defined tzinfo classes may return absolutely anything.
173  # Hence this paranoid conversion to create a valid cache key.
174  tz_name = force_text(get_current_timezone_name(), errors='ignore')
175  cache_key += '.%s' % tz_name.encode('ascii', 'ignore').decode('ascii').replace(' ', '_')
176  return cache_key

178 def _generate_cache_key(request, method, headerlist, key_prefix):
179  """Returns a cache key from the headers given in the header list."""
180  ctx = hashlib.md5()
181  for header in headerlist:
182  value = request.META.get(header, None)
183  if value is not None:
184  ctx.update(force_bytes(value))
185  path = hashlib.md5(force_bytes(iri_to_uri(request.get_full_path())))
186  cache_key = 'views.decorators.cache.cache_page.%s.%s.%s.%s' % (
187  key_prefix, method, path.hexdigest(), ctx.hexdigest())
188  return _i18n_cache_key_suffix(request, cache_key)

197 def get_cache_key(request, key_prefix=None, method='GET', cache=None):
198  """
199  Returns a cache key based on the request path and query. It can be used
200  in the request phase because it pulls the list of headers to take into
201  account from the global path registry and uses those to build a cache key
202  to check against.
203

205  function returns None.
206  """

218 def learn_cache_key(request, response, cache_timeout=None, key_prefix=None, cache=None):
219  """
220  Learns what headers to take into account for some request path from the
221  response object. It stores those headers in a global path registry so that
222  later access to that path will know what headers to take into account
223  without building the response object itself. The headers are named in the
224  Vary header of the response, but we want to prevent response generation.
225 
226  The list of headers to use for cache key generation is stored in the same
227  cache as the pages themselves. If the cache ages some data out of the
228  cache, this just means that we have to build the response once to get at
229  the Vary header and so at the list of headers to use for the cache key.
230  """
231  if key_prefix is None:
232  key_prefix = settings.CACHE_MIDDLEWARE_KEY_PREFIX
233  if cache_timeout is None:
234  cache_timeout = settings.CACHE_MIDDLEWARE_SECONDS
235  cache_key = _generate_cache_header_key(key_prefix, request)
236  if cache is None:
237  cache = get_cache(settings.CACHE_MIDDLEWARE_ALIAS)
238  if response.has_header('Vary'):
239  is_accept_language_redundant = settings.USE_I18N or settings.USE_L10N
240  # If i18n or l10n are used, the generated cache key will be suffixed
241  # with the current locale. Adding the raw value of Accept-Language is
242  # redundant in that case and would result in storing the same content
243  # under multiple keys in the cache. See #18191 for details.
244  headerlist = []
245  for header in cc_delim_re.split(response['Vary']):
246  header = header.upper().replace('-', '_')
247  if header == 'ACCEPT_LANGUAGE' and is_accept_language_redundant:
248  continue
249  headerlist.append('HTTP_' + header)
250  headerlist.sort()
251  cache.set(cache_key, headerlist, cache_timeout)
252  return _generate_cache_key(request, request.method, headerlist, key_prefix)
253  else:
254  # if there is no Vary header, we still need a cache key
255  # for the request.get_full_path()
256  cache.set(cache_key, [], cache_timeout)
257  return _generate_cache_key(request, request.method, [], key_prefix)
