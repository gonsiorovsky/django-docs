11  Cleans a IPv6 address string.
12 
13  Validity is checked by calling is_valid_ipv6_address() - if an
14  invalid address is passed, ValidationError is raised.
15 
16  Replaces the longest continious zero-sequence with "::" and
17  removes leading zeroes and makes sure all hextets are lowercase.
18 
19  Args:
20  ip_str: A valid IPv6 address.
21  unpack_ipv4: if an IPv4-mapped address is found,
22  return the plain IPv4 address (default=False).
23  error_message: A error message for in the ValidationError.
24 
25  Returns:
26  A compressed IPv6 address, or the same value
27 
28  """

30  best_doublecolon_len = 0
31  doublecolon_start = -1
32  doublecolon_len = 0
33 
34  if not is_valid_ipv6_address(ip_str):
35  raise ValidationError(error_message, code='invalid')
36

38  # IP strings
39  ip_str = _explode_shorthand_ip_string(ip_str)
40 
41  ip_str = _sanitize_ipv4_mapping(ip_str)
42 
43  # If needed, unpack the IPv4 and return straight away
44  # - no need in running the rest of the algorithm
45  if unpack_ipv4:
46  ipv4_unpacked = _unpack_ipv4(ip_str)
47 
48  if ipv4_unpacked:
49  return ipv4_unpacked

52

54  # Remove leading zeroes
55  hextets[index] = hextets[index].lstrip('0')
56  if not hextets[index]:
57  hextets[index] = '0'

63  # Start of a sequence of zeros.
64  doublecolon_start = index

144 def is_valid_ipv6_address(ip_str):
145  """
146  Ensure we have a valid IPv6 address.
147 
148  Args:
149  ip_str: A string, the IPv6 address.
150 
151  Returns:
152  A boolean, True if this is a valid IPv6 address.
153 
154  """
155  from django.core.validators import validate_ipv4_address
156 
157  # We need to have at least one ':'.
158  if ':' not in ip_str:
159  return False
160 
161  # We can only have one '::' shortener.
162  if ip_str.count('::') > 1:
163  return False
164 
165  # '::' should be encompassed by start, digits or end.
166  if ':::' in ip_str:
167  return False
168 
169  # A single colon can neither start nor end an address.
170  if ((ip_str.startswith(':') and not ip_str.startswith('::')) or
171  (ip_str.endswith(':') and not ip_str.endswith('::'))):
172  return False
173 
174  # We can never have more than 7 ':' (1::2:3:4:5:6:7:8 is invalid)
175  if ip_str.count(':') > 7:
176  return False
177 
178  # If we have no concatenation, we need to have 8 fields with 7 ':'.
179  if '::' not in ip_str and ip_str.count(':') != 7:
180  # We might have an IPv4 mapped address.
181  if ip_str.count('.') != 3:
182  return False
183 
184  ip_str = _explode_shorthand_ip_string(ip_str)
185 
186  # Now that we have that all squared away, let's check that each of the
187  # hextets are between 0x0 and 0xFFFF.
188  for hextet in ip_str.split(':'):
189  if hextet.count('.') == 3:
190  # If we have an IPv4 mapped address, the IPv4 portion has to
191  # be at the end of the IPv6 portion.
192  if not ip_str.split(':')[-1] == hextet:
193  return False
194  try:
195  validate_ipv4_address(hextet)
196  except ValidationError:
197  return False
198  else:
199  try:
200  # a value error here means that we got a bad hextet,
201  # something like 0xzzzz
202  if int(hextet, 16) < 0x0 or int(hextet, 16) > 0xFFFF:
203  return False
204  except ValueError:
205  return False
206  return True
