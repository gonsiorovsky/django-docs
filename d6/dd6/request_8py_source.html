385  def urlencode(self, safe=None):
386  """
387  Returns an encoded string of all query string arguments.
388 
389  :arg safe: Used to specify characters which do not require quoting, for
390  example::
391 
392  >>> q = QueryDict('', mutable=True)
393  >>> q['next'] = '/a&b/'
394  >>> q.urlencode()
395  'next=%2Fa%26b%2F'
396  >>> q.urlencode(safe='/')
397  'next=/a%26b/'
398 
399  """
400  output = []
401  if safe:
402  safe = force_bytes(safe, self.encoding)
403  encode = lambda k, v: '%s=%s' % ((quote(k, safe), quote(v, safe)))
404  else:
405  encode = lambda k, v: urlencode({k: v})
406  for k, list_ in self.lists():
407  k = force_bytes(k, self.encoding)
408  output.extend([encode(k, force_bytes(v, self.encoding))
409  for v in list_])
410  return '&'.join(output)

497 def validate_host(host, allowed_hosts):
498  """
499  Validate the given host for this site.
500 
501  Check that the host looks valid and matches a host or host pattern in the
502  given list of ``allowed_hosts``. Any pattern beginning with a period
503  matches a domain and all its subdomains (e.g. ``.example.com`` matches
504  ``example.com`` and any subdomain), ``*`` matches anything, and anything
505  else must match exactly.
506 
507  Note: This function assumes that the given host is lower-cased and has
508  already had the port, if any, stripped off.
509 
510  Return ``True`` for a valid host, ``False`` otherwise.
511 
512  """
513  for pattern in allowed_hosts:
514  pattern = pattern.lower()
515  match = (
516  pattern == '*' or
517  pattern.startswith('.') and (
518  host.endswith(pattern) or host == pattern[1:]
519  ) or
520  pattern == host
521  )
522  if match:
523  return True
524 
525  return False
