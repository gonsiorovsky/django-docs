41 def _check_permission_clashing(custom, builtin, ctype):
42  """
43  Check that permissions for a model do not clash. Raises CommandError if
44  there are duplicate permissions.
45  """
46  pool = set()
47  builtin_codenames = set(p[0] for p in builtin)
48  for codename, _name in custom:
49  if codename in pool:
50  raise CommandError(
51  "The permission codename '%s' is duplicated for model '%s.%s'." %

53  elif codename in builtin_codenames:
54  raise CommandError(
55  "The permission codename '%s' clashes with a builtin permission "

153 def get_default_username(check_db=True):
154  """
155  Try to determine the current system user's username to use as a default.
156 
157  :param check_db: If ``True``, requires that the username does not match an
158  existing ``auth.User`` (otherwise returns an empty string).
159  :returns: The username, or an empty string if no username can be
160  determined.
161  """
162  # If the User model has been swapped out, we can't make any assumptions
163  # about the default user name.
164  if auth_app.User._meta.swapped:
165  return ''
166 
167  default_username = get_system_username()
168  try:
169  default_username = unicodedata.normalize('NFKD', default_username)\
170  .encode('ascii', 'ignore').decode('ascii').replace(' ', '').lower()
171  except UnicodeDecodeError:
172  return ''
173

176  auth_app.User._meta.get_field('username').run_validators(default_username)
177  except exceptions.ValidationError:
178  return ''

181  if check_db and default_username:
