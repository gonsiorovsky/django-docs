411  def combine(self, rhs, connector):
412  """
413  Merge the 'rhs' query into the current one (with any 'rhs' effects
414  being applied *after* (that is, "to the right of") anything in the
415  current query. 'rhs' is not modified during a call to this function.
416 
417  The 'connector' parameter describes how to connect filters from the
418  'rhs' query.
419  """
420  assert self.model == rhs.model, \
421  "Cannot combine queries on two different base models."
422  assert self.can_filter(), \
423  "Cannot combine queries once a slice has been taken."
424  assert self.distinct == rhs.distinct, \
425  "Cannot combine a unique query with a non-unique query."
426  assert self.distinct_fields == rhs.distinct_fields, \
427  "Cannot combine queries with different distinct fields."
428 
429  self.remove_inherited_models()
430  # Work out how to relabel the rhs aliases, if necessary.
431  change_map = {}
432  conjunction = (connector == AND)
433 
434  # Determine which existing joins can be reused. When combining the
435  # query with AND we must recreate all joins for m2m filters. When
436  # combining with OR we can reuse joins. The reason is that in AND
437  # case a single row can't fulfill a condition like:
438  # revrel__col=1 & revrel__col=2
439  # But, there might be two different related rows matching this
440  # condition. In OR case a single True is enough, so single row is
441  # enough, too.
442  #
443  # Note that we will be creating duplicate joins for non-m2m joins in
444  # the AND case. The results will be correct but this creates too many
445  # joins. This is something that could be fixed later on.
446  reuse = set() if conjunction else set(self.tables)
447  # Base table must be present in the query - this is the same
448  # table on both sides.
449  self.get_initial_alias()
450  # Now, add the joins from rhs query into the new query (skipping base
451  # table).
452  for alias in rhs.tables[1:]:
453  table, _, join_type, lhs, join_cols, nullable, join_field = rhs.alias_map[alias]
454  promote = (join_type == self.LOUTER)
455  # If the left side of the join was already relabeled, use the
456  # updated alias.
457  lhs = change_map.get(lhs, lhs)
458  new_alias = self.join(
459  (lhs, table, join_cols), reuse=reuse,
460  outer_if_first=not conjunction, nullable=nullable,
461  join_field=join_field)
462  if promote:
463  self.promote_joins([new_alias])
464  # We can't reuse the same join again in the query. If we have two
465  # distinct joins for the same connection in rhs query, then the
466  # combined query must have two joins, too.
467  reuse.discard(new_alias)
468  change_map[alias] = new_alias
469  if not rhs.alias_refcount[alias]:
470  # The alias was unused in the rhs query. Unref it so that it
471  # will be unused in the new query, too. We have to add and
472  # unref the alias so that join promotion has information of
473  # the join type for the unused alias.
474  self.unref_alias(new_alias)
475

477  # all joins exclusive to either the lhs or the rhs must be converted
478  # to an outer join. RHS joins were already set to outer joins above,
479  # so check which joins were used only in the lhs query.
480  if not conjunction:
481  rhs_used_joins = set(change_map.values())
482  to_promote = [alias for alias in self.tables
483  if alias not in rhs_used_joins]
484  self.promote_joins(to_promote, True)
485 
486  # Now relabel a copy of the rhs where-clause and add it to the current
487  # one.
488  if rhs.where:
489  w = rhs.where.clone()
490  w.relabel_aliases(change_map)
491  if not self.where:
492  # Since 'self' matches everything, add an explicit "include
493  # everything" where-constraint so that connections between the
494  # where clauses won't exclude valid results.
495  self.where.add(EverythingNode(), AND)
496  elif self.where:
497  # rhs has an empty where clause.
498  w = self.where_class()
499  w.add(EverythingNode(), AND)
500  else:
501  w = self.where_class()
502  self.where.add(w, connector)
503 
504  # Selection columns and extra extensions are those provided by 'rhs'.
505  self.select = []
506  for col, field in rhs.select:
507  if isinstance(col, (list, tuple)):
508  new_col = change_map.get(col[0], col[0]), col[1]
509  self.select.append(SelectInfo(new_col, field))
510  else:
511  new_col = col.relabeled_clone(change_map)
512  self.select.append(SelectInfo(new_col, field))
513 
514  if connector == OR:
515  # It would be nice to be able to handle this, but the queries don't
516  # really make sense (or return consistent value sets). Not worth
517  # the extra complexity when you can write a real query instead.
518  if self.extra and rhs.extra:
519  raise ValueError("When merging querysets using 'or', you "
520  "cannot have extra(select=...) on both sides.")
521  self.extra.update(rhs.extra)
522  extra_select_mask = set()
523  if self.extra_select_mask is not None:
524  extra_select_mask.update(self.extra_select_mask)
525  if rhs.extra_select_mask is not None:
526  extra_select_mask.update(rhs.extra_select_mask)
527  if extra_select_mask:
528  self.set_extra_mask(extra_select_mask)
529  self.extra_tables += rhs.extra_tables

536  def deferred_to_data(self, target, callback):
537  """
538  Converts the self.deferred_loading data structure to an alternate data
539  structure, describing the field that *will* be loaded. This is used to
540  compute the columns to select from the database and also by the
541  QuerySet class to work out which fields are being initialised on each
542  model. Models that have all their fields included aren't mentioned in
543  the result, only those that have field restrictions in place.
544 
545  The "target" parameter is the instance that is populated (in place).
546  The "callback" is a function that is called whenever a (model, field)
547  pair need to be added to "target". It accepts three parameters:
548  "target", and the model and list of fields being added for that model.
549  """
550  field_names, defer = self.deferred_loading
551  if not field_names:
552  return

554  seen = {}
555  must_include = {orig_opts.concrete_model: set([orig_opts.pk])}
556  for field_name in field_names:
557  parts = field_name.split(LOOKUP_SEP)
558  cur_model = self.model
559  opts = orig_opts
560  for name in parts[:-1]:
561  old_model = cur_model
562  source = opts.get_field_by_name(name)[0]
563  if is_reverse_o2o(source):
564  cur_model = source.model
565  else:
566  cur_model = source.rel.to
567  opts = cur_model._meta
568  # Even if we're "just passing through" this model, we must add
569  # both the current model's pk and the related reference field
570  # (if it's not a reverse relation) to the things we select.
571  if not is_reverse_o2o(source):
572  must_include[old_model].add(source)
573  add_to_dict(must_include, cur_model, opts.pk)
574  field, model, _, _ = opts.get_field_by_name(parts[-1])
575  if model is None:
576  model = cur_model
577  if not is_reverse_o2o(field):
578  add_to_dict(seen, model, field)
579 
580  if defer:

582  # appear in "seen" (for all models that appear in "seen"). The only
583  # slight complexity here is handling fields that exist on parent
584  # models.
585  workset = {}
586  for model, values in six.iteritems(seen):
587  for field, m in model._meta.get_fields_with_model():
588  if field in values:
589  continue
590  add_to_dict(workset, m or model, field)
591  for model, values in six.iteritems(must_include):
592  # If we haven't included a model in workset, we don't add the
593  # corresponding must_include fields for that model, since an
594  # empty set means "include all fields". That's why there's no
595  # "else" branch here.
596  if model in workset:
597  workset[model].update(values)
598  for model, values in six.iteritems(workset):
599  callback(target, model, values)
600  else:
601  for model, values in six.iteritems(must_include):
602  if model in seen:
603  seen[model].update(values)
604  else:

606  # included any fields, we have to make sure it's mentioned
607  # so that only the "must include" fields are pulled in.
608  seen[model] = values

610  # in the parent list. Again, it must be mentioned to ensure that
611  # only "must include" fields are pulled in.
612  for model in orig_opts.get_parent_list():
613  if model not in seen:
614  seen[model] = set()

616  callback(target, model, values)

665  def promote_joins(self, aliases, unconditional=False):
666  """
667  Promotes recursively the join type of given aliases and its children to
668  an outer join. If 'unconditional' is False, the join is only promoted if
669  it is nullable or the parent join is an outer join.
670 
671  Note about join promotion: When promoting any alias, we make sure all
672  joins which start from that alias are promoted, too. When adding a join
673  in join(), we make sure any join added to already existing LOUTER join
674  is generated as LOUTER. This ensures we don't ever have broken join
675  chains which contain first a LOUTER join, then an INNER JOIN, that is
676  this kind of join should never be generated: a LOUTER b INNER c. The
677  reason for avoiding this type of join chain is that the INNER after
678  the LOUTER will effectively remove any effect the LOUTER had.
679  """
680  aliases = list(aliases)
681  while aliases:
682  alias = aliases.pop(0)
683  if self.alias_map[alias].join_cols[0][1] is None:
684  # This is the base table (first FROM entry) - this table
685  # isn't really joined at all in the query, so we should not
686  # alter its join type.
687  continue
688  parent_alias = self.alias_map[alias].lhs_alias
689  parent_louter = (parent_alias
690  and self.alias_map[parent_alias].join_type == self.LOUTER)
691  already_louter = self.alias_map[alias].join_type == self.LOUTER
692  if ((unconditional or self.alias_map[alias].nullable
693  or parent_louter) and not already_louter):
694  data = self.alias_map[alias]._replace(join_type=self.LOUTER)
695  self.alias_map[alias] = data
696  # Join type of 'alias' changed, so re-examine all aliases that
697  # refer to this one.
698  aliases.extend(
699  join for join in self.alias_map.keys()
700  if (self.alias_map[join].lhs_alias == alias
701  and join not in aliases))

725  def change_aliases(self, change_map):
726  """
727  Changes the aliases in change_map (which maps old-alias -> new-alias),
728  relabelling any references to them in select columns and the where
729  clause.
730  """
731  assert set(change_map.keys()).intersection(set(change_map.values())) == set()
732 
733  def relabel_column(col):
734  if isinstance(col, (list, tuple)):
735  old_alias = col[0]
736  return (change_map.get(old_alias, old_alias), col[1])
737  else:
738  return col.relabeled_clone(change_map)
739  # 1. Update references in "select" (normal columns plus aliases),
740  # "group by", "where" and "having".
741  self.where.relabel_aliases(change_map)
742  self.having.relabel_aliases(change_map)
743  if self.group_by:
744  self.group_by = [relabel_column(col) for col in self.group_by]
745  self.select = [SelectInfo(relabel_column(s.col), s.field)
746  for s in self.select]
747  self.aggregates = SortedDict(
748  (key, relabel_column(col)) for key, col in self.aggregates.items())
749 
750  # 2. Rename the alias in the internal table/alias datastructures.
751  for ident, aliases in self.join_map.items():
752  del self.join_map[ident]
753  aliases = tuple([change_map.get(a, a) for a in aliases])
754  ident = (change_map.get(ident[0], ident[0]),) + ident[1:]
755  self.join_map[ident] = aliases
756  for old_alias, new_alias in six.iteritems(change_map):
757  alias_data = self.alias_map[old_alias]
758  alias_data = alias_data._replace(rhs_alias=new_alias)
759  self.alias_refcount[new_alias] = self.alias_refcount[old_alias]
760  del self.alias_refcount[old_alias]
761  self.alias_map[new_alias] = alias_data
762  del self.alias_map[old_alias]
763 
764  table_aliases = self.table_map[alias_data.table_name]
765  for pos, alias in enumerate(table_aliases):
766  if alias == old_alias:
767  table_aliases[pos] = new_alias
768  break
769  for pos, alias in enumerate(self.tables):
770  if alias == old_alias:
771  self.tables[pos] = new_alias
772  break
773  for key, alias in self.included_inherited_models.items():
774  if alias in change_map:
775  self.included_inherited_models[key] = change_map[alias]
776 
777  # 3. Update any joins that refer to the old alias.
778  for alias, data in six.iteritems(self.alias_map):
779  lhs = data.lhs_alias
780  if lhs in change_map:
781  data = data._replace(lhs_alias=change_map[lhs])
782  self.alias_map[alias] = data
783  # 4. Update the temporary _lookup_joins list
784  if hasattr(self, '_lookup_joins'):
785  self._lookup_joins = [change_map.get(lj, lj) for lj in self._lookup_joins]

787  def bump_prefix(self, exceptions=()):
788  """
789  Changes the alias prefix to the next letter in the alphabet and
790  relabels all the aliases. Even tables that previously had no alias will
791  get an alias after this call (it's mostly used for nested queries and
792  the outer query will already be using the non-aliased table name).

833  def join(self, connection, reuse=None, outer_if_first=False,
834  nullable=False, join_field=None):
835  """
836  Returns an alias for the join in 'connection', either reusing an
837  existing alias for that join or creating a new one. 'connection' is a
838  tuple (lhs, table, join_cols) where 'lhs' is either an existing
839  table alias or a table name. 'join_cols' is a tuple of tuples containing
840  columns to join on ((l_id1, r_id1), (l_id2, r_id2)). The join corresponds
841  to the SQL equivalent of::
842 
843  lhs.l_id1 = table.r_id1 AND lhs.l_id2 = table.r_id2
844 
845  The 'reuse' parameter can be either None which means all joins
846  (matching the connection) are reusable, or it can be a set containing
847  the aliases that can be reused.
848 
849  If 'outer_if_first' is True and a new join is created, it will have the
850  LOUTER join type.
851 
852  A join is always created as LOUTER if the lhs alias is LOUTER to make
853  sure we do not generate chains like t1 LOUTER t2 INNER t3.
854 
855  If 'nullable' is True, the join can potentially involve NULL values and
856  is a candidate for promotion (to "left outer") when combining querysets.
857 
858  The 'join_field' is the field we are joining along (if any).
859  """
860  lhs, table, join_cols = connection
861  assert lhs is None or join_field is not None
862  existing = self.join_map.get(connection, ())
863  if reuse is None:
864  reuse = existing
865  else:
866  reuse = [a for a in existing if a in reuse]
867  for alias in reuse:
868  if join_field and self.alias_map[alias].join_field != join_field:
869  # The join_map doesn't contain join_field (mainly because
870  # fields in Query structs are problematic in pickling), so
871  # check that the existing join is created using the same
872  # join_field used for the under work join.
873  continue
874  self.ref_alias(alias)
875  return alias
876 
877  # No reuse is possible, so we need a new alias.
878  alias, _ = self.table_alias(table, True)
879  if not lhs:
880  # Not all tables need to be joined to anything. No join type
881  # means the later columns are ignored.
882  join_type = None
883  elif outer_if_first or self.alias_map[lhs].join_type == self.LOUTER:
884  # We need to use LOUTER join if asked by outer_if_first or if the
885  # LHS table is left-joined in the query.
886  join_type = self.LOUTER
887  else:
888  join_type = self.INNER
889  join = JoinInfo(table, alias, join_type, lhs, join_cols or ((None, None),), nullable,
890  join_field)
891  self.alias_map[alias] = join
892  if connection in self.join_map:
893  self.join_map[connection] += (alias,)
894  else:
895  self.join_map[connection] = (alias,)
896  return alias

919  def join_parent_model(self, opts, model, alias, seen):

922  a parent of 'opts' or if it is None this method is a no-op.

1011  def build_filter(self, filter_expr, branch_negated=False, current_negated=False,
1012  can_reuse=None):
1013  """
1014  Builds a WhereNode for a single filter clause, but doesn't add it
1015  to this Query. Query.add_q() will then add this filter to the where
1016  or having Node.
1017 
1018  The 'branch_negated' tells us if the current branch contains any
1019  negations. This will be used to determine if subqueries are needed.
1020 
1021  The 'current_negated' is used to determine if the current filter is
1022  negated or not and this will be used to determine if IS NULL filtering
1023  is needed.
1024 
1025  The difference between current_netageted and branch_negated is that
1026  branch_negated is set on first negation, but current_negated is
1027  flipped for each negation.
1028 
1029  Note that add_filter will not do any negating itself, that is done
1030  upper in the code by add_q().
1031 
1032  The 'can_reuse' is a set of reusable joins for multijoins.
1033 
1034  The method will create a filter clause that can be added to the current
1035  query. However, if the filter isn't added to the query then the caller
1036  is responsible for unreffing the joins used.
1037  """
1038  arg, value = filter_expr
1039  parts = arg.split(LOOKUP_SEP)
1040  if not parts:
1041  raise FieldError("Cannot parse keyword query %r" % arg)
1042 
1043  # Work out the lookup type and remove it from the end of 'parts',
1044  # if necessary.
1045  lookup_type = 'exact' # Default lookup type
1046  num_parts = len(parts)
1047  if (len(parts) > 1 and parts[-1] in self.query_terms
1048  and arg not in self.aggregates):
1049  # Traverse the lookup query to distinguish related fields from
1050  # lookup types.
1051  lookup_model = self.model
1052  for counter, field_name in enumerate(parts):
1053  try:
1054  lookup_field = lookup_model._meta.get_field(field_name)
1055  except FieldDoesNotExist:
1056  # Not a field. Bail out.
1057  lookup_type = parts.pop()
1058  break
1059  # Unless we're at the end of the list of lookups, let's attempt
1060  # to continue traversing relations.
1061  if (counter + 1) < num_parts:
1062  try:
1063  lookup_model = lookup_field.rel.to
1064  except AttributeError:
1065  # Not a related field. Bail out.
1066  lookup_type = parts.pop()
1067  break
1068 
1069  clause = self.where_class()
1070  # Interpret '__exact=None' as the sql 'is NULL'; otherwise, reject all
1071  # uses of None as a query value.
1072  if value is None:
1073  if lookup_type != 'exact':
1074  raise ValueError("Cannot use None as a query value")
1075  lookup_type = 'isnull'
1076  value = True
1077  elif callable(value):
1078  value = value()
1079  elif isinstance(value, ExpressionNode):
1080  # If value is a query expression, evaluate it
1081  value = SQLEvaluator(value, self, reuse=can_reuse)
1082  # For Oracle '' is equivalent to null. The check needs to be done
1083  # at this stage because join promotion can't be done at compiler
1084  # stage. Using DEFAULT_DB_ALIAS isn't nice, but it is the best we
1085  # can do here. Similar thing is done in is_nullable(), too.
1086  if (connections[DEFAULT_DB_ALIAS].features.interprets_empty_strings_as_nulls and
1087  lookup_type == 'exact' and value == ''):
1088  value = True
1089  lookup_type = 'isnull'
1090 
1091  for alias, aggregate in self.aggregates.items():
1092  if alias in (parts[0], LOOKUP_SEP.join(parts)):
1093  clause.add((aggregate, lookup_type, value), AND)
1094  return clause
1095 
1096  opts = self.get_meta()
1097  alias = self.get_initial_alias()
1098  allow_many = not branch_negated
1099 
1100  try:
1101  field, sources, opts, join_list, path = self.setup_joins(
1102  parts, opts, alias, can_reuse, allow_many,
1103  allow_explicit_fk=True)
1104  if can_reuse is not None:
1105  can_reuse.update(join_list)
1106  # split_exclude() needs to know which joins were generated for the
1107  # lookup parts
1108  self._lookup_joins = join_list
1109  except MultiJoin as e:
1110  return self.split_exclude(filter_expr, LOOKUP_SEP.join(parts[:e.level]),
1111  can_reuse, e.names_with_path)
1112 
1113  if (lookup_type == 'isnull' and value is True and not current_negated and
1114  len(join_list) > 1):
1115  # If the comparison is against NULL, we may need to use some left
1116  # outer joins when creating the join chain. This is only done when
1117  # needed, as it's less efficient at the database level.
1118  self.promote_joins(join_list)
1119 
1120  # Process the join list to see if we can remove any inner joins from
1121  # the far end (fewer tables in a query is better). Note that join
1122  # promotion must happen before join trimming to have the join type
1123  # information available when reusing joins.
1124  targets, alias, join_list = self.trim_joins(sources, join_list, path)
1125 
1126  if hasattr(field, 'get_lookup_constraint'):
1127  constraint = field.get_lookup_constraint(self.where_class, alias, targets, sources,
1128  lookup_type, value)
1129  else:
1130  constraint = (Constraint(alias, targets[0].column, field), lookup_type, value)
1131  clause.add(constraint, AND)
1132  if current_negated and (lookup_type != 'isnull' or value is False):
1133  self.promote_joins(join_list)
1134  if (lookup_type != 'isnull' and (
1135  self.is_nullable(targets[0]) or
1136  self.alias_map[join_list[-1]].join_type == self.LOUTER)):
1137  # The condition added here will be SQL like this:
1138  # NOT (col IS NOT NULL), where the first NOT is added in
1139  # upper layers of code. The reason for addition is that if col
1140  # is null, then col != someval will result in SQL "unknown"
1141  # which isn't the same as in Python. The Python None handling
1142  # is wanted, and it can be gotten by
1143  # (col IS NULL OR col != someval)
1144  # <=>
1145  # NOT (col IS NOT NULL AND col = someval).
1146  clause.add((Constraint(alias, targets[0].column, None), 'isnull', False), AND)
1147  return clause

1163  def split_having_parts(self, q_object, negated=False):
1164  """
1165  Returns a list of q_objects which need to go into the having clause
1166  instead of the where clause. Removes the splitted out nodes from the
1167  given q_object. Note that the q_object is altered, so cloning it is
1168  needed.
1169  """
1170  having_parts = []
1171  for c in q_object.children[:]:
1172  # When constucting the having nodes we need to take care to
1173  # preserve the negation status from the upper parts of the tree
1174  if isinstance(c, Node):
1175  # For each negated child, flip the in_negated flag.
1176  in_negated = c.negated ^ negated
1177  if c.connector == OR and self.need_having(c):
1178  # A subtree starting from OR clause must go into having in
1179  # whole if any part of that tree references an aggregate.
1180  q_object.children.remove(c)
1181  having_parts.append(c)
1182  c.negated = in_negated
1183  else:
1184  having_parts.extend(
1185  self.split_having_parts(c, in_negated)[1])
1186  elif self.need_having(c):
1187  q_object.children.remove(c)

1213  current_negated=False):

1216  """

1225  if current_negated:

1251  def names_to_path(self, names, opts, allow_many, allow_explicit_fk):

1255  example).

1263  contain the same value as the final field).

1277  # feature, this could be removed.

1331  def setup_joins(self, names, opts, alias, can_reuse=None, allow_many=True,
1332  allow_explicit_fk=False, outer_if_first=False):
1333  """
1334  Compute the necessary table joins for the passage through the fields
1335  given in 'names'. 'opts' is the Options class for the current model
1336  (which gives the table we are starting from), 'alias' is the alias for
1337  the table to start the joining from.
1338

1340  can be None in which case all joins are reusable or a set of aliases
1341  that can be reused. Note that non-reverse foreign keys are always
1342  reusable when using setup_joins().
1343 
1344  If 'allow_many' is False, then any reverse foreign key seen will
1345  generate a MultiJoin exception.
1346 
1347  The 'allow_explicit_fk' controls if field.attname is allowed in the
1348  lookups.
1349

1351  for any 'where' constraint), the final 'opts' value, the joins and the
1352  field path travelled to generate the joins.
1353 
1354  The target field is the field containing the concrete value. Final
1355  field can be something different, for example foreign key pointing to
1356  that value. Final field is needed for example in some value
1357  conversions (convert 'obj' in fk__id=obj to pk val using the foreign
1358  key field for example).
1359  """
1360  joins = [alias]
1361  # First, generate the path for the names
1362  path, final_field, targets = self.names_to_path(
1363  names, opts, allow_many, allow_explicit_fk)

1365  # joins at this stage - we will need the information about join type
1366  # of the trimmed joins.
1367  for pos, join in enumerate(path):

1369  if join.direct:
1370  nullable = self.is_nullable(join.join_field)
1371  else:
1372  nullable = True
1373  connection = alias, opts.db_table, join.join_field.get_joining_columns()
1374  reuse = can_reuse if join.m2m else None
1375  alias = self.join(
1376  connection, reuse=reuse, nullable=nullable, join_field=join.join_field,
1377  outer_if_first=outer_if_first)
1378  joins.append(alias)
1379  if hasattr(final_field, 'field'):
1380  final_field = final_field.field
1381  return final_field, targets, opts, joins, path

1383  def trim_joins(self, targets, joins, path):
1384  """
1385  The 'target' parameter is the final field being joined to, 'joins'
1386  is the full list of join aliases. The 'path' contain the PathInfos
1387  used to create the joins.
1388 
1389  Returns the final target field and table alias and the new active
1390  joins.
1391 
1392  We will always trim any direct join if we have the target column
1393  available already in the previous table. Reverse joins can't be
1394  trimmed as we don't know if there is anything on the other side of
1395  the join.
1396  """
1397  for pos, info in enumerate(reversed(path)):

1399  break

1408  def split_exclude(self, filter_expr, prefix, can_reuse, names_with_path):
1409  """
1410  When doing an exclude against any kind of N-to-many relation, we need
1411  to use a subquery. This method constructs the nested query, given the
1412  original exclude filter (filter_expr) and the portion up to the first
1413  N-to-many relation field.
1414 
1415  As an example we could have original filter ~Q(child__name='foo').
1416  We would get here with filter_expr = child__name, prefix = child and
1417  can_reuse is a set of joins usable for filters in the original query.
1418 
1419  We will turn this into equivalent of:
1420  WHERE NOT (pk IN (SELECT parent_id FROM thetable
1421  WHERE name = 'foo' AND parent_id IS NOT NULL))
1422 
1423  It might be worth it to consider using WHERE NOT EXISTS as that has
1424  saner null handling, and is easier for the backend's optimizer to
1425  handle.
1426  """
1427  # Generate the inner query.
1428  query = Query(self.model)
1429  query.where.add(query.build_filter(filter_expr), AND)
1430  query.bump_prefix()
1431  query.clear_ordering(True)
1432  # Try to have as simple as possible subquery -> trim leading joins from
1433  # the subquery.
1434  trimmed_prefix, contains_louter = query.trim_start(names_with_path)
1435  query.remove_inherited_models()
1436 
1437  # Add extra check to make sure the selected field will not be null
1438  # since we are adding a IN <subquery> clause. This prevents the
1439  # database from tripping over IN (...,NULL,...) selects and returning
1440  # nothing
1441  if self.is_nullable(query.select[0].field):
1442  alias, col = query.select[0].col
1443  query.where.add((Constraint(alias, col, query.select[0].field), 'isnull', False), AND)
1444 
1445  condition = self.build_filter(
1446  ('%s__in' % trimmed_prefix, query),
1447  current_negated=True, branch_negated=True, can_reuse=can_reuse)
1448  if contains_louter:
1449  or_null_condition = self.build_filter(
1450  ('%s__isnull' % trimmed_prefix, True),
1451  current_negated=True, branch_negated=True, can_reuse=can_reuse)
1452  condition.add(or_null_condition, OR)
1453  # Note that the end result will be:
1454  # (outercol NOT IN innerq AND outercol IS NOT NULL) OR outercol IS NULL.
1455  # This might look crazy but due to how IN works, this seems to be
1456  # correct. If the IS NOT NULL check is removed then outercol NOT
1457  # IN will return UNKNOWN. If the IS NULL check is removed, then if
1458  # outercol IS NULL we will not match the row.
1459  return condition
