30  pass
31  else:
32  raise TypeError('Data source parameter must be a string or a DataSource object.')
33 
34  # Creating the dictionary.
35  _mapping = {}
36 
37  # Generating the field name for each field in the layer.
38  for field in data_source[layer_key].fields:

68  model_name = 'WorldBorders'

75
