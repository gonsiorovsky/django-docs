109  """
110  Determines if the objects in the given queryset-like can be
111  fast-deleted. This can be done if there are no cascades, no
112  parents and no signal listeners for the object class.
113 
114  The 'from_field' tells where we are coming from - we need this to
115  determine if the objects are in fact to be deleted. Allows also
116  skipping parent -> child -> parent chain preventing fast delete of
117  the child.
118  """
119  if from_field and from_field.rel.on_delete is not CASCADE:
120  return False
121  if not (hasattr(objs, 'model') and hasattr(objs, '_raw_delete')):
122  return False
123  model = objs.model

125  or signals.post_delete.has_listeners(model)
126  or signals.m2m_changed.has_listeners(model)):
127  return False

129  # parent when parent delete is cascading to child.
