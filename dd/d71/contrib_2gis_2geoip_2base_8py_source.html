24  # The flags for GeoIP memory caching.
25  # GEOIP_STANDARD - read database from filesystem, uses least memory.
26  #
27  # GEOIP_MEMORY_CACHE - load database into memory, faster performance
28  # but uses more memory
29  #
30  # GEOIP_CHECK_CACHE - check for updated database. If database has been
31  # updated, reload filehandle and/or memory cache. This option
32  # is not thread safe.
33  #
34  # GEOIP_INDEX_CACHE - just cache the most frequently accessed index
35  # portion of the database, resulting in faster lookups than
36  # GEOIP_STANDARD, but less memory usage than GEOIP_MEMORY_CACHE -
37  # useful for larger databases such as GeoIP Organization and
38  # GeoIP City. Note, for GeoIP Country, Region and Netspeed
39  # databases, GEOIP_INDEX_CACHE is equivalent to GEOIP_MEMORY_CACHE
40  #
41  # GEOIP_MMAP_CACHE - load database into mmap shared memory ( not available
42  # on Windows).
43  GEOIP_STANDARD = 0
44  GEOIP_MEMORY_CACHE = 1
45  GEOIP_CHECK_CACHE = 2
46  GEOIP_INDEX_CACHE = 4
47  GEOIP_MMAP_CACHE = 8

64  * path: Base directory to where GeoIP data is located or the full path

68

70  and may be an integer in (0, 1, 2, 4, 8) corresponding to
71  the GEOIP_STANDARD, GEOIP_MEMORY_CACHE, GEOIP_CHECK_CACHE,
72  GEOIP_INDEX_CACHE, and GEOIP_MMAP_CACHE, `GeoIPOptions` C API
73  settings, respectively. Defaults to 0, meaning that the data is read
74  from the disk.

77  'GeoIP.dat'; overrides the GEOIP_COUNTRY settings attribute.

80  'GeoLiteCity.dat'; overrides the GEOIP_CITY settings attribute.
