23 def serve(request, path, document_root=None, show_indexes=False):
24  """
25  Serve static files below a given point in the directory structure.
26 
27  To use, put a URL pattern such as::
28 
29  (r'^(?P<path>.*)$', 'django.views.static.serve', {'document_root' : '/path/to/my/files/'})
30 
31  in your URLconf. You must provide the ``document_root`` param. You may
32  also set ``show_indexes`` to ``True`` if you'd like to serve a basic index
33  of the directory. This index view will use the template hardcoded below,
34  but if you'd like to override it, you can create a template called
35  ``static/directory_index.html``.
36  """
37  path = posixpath.normpath(unquote(path))
38  path = path.lstrip('/')
39  newpath = ''
40  for part in path.split('/'):
41  if not part:
42  # Strip empty path components.
43  continue
44  drive, part = os.path.splitdrive(part)
45  head, part = os.path.split(part)
46  if part in (os.curdir, os.pardir):
47  # Strip '.' and '..' in path.
48  continue
49  newpath = os.path.join(newpath, part).replace('\\', '/')
50  if newpath and path != newpath:
51  return HttpResponseRedirect(newpath)
52  fullpath = os.path.join(document_root, newpath)
53  if os.path.isdir(fullpath):
54  if show_indexes:
55  return directory_index(newpath, fullpath)
56  raise Http404(_("Directory indexes are not allowed here."))
57  if not os.path.exists(fullpath):
58  raise Http404(_('"%(path)s" does not exist') % {'path': fullpath})
59  # Respect the If-Modified-Since header.
60  statobj = os.stat(fullpath)
61  if not was_modified_since(request.META.get('HTTP_IF_MODIFIED_SINCE'),
62  statobj.st_mtime, statobj.st_size):
63  return HttpResponseNotModified()
64  content_type, encoding = mimetypes.guess_type(fullpath)
65  content_type = content_type or 'application/octet-stream'
66  f = open(fullpath, 'rb')
67  response = CompatibleStreamingHttpResponse(iter(lambda: f.read(STREAM_CHUNK_SIZE), b''),
68  content_type=content_type)
69  response["Last-Modified"] = http_date(statobj.st_mtime)
70  if stat.S_ISREG(statobj.st_mode):
71  response["Content-Length"] = statobj.st_size
72  if encoding:
73  response["Content-Encoding"] = encoding
74  return response

120 def was_modified_since(header=None, mtime=0, size=0):
121  """
122  Was something modified since the user last downloaded it?
123 
124  header
125  This is the value of the If-Modified-Since header. If this is None,
126  I'll just return True.
127 
128  mtime
129  This is the modification time of the item we're talking about.
130 
131  size
132  This is the size of the item we're talking about.
133  """
134  try:
135  if header is None:
136  raise ValueError
137  matches = re.match(r"^([^;]+)(; length=([0-9]+))?$", header,
138  re.IGNORECASE)
139  header_mtime = parse_http_date(matches.group(1))
140  header_len = matches.group(3)
141  if header_len and int(header_len) != size:
142  raise ValueError
143  if int(mtime) > header_mtime:
144  raise ValueError
145  except (AttributeError, ValueError, OverflowError):
146  return True
147  return False
