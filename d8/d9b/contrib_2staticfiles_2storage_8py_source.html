170  of the storage.
171  """

174  # fragments and data-uri URLs

182  if url.startswith('/'):
183  sub_level -= 1
184  url_parts = url_parts[1:]

186  start, end = parent_level + 1, parent_level
187  else:

204  def post_process(self, paths, dry_run=False, **options):

206  Post process the given SortedDict of files (called from collectstatic).
207 
208  Processing is actually two separate operations:
209 
210  1. renaming files to include a hash of their content for cache-busting,
211  and copying those files to the target storage.
212  2. adjusting files which contain references to other files so they
213  refer to the cache-busting filenames.
214 
215  If either of these are performed on a file, then that file is considered
216  post-processed.
217  """

220  return

223  hashed_paths = {}

227  adjustable_paths = [path for path in paths if matches(path)]
228

230  path_level = lambda name: len(name.split(os.sep))
231  for name in sorted(paths.keys(), key=path_level, reverse=True):
232

234  # file, which might be somewhere far away, like S3
235  storage, path = paths[name]
236  with storage.open(path) as original_file:
237 
238  # generate the hash with the original content, even for
239  # adjustable files.
240  hashed_name = self.hashed_name(name, original_file)
241 
242  # then get the original's file content..
243  if hasattr(original_file, 'seek'):

245

247  processed = False
248

252  for patterns in self._patterns.values():

265  processed = True

272  hashed_name = force_text(saved_name.replace('\\', '/'))
