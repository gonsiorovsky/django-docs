174  """
175  Define a block that can be overridden by child templates.
176  """
177  # token.split_contents() isn't useful here because this tag doesn't accept variable as arguments
178  bits = token.contents.split()
179  if len(bits) != 2:
180  raise TemplateSyntaxError("'%s' tag takes only one argument" % bits[0])
181  block_name = bits[1]
182  # Keep track of the names of BlockNodes found in this template, so we can
183  # check for duplication.
184  try:
185  if block_name in parser.__loaded_blocks:
186  raise TemplateSyntaxError("'%s' tag with name '%s' appears more than once" % (bits[0], block_name))
187  parser.__loaded_blocks.append(block_name)
188  except AttributeError: # parser.__loaded_blocks isn't a list yet
189  parser.__loaded_blocks = [block_name]
190  nodelist = parser.parse(('endblock',))
191 
192  # This check is kept for backwards-compatibility. See #3100.
193  endblock = parser.next_token()
194  acceptable_endblocks = ('endblock', 'endblock %s' % block_name)
195  if endblock.contents not in acceptable_endblocks:
196  parser.invalid_block_tag(endblock, 'endblock', acceptable_endblocks)
197 
198  return BlockNode(block_name, nodelist)

225

227

235  {% include "foo/some_include" with bar="1" only %}
236  """

239  raise TemplateSyntaxError("%r tag takes at least one argument: the name of the template to be included." % bits[0])
240  options = {}

242  while remaining_bits:
