166 def build_test(label):
167  """
168  Construct a test case with the specified label. Label should be of the
169  form model.TestClass or model.TestClass.test_method. Returns an
170  instantiated test or test suite corresponding to the label provided.
171 
172  """
173  parts = label.split('.')
174  if len(parts) < 2 or len(parts) > 3:
175  raise ValueError("Test label '%s' should be of the form app.TestCase "
176  "or app.TestCase.test_method" % label)
177 
178  #
179  # First, look for TestCase instances with a name that matches
180  #
181  app_module = get_app(parts[0])
182  test_module = get_tests(app_module)
183  TestClass = getattr(app_module, parts[1], None)
184 
185  # Couldn't find the test class in models.py; look in tests.py
186  if TestClass is None:
187  if test_module:
188  TestClass = getattr(test_module, parts[1], None)
189 
190  try:
191  if issubclass(TestClass, (unittest.TestCase, real_unittest.TestCase)):
192  if len(parts) == 2: # label is app.TestClass
193  try:
194  return unittest.TestLoader().loadTestsFromTestCase(
195  TestClass)
196  except TypeError:
197  raise ValueError(
198  "Test label '%s' does not refer to a test class"
199  % label)
200  else: # label is app.TestClass.test_method
201  return TestClass(parts[2])
202  except TypeError:
203  # TestClass isn't a TestClass - it must be a method or normal class
204  pass
205 
206  #
207  # If there isn't a TestCase, look for a doctest that matches
208  #
209  tests = []
210  for module in app_module, test_module:
211  try:
212  doctests = make_doctest(module)
213  # Now iterate over the suite, looking for doctests whose name
214  # matches the pattern that was given
215  for test in doctests:
216  if test._dt_test.name in (
217  '%s.%s' % (module.__name__, '.'.join(parts[1:])),
218  '%s.__test__.%s' % (
219  module.__name__, '.'.join(parts[1:]))):
220  tests.append(test)
221  except ValueError:
222  # No doctests found.
223  pass
224 
225  # If no tests were found, then we were given a bad test label.
226  if not tests:
227  raise ValueError("Test label '%s' does not refer to a test" % label)
228 
229  # Construct a suite out of the tests that matched.
230  return unittest.TestSuite(tests)
