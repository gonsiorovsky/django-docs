429  already_seen=None):
430  """
431  Returns the table alias (the name might be ambiguous, the alias will
432  not be) and column name for ordering by the given 'name' parameter.
433  The 'name' is of the form 'field1__field2__...__fieldN'.

458  def _setup_joins(self, pieces, opts, alias):
459  """
460  A helper method for get_ordering and get_distinct. This method will
461  call query.setup_joins, handle refcounts and then promote the joins.
462 
463  Note that get_ordering and get_distinct must produce same target
464  columns on same input, as the prefixes of get_ordering and get_distinct
465  must match. Executing SQL where this is not true is an error.
466  """
467  if not alias:
468  alias = self.query.get_initial_alias()
469  field, targets, opts, joins, _ = self.query.setup_joins(
470  pieces, opts, alias)
471  # We will later on need to promote those joins that were added to the
472  # query afresh above.
473  joins_to_promote = [j for j in joins if self.query.alias_refcount[j] < 2]
474  alias = joins[-1]
475  cols = [target.column for target in targets]
476  if not field.rel:
477  # To avoid inadvertent trimming of a necessary alias, use the
478  # refcount to show that we are referencing a non-relation field on
479  # the model.
480  self.query.ref_alias(alias)
481 
482  # Must use left outer joins for nullable fields and their relations.
483  # Ordering or distinct must not affect the returned set, and INNER
484  # JOINS for nullable fields could do this.
485  self.query.promote_joins(joins_to_promote)
486  return field, cols, alias, joins, opts

488  def _final_join_removal(self, cols, alias):
489  """
490  A helper method for get_distinct and get_ordering. This method will
491  trim extra not-needed joins from the tail of the join chain.
492 
493  This is very similar to what is done in trim_joins, but we will
494  trim LEFT JOINS here. It would be a good idea to consolidate this
495  method and query.trim_joins().
496  """
497  if alias:
498  while 1:
499  join = self.query.alias_map[alias]
500  lhs_cols, rhs_cols = zip(*[(lhs_col, rhs_col) for lhs_col, rhs_col in join.join_cols])
501  if set(cols) != set(rhs_cols):
502  break
503 
504  cols = [lhs_cols[rhs_cols.index(col)] for col in cols]
505  self.query.unref_alias(alias)
506  alias = join.lhs_alias
507  return cols, alias

509  def get_from_clause(self):
510  """
511  Returns a list of strings that are joined together to go after the
512  "FROM" part of the query, as well as a list any extra parameters that
513  need to be included. Sub-classes, can override this to create a
514  from-clause via a "select".
515 
516  This should only be called after any SQL construction methods that
517  might change the tables we need. This means the select columns,
518  ordering and distinct must be done first.
519  """
520  result = []
521  qn = self.quote_name_unless_alias
522  qn2 = self.connection.ops.quote_name
523  first = True
524  from_params = []
525  for alias in self.query.tables:
526  if not self.query.alias_refcount[alias]:
527  continue
528  try:
529  name, alias, join_type, lhs, join_cols, _, join_field = self.query.alias_map[alias]
530  except KeyError:
531  # Extra tables can end up in self.tables, but not in the
532  # alias_map if they aren't in a join. That's OK. We skip them.
533  continue
534  alias_str = '' if alias == name else (' %s' % alias)
535  if join_type and not first:
536  extra_cond = join_field.get_extra_restriction(
537  self.query.where_class, alias, lhs)
538  if extra_cond:
539  extra_sql, extra_params = extra_cond.as_sql(
540  qn, self.connection)
541  extra_sql = 'AND (%s)' % extra_sql
542  from_params.extend(extra_params)
543  else:
544  extra_sql = ""
545  result.append('%s %s%s ON ('
546  % (join_type, qn(name), alias_str))
547  for index, (lhs_col, rhs_col) in enumerate(join_cols):
548  if index != 0:
549  result.append(' AND ')
550  result.append('%s.%s = %s.%s' %
551  (qn(lhs), qn2(lhs_col), qn(alias), qn2(rhs_col)))
552  result.append('%s)' % extra_sql)
553  else:
554  connector = '' if first else ', '
555  result.append('%s%s%s' % (connector, qn(name), alias_str))
556  first = False
557  for t in self.query.extra_tables:
558  alias, unused = self.query.table_alias(t)
559  # Only add the alias if it's not already present (the table_alias()
560  # calls increments the refcount, so an alias refcount of one means
561  # this is the only reference.
562  if alias not in self.query.alias_map or self.query.alias_refcount[alias] == 1:
563  connector = '' if first else ', '
564  result.append('%s%s' % (connector, qn(alias)))
565  first = False
566  return result, from_params

762  def execute_sql(self, result_type=MULTI):
763  """

765  return value is a single data item if result_type is SINGLE, or an

776  sql, params = self.as_sql()

804  # structure as normally, but ensure it is all read into memory
805  # before going any further.

991  def pre_sql_setup(self):

995  (portable) SQL updates. That is done here.

1003  super(SQLUpdateCompiler, self).pre_sql_setup()

1017  count = query.count_active_tables()

1019  return

1036  # The fast path. Filters and updates in one query.
1037  self.query.add_filter(('pk__in', query))

1039  self.query.alias_refcount[alias] = 0
