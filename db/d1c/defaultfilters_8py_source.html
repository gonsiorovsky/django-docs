31 def stringfilter(func):
32  """
33  Decorator for filters which should only receive unicode objects. The object
34  passed as the first positional argument will be converted to a unicode
35  object.
36  """
37  def _dec(*args, **kwargs):
38  if args:
39  args = list(args)
40  args[0] = force_text(args[0])
41  if (isinstance(args[0], SafeData) and
42  getattr(_dec._decorated_function, 'is_safe', False)):
43  return mark_safe(func(*args, **kwargs))
44  return func(*args, **kwargs)
45 
46  # Include a reference to the real function (used to check original
47  # arguments by the template parser, and to bear the 'is_safe' attribute
48  # when multiple decorators are applied).
49  _dec._decorated_function = getattr(func, '_decorated_function', func)
50 
51  return wraps(func)(_dec)

100  """
101  Displays a float to a specified number of decimal places.
102 
103  If called without an argument, it displays the floating point number with
104  one decimal place -- but only if there's a decimal place to be displayed:
105 
106  * num1 = 34.23234
107  * num2 = 34.00000
108  * num3 = 34.26000
109  * {{ num1|floatformat }} displays "34.2"
110  * {{ num2|floatformat }} displays "34"
111  * {{ num3|floatformat }} displays "34.3"
112 
113  If arg is positive, it will always display exactly arg number of decimal
114  places:
115 
116  * {{ num1|floatformat:3 }} displays "34.232"
117  * {{ num2|floatformat:3 }} displays "34.000"
118  * {{ num3|floatformat:3 }} displays "34.260"
119 
120  If arg is negative, it will display arg number of decimal places -- but
121  only if there are places to be displayed:
122 
123  * {{ num1|floatformat:"-3" }} displays "34.232"
124  * {{ num2|floatformat:"-3" }} displays "34"
125  * {{ num3|floatformat:"-3" }} displays "34.260"
126 
127  If the input float is infinity or NaN, the (platform-dependent) string
128  representation of that value will be displayed.
129  """

132  input_val = force_text(text)
133  d = Decimal(input_val)
134  except UnicodeEncodeError:
135  return ''
136  except InvalidOperation:
137  if input_val in special_floats:
138  return input_val
139  try:
140  d = Decimal(force_text(float(text)))
141  except (ValueError, InvalidOperation, TypeError, UnicodeEncodeError):
142  return ''
143  try:
144  p = int(arg)
145  except ValueError:

147 
148  try:

150  except (ValueError, OverflowError, InvalidOperation):

162  tupl = d.as_tuple()

164  prec = abs(p) + units + 1

167  # and `exponent` from `Decimal.as_tuple()` directly.

173  digits.insert(-exponent, '.')
