160  def iterator(self):
161  """
162  An iterator over the results from applying this QuerySet to the
163  database.
164  """
165  fill_cache = False
166  if connections[self.db].features.supports_select_related:
167  fill_cache = self.query.select_related
168  if isinstance(fill_cache, dict):
169  requested = fill_cache
170  else:
171  requested = None
172  max_depth = self.query.max_depth
173 
174  extra_select = list(self.query.extra_select)
175  aggregate_select = list(self.query.aggregate_select)
176 
177  only_load = self.query.get_loaded_field_names()
178  if not fill_cache:
179  fields = self.model._meta.concrete_fields
180 
181  load_fields = []
182  # If only/defer clauses have been specified,
183  # build the list of fields that are to be loaded.
184  if only_load:
185  for field, model in self.model._meta.get_concrete_fields_with_model():
186  if model is None:
187  model = self.model
188  try:
189  if field.name in only_load[model]:
190  # Add a field that has been explicitly included
191  load_fields.append(field.name)
192  except KeyError:
193  # Model wasn't explicitly listed in the only_load table
194  # Therefore, we need to load all fields from this model
195  load_fields.append(field.name)
196 
197  index_start = len(extra_select)
198  aggregate_start = index_start + len(load_fields or self.model._meta.concrete_fields)
199 
200  skip = None
201  if load_fields and not fill_cache:
202  # Some fields have been deferred, so we have to initialise
203  # via keyword arguments.
204  skip = set()
205  init_list = []
206  for field in fields:
207  if field.name not in load_fields:
208  skip.add(field.attname)
209  else:
210  init_list.append(field.attname)
211  model_cls = deferred_class_factory(self.model, skip)
212 
213  # Cache db and model outside the loop
214  db = self.db
215  model = self.model
216  compiler = self.query.get_compiler(using=db)
217  if fill_cache:
218  klass_info = get_klass_info(model, max_depth=max_depth,
219  requested=requested, only_load=only_load)
220  for row in compiler.results_iter():
221  if fill_cache:
222  obj, _ = get_cached_row(row, index_start, db, klass_info,
223  offset=len(aggregate_select))
224  else:
225  # Omit aggregates in object creation.
226  row_data = row[index_start:aggregate_start]
227  if skip:
228  obj = model_cls(**dict(zip(init_list, row_data)))
229  else:
230  obj = model(*row_data)
231 
232  # Store the source database of the object
233  obj._state.db = db
234  # This object came from the database; it's not being added.
235  obj._state.adding = False
236 
237  if extra_select:
238  for i, k in enumerate(extra_select):
239  setattr(obj, k, row[i])
240 
241  # Add the aggregates to the model
242  if aggregate_select:
243  for i, aggregate in enumerate(aggregate_select):
244  setattr(obj, aggregate, row[i + aggregate_start])
245 
246  # Add the known related objects to the model, if there are any
247  if self._known_related_objects:
248  for field, rel_objs in self._known_related_objects.items():
249  # Avoid overwriting objects loaded e.g. by select_related
250  if hasattr(obj, field.get_cache_name()):
251  continue
252  pk = getattr(obj, field.get_attname())
253  try:
254  rel_obj = rel_objs[pk]

325  def bulk_create(self, objs, batch_size=None):
326  """
327  Inserts each of the instances into the database. This does *not* call
328  save() on each of the instances, does not send any pre/post save
329  signals, and does not set the primary key attribute if it is an
330  autoincrement field.
331  """
332  # So this case is fun. When you bulk insert you don't get the primary
333  # keys back (if it's an autoincrement), so you can't insert into the
334  # child tables which references this. There are two workarounds, 1)
335  # this could be implemented if you didn't have an autoincrement pk,
336  # and 2) you could do it by doing O(n) normal inserts into the parent
337  # tables to get the primary keys back, and then doing a single bulk
338  # insert into the childmost table. Some databases might allow doing
339  # this by using RETURNING clause for the insert query. We're punting
340  # on these for now because they are relatively rare cases.
341  assert batch_size is None or batch_size > 0
342  if self.model._meta.parents:
343  raise ValueError("Can't bulk create an inherited model")
344  if not objs:
345  return objs
346  self._for_write = True
347  connection = connections[self.db]
348  fields = self.model._meta.local_concrete_fields
349  with transaction.commit_on_success_unless_managed(using=self.db):
350  if (connection.features.can_combine_inserts_with_and_without_auto_increment_pk
351  and self.model._meta.has_auto_field):
352  self._batched_insert(objs, fields, batch_size)
353  else:
354  objs_with_pk, objs_without_pk = partition(lambda o: o.pk is None, objs)
355  if objs_with_pk:
356  self._batched_insert(objs_with_pk, fields, batch_size)
357  if objs_without_pk:
358  fields= [f for f in fields if not isinstance(f, AutoField)]
359  self._batched_insert(objs_without_pk, fields, batch_size)
360 
361  return objs

1153 def get_klass_info(klass, max_depth=0, cur_depth=0, requested=None,
1154  only_load=None, from_parent=None):
1155  """
1156  Helper function that recursively returns an information for a klass, to be
1157  used in get_cached_row. It exists just to compute this information only
1158  once for entire queryset. Otherwise it would be computed for each row, which
1159  leads to poor perfomance on large querysets.
1160 
1161  Arguments:
1162  * klass - the class to retrieve (and instantiate)
1163  * max_depth - the maximum depth to which a select_related()
1164  relationship should be explored.
1165  * cur_depth - the current depth in the select_related() tree.
1166  Used in recursive calls to determin if we should dig deeper.
1167  * requested - A dictionary describing the select_related() tree
1168  that is to be retrieved. keys are field names; values are
1169  dictionaries describing the keys on that related object that
1170  are themselves to be select_related().
1171  * only_load - if the query has had only() or defer() applied,
1172  this is the list of field names that will be returned. If None,
1173  the full field list for `klass` can be assumed.
1174  * from_parent - the parent model used to get to this model
1175 
1176  Note that when travelling from parent to child, we will only load child
1177  fields which aren't in the parent.
1178  """
1179  if max_depth and requested is None and cur_depth > max_depth:
1180  # We've recursed deeply enough; stop now.
1181  return None
1182 
1183  if only_load:
1184  load_fields = only_load.get(klass) or set()
1185  # When we create the object, we will also be creating populating
1186  # all the parent classes, so traverse the parent classes looking
1187  # for fields that must be included on load.
1188  for parent in klass._meta.get_parent_list():
1189  fields = only_load.get(parent)
1190  if fields:
1191  load_fields.update(fields)
1192  else:
1193  load_fields = None
1194 
1195  if load_fields:
1196  # Handle deferred fields.
1197  skip = set()
1198  init_list = []
1199  # Build the list of fields that *haven't* been requested
1200  for field, model in klass._meta.get_concrete_fields_with_model():
1201  if from_parent and model and issubclass(from_parent, model):
1202  # Avoid loading fields already loaded for parent model for
1203  # child models.
1204  continue
1205  elif field.name not in load_fields:
1206  skip.add(field.attname)
1207  else:
1208  init_list.append(field.attname)
1209  # Retrieve all the requested fields
1210  field_count = len(init_list)
1211  if skip:
1212  klass = deferred_class_factory(klass, skip)
1213  field_names = init_list
1214  else:
1215  field_names = ()
1216  else:
1217  # Load all fields on klass
1218 
1219  field_count = len(klass._meta.concrete_fields)
1220  # Check if we need to skip some parent fields.
1221  if from_parent and len(klass._meta.local_concrete_fields) != len(klass._meta.concrete_fields):
1222  # Only load those fields which haven't been already loaded into
1223  # 'from_parent'.
1224  non_seen_models = [p for p in klass._meta.get_parent_list()
1225  if not issubclass(from_parent, p)]
1226  # Load local fields, too...
1227  non_seen_models.append(klass)
1228  field_names = [f.attname for f in klass._meta.concrete_fields
1229  if f.model in non_seen_models]
1230  field_count = len(field_names)
1231  # Try to avoid populating field_names variable for perfomance reasons.
1232  # If field_names variable is set, we use **kwargs based model init
1233  # which is slower than normal init.
1234  if field_count == len(klass._meta.concrete_fields):
1235  field_names = ()
1236 
1237  restricted = requested is not None
1238 
1239  related_fields = []
1240  for f in klass._meta.fields:
1241  if select_related_descend(f, restricted, requested, load_fields):
1242  if restricted:
1243  next = requested[f.name]
1244  else:
1245  next = None
1246  klass_info = get_klass_info(f.rel.to, max_depth=max_depth, cur_depth=cur_depth+1,
1247  requested=next, only_load=only_load)
1248  related_fields.append((f, klass_info))
1249 
1250  reverse_related_fields = []
1251  if restricted:
1252  for o in klass._meta.get_all_related_objects():
1253  if o.field.unique and select_related_descend(o.field, restricted, requested,
1254  only_load.get(o.model), reverse=True):
1255  next = requested[o.field.related_query_name()]
1256  parent = klass if issubclass(o.model, klass) else None
1257  klass_info = get_klass_info(o.model, max_depth=max_depth, cur_depth=cur_depth+1,
1258  requested=next, only_load=only_load, from_parent=parent)
1259  reverse_related_fields.append((o.field, klass_info))
1260  if field_names:
1261  pk_idx = field_names.index(klass._meta.pk.attname)
1262  else:
1263  pk_idx = klass._meta.pk_index()
1264 
1265  return klass, field_names, field_count, related_fields, reverse_related_fields, pk_idx

1519  Helper function for prefetch_related functionality
1520 
1521  Populates prefetched objects caches for a list of results
1522  from a QuerySet
1523  """
1524  if len(result_cache) == 0:
1525  return # nothing to do
1526

1528 
1529  # We need to be able to dynamically add to the list of prefetch_related
1530  # lookups that we look up (see below). So we need some book keeping to
1531  # ensure we don't do duplicate work.
1532  done_lookups = set() # list of lookups like foo__bar__baz
1533  done_queries = {} # dictionary of things like 'foo__bar': [results]
1534

1536  followed_descriptors = set() # recursion protection

1546  # from the primary QuerySet. It won't be for deeper levels.
1547  obj_list = result_cache
