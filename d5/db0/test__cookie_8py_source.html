27 def stored_cookie_messages_count(storage, response):
28  """
29  Returns an integer containing the number of messages stored.
30  """
31  # Get a list of cookies, excluding ones with a max-age of 0 (because
32  # they have been marked for deletion).
33  cookie = response.cookies.get(storage.cookie_name)
34  if not cookie or cookie['max-age'] == 0:
35  return 0
36  data = storage._decode(cookie.value)
37  if not data:
38  return 0

102  # When storing as a cookie, the cookie has constant overhead of approx
