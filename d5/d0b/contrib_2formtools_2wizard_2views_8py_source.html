123  @classmethod
124  def get_initkwargs(cls, form_list=None, initial_dict=None,
125  instance_dict=None, condition_dict=None, *args, **kwargs):
126  """
127  Creates a dict with all needed parameters for the form wizard instances.
128 
129  * `form_list` - is a list of forms. The list entries can be single form
130  classes or tuples of (`step_name`, `form_class`). If you pass a list
131  of forms, the wizardview will convert the class list to
132  (`zero_based_counter`, `form_class`). This is needed to access the
133  form for a specific step.
134  * `initial_dict` - contains a dictionary of initial data dictionaries.
135  The key should be equal to the `step_name` in the `form_list` (or
136  the str of the zero based counter - if no step_names added in the
137  `form_list`)
138  * `instance_dict` - contains a dictionary whose values are model
139  instances if the step is based on a ``ModelForm`` and querysets if
140  the step is based on a ``ModelFormSet``. The key should be equal to
141  the `step_name` in the `form_list`. Same rules as for `initial_dict`
142  apply.
143  * `condition_dict` - contains a dictionary of boolean values or
144  callables. If the value of for a specific `step_name` is callable it
145  will be called with the wizardview instance as the only argument.
146  If the return value is true, the step's form will be used.
147  """
148 
149  kwargs.update({
150  'initial_dict': initial_dict or kwargs.pop('initial_dict',
151  getattr(cls, 'initial_dict', None)) or {},
152  'instance_dict': instance_dict or kwargs.pop('instance_dict',
153  getattr(cls, 'instance_dict', None)) or {},
154  'condition_dict': condition_dict or kwargs.pop('condition_dict',
155  getattr(cls, 'condition_dict', None)) or {}
156  })
157 
158  form_list = form_list or kwargs.pop('form_list',
159  getattr(cls, 'form_list', None)) or []
160 
161  computed_form_list = SortedDict()
162 
163  assert len(form_list) > 0, 'at least one form is needed'
164 
165  # walk through the passed form list
166  for i, form in enumerate(form_list):
167  if isinstance(form, (list, tuple)):
168  # if the element is a tuple, add the tuple to the new created
169  # sorted dictionary.
170  computed_form_list[six.text_type(form[0])] = form[1]
171  else:
172  # if not, add the form with a zero based counter as unicode
173  computed_form_list[six.text_type(i)] = form
174 
175  # walk through the new created list of forms
176  for form in six.itervalues(computed_form_list):
177  if issubclass(form, formsets.BaseFormSet):
178  # if the element is based on BaseFormSet (FormSet/ModelFormSet)
179  # we need to override the form variable.

182  # file_storage added to the wizardview (by subclassing).
183  for field in six.itervalues(form.base_fields):
184  if (isinstance(field, forms.FileField) and
185  not hasattr(cls, 'file_storage')):
186  raise NoFileStorageConfigured(
187  "You need to define 'file_storage' in your "
188  "wizard view in order to handle file uploads.")
189

191  kwargs['form_list'] = computed_form_list

198  def get_form_list(self):
199  """
200  This method returns a form_list based on the initial form list but
201  checks if there is a condition method/value in the condition_list.
202  If an entry exists in the condition list, it will call/read the value
203  and respect the result. (True means add the form, False means ignore
204  the form)
205 
206  The form_list is always generated on the fly because condition methods
207  could use data from other (maybe previous forms).
208  """
209  form_list = SortedDict()
210  for form_key, form_class in six.iteritems(self.form_list):
211  # try to fetch the value from condition list, by default, the form
212  # gets passed to the new list.
213  condition = self.condition_dict.get(form_key, True)
214  if callable(condition):
215  # call the value if needed, passes the current instance.
216  condition = condition(self)
217  if condition:
218  form_list[form_key] = form_class
219  return form_list
