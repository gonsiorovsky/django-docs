160  """
161  Resolve strings to objects using standard import and attribute
162  syntax.
163  """
164  name = s.split('.')
165  used = name.pop(0)
166  try:
167  found = self.importer(used)
168  for frag in name:
169  used += '.' + frag
170  try:
171  found = getattr(found, frag)

186  def cfg_convert(self, value):
187  """Default converter for the cfg:// protocol."""
188  rest = value
189  m = self.WORD_PATTERN.match(rest)
190  if m is None:
191  raise ValueError("Unable to convert %r" % value)
192  else:
193  rest = rest[m.end():]
194  d = self.config[m.groups()[0]]
195  #print d, rest
196  while rest:
197  m = self.DOT_PATTERN.match(rest)
198  if m:
199  d = d[m.groups()[0]]
200  else:
201  m = self.INDEX_PATTERN.match(rest)
202  if m:
203  idx = m.groups()[0]
204  if not self.DIGIT_PATTERN.match(idx):
205  d = d[idx]
206  else:
207  try:
208  n = int(idx) # try as number first (most likely)
209  d = d[n]

211  d = d[idx]

213  rest = rest[m.end():]

220  def convert(self, value):
221  """
222  Convert values to an appropriate type. dicts, lists and tuples are
223  replaced by their converting alternatives. Strings are checked to
224  see if they have a conversion format and are converted if they do.
225  """
226  if not isinstance(value, ConvertingDict) and isinstance(value, dict):
227  value = ConvertingDict(value)
228  value.configurator = self
229  elif not isinstance(value, ConvertingList) and isinstance(value, list):
230  value = ConvertingList(value)
231  value.configurator = self
232  elif not isinstance(value, ConvertingTuple) and\
233  isinstance(value, tuple):
234  value = ConvertingTuple(value)
235  value.configurator = self
236  elif isinstance(value, six.string_types): # str for py3k
237  m = self.CONVERT_PATTERN.match(value)
238  if m:
239  d = m.groupdict()
240  prefix = d['prefix']
241  converter = self.value_converters.get(prefix, None)
242  if converter:
243  suffix = d['suffix']
244  converter = getattr(self, converter)
245  value = converter(suffix)
246  return value

276 
277  config = self.config
278  if 'version' not in config:
279  raise ValueError("dictionary doesn't specify a version")
280  if config['version'] != 1:
281  raise ValueError("Unsupported version: %s" % config['version'])
282  incremental = config.pop('incremental', False)
283  EMPTY_DICT = {}
284  logging._acquireLock()
285  try:
286  if incremental:
287  handlers = config.get('handlers', EMPTY_DICT)
288  # incremental handler config only if handler name
289  # ties in to logging._handlers (Python 2.7)
290  if sys.version_info[:2] == (2, 7):
291  for name in handlers:
292  if name not in logging._handlers:
293  raise ValueError('No handler found with '
294  'name %r' % name)
295  else:
296  try:
297  handler = logging._handlers[name]
298  handler_config = handlers[name]
299  level = handler_config.get('level', None)
300  if level:
301  handler.setLevel(_checkLevel(level))
302  except StandardError as e:
303  raise ValueError('Unable to configure handler '
304  '%r: %s' % (name, e))
305  loggers = config.get('loggers', EMPTY_DICT)
306  for name in loggers:
307  try:
308  self.configure_logger(name, loggers[name], True)
309  except StandardError as e:
310  raise ValueError('Unable to configure logger '
311  '%r: %s' % (name, e))
312  root = config.get('root', None)
313  if root:
314  try:
315  self.configure_root(root, True)
316  except StandardError as e:
317  raise ValueError('Unable to configure root '
318  'logger: %s' % e)
319  else:
320  disable_existing = config.pop('disable_existing_loggers', True)
321 
322  logging._handlers.clear()
323  del logging._handlerList[:]
324 
325  # Do formatters first - they don't refer to anything else
326  formatters = config.get('formatters', EMPTY_DICT)
327  for name in formatters:
328  try:
329  formatters[name] = self.configure_formatter(
330  formatters[name])
331  except StandardError as e:
332  raise ValueError('Unable to configure '
333  'formatter %r: %s' % (name, e))
334  # Next, do filters - they don't refer to anything else, either
335  filters = config.get('filters', EMPTY_DICT)
336  for name in filters:
337  try:
338  filters[name] = self.configure_filter(filters[name])
339  except StandardError as e:
340  raise ValueError('Unable to configure '
341  'filter %r: %s' % (name, e))
342 
343  # Next, do handlers - they refer to formatters and filters
344  # As handlers can refer to other handlers, sort the keys
345  # to allow a deterministic order of configuration
346  handlers = config.get('handlers', EMPTY_DICT)
347  for name in sorted(handlers):
348  try:
349  handler = self.configure_handler(handlers[name])
350  handler.name = name
351  handlers[name] = handler
352  except StandardError as e:
353  raise ValueError('Unable to configure handler '
354  '%r: %s' % (name, e))
355  # Next, do loggers - they refer to handlers and filters
356 
357  #we don't want to lose the existing loggers,
358  #since other threads may have pointers to them.
359  #existing is set to contain all existing loggers,
360  #and as we go through the new configuration we
361  #remove any which are configured. At the end,
362  #what's left in existing is the set of loggers
363  #which were in the previous configuration but
364  #which are not in the new configuration.
365  root = logging.root
366  existing = list(root.manager.loggerDict)
367  #The list needs to be sorted so that we can
368  #avoid disabling child loggers of explicitly
369  #named loggers. With a sorted list it is easier
370  #to find the child loggers.
371  existing.sort()
372  #We'll keep the list of existing loggers
373  #which are children of named loggers here...
374  child_loggers = []
375  #now set up the new ones...
376  loggers = config.get('loggers', EMPTY_DICT)
377  for name in loggers:
378  if name in existing:
379  i = existing.index(name)
380  prefixed = name + "."
381  pflen = len(prefixed)
382  num_existing = len(existing)
383  i = i + 1 # look at the entry after name
384  while (i < num_existing) and\
385  (existing[i][:pflen] == prefixed):
386  child_loggers.append(existing[i])
387  i = i + 1
388  existing.remove(name)
389  try:
390  self.configure_logger(name, loggers[name])
391  except StandardError as e:
392  raise ValueError('Unable to configure logger '
393  '%r: %s' % (name, e))
394 
395  #Disable any old loggers. There's no point deleting
396  #them as other threads may continue to hold references
397  #and by disabling them, you stop them doing any logging.
398  #However, don't disable children of named loggers, as that's
399  #probably not what was intended by the user.
400  for log in existing:
401  logger = root.manager.loggerDict[log]
402  if log in child_loggers:
403  logger.level = logging.NOTSET
404  logger.handlers = []
405  logger.propagate = True
406  elif disable_existing:
407  logger.disabled = True
408 
409  # And finally, do the root logger
410  root = config.get('root', None)
411  if root:
412  try:
413  self.configure_root(root)
414  except StandardError as e:
415  raise ValueError('Unable to configure root '
416  'logger: %s' % e)
417  finally:
418  logging._releaseLock()
