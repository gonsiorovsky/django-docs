203 class override(object):
204  """
205  Temporarily set the time zone for the current thread.
206 
207  This is a context manager that uses ``~django.utils.timezone.activate()``
208  to set the timezone on entry, and restores the previously active timezone
209  on exit.
210 
211  The ``timezone`` argument must be an instance of a ``tzinfo`` subclass, a
212  time zone name, or ``None``. If is it a time zone name, pytz is required.
213  If it is ``None``, Django enables the default time zone.
214  """
215  def __init__(self, timezone):
216  self.timezone = timezone
217  self.old_timezone = getattr(_active, 'value', None)

221  deactivate()
