141 def sort_dependencies(app_list):
142  """Sort a list of app,modellist pairs into a single list of models.
143 
144  The single list of models is sorted so that any model with a natural key
145  is serialized before a normal model, and any model with a natural key
146  dependency has it's dependencies serialized first.
147  """
148  from django.db.models import get_model, get_models
149  # Process the list of models, and get the list of dependencies
150  model_dependencies = []
151  models = set()
152  for app, model_list in app_list:
153  if model_list is None:
154  model_list = get_models(app)
155 
156  for model in model_list:
157  models.add(model)
158  # Add any explicitly defined dependencies
159  if hasattr(model, 'natural_key'):
160  deps = getattr(model.natural_key, 'dependencies', [])
161  if deps:
162  deps = [get_model(*d.split('.')) for d in deps]
163  else:
164  deps = []
165 
166  # Now add a dependency for any FK or M2M relation with
167  # a model that defines a natural key
168  for field in model._meta.fields:
169  if hasattr(field.rel, 'to'):
170  rel_model = field.rel.to
171  if hasattr(rel_model, 'natural_key') and rel_model != model:
172  deps.append(rel_model)
173  for field in model._meta.many_to_many:
174  rel_model = field.rel.to
175  if hasattr(rel_model, 'natural_key') and rel_model != model:
176  deps.append(rel_model)
177  model_dependencies.append((model, deps))
178 
179  model_dependencies.reverse()
180  # Now sort the models to ensure that dependencies are met. This
181  # is done by repeatedly iterating over the input list of models.
182  # If all the dependencies of a given model are in the final list,
183  # that model is promoted to the end of the final list. This process
184  # continues until the input list is empty, or we do a full iteration
185  # over the input models without promoting a model to the final list.
186  # If we do a full iteration without a promotion, that means there are
187  # circular dependencies in the list.
188  model_list = []
189  while model_dependencies:

191  changed = False
192  while model_dependencies:
193  model, deps = model_dependencies.pop()
194

196  # on the final model list, or not on the original serialization list,
197  # then we've found another model with all it's dependencies satisfied.
198  found = True

200  if not candidate:
