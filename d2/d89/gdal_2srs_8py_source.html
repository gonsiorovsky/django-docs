49  def __init__(self, srs_input=''):
50  """
51  Creates a GDAL OSR Spatial Reference object from the given input.
52  The input may be string of OGC Well Known Text (WKT), an integer
53  EPSG code, a PROJ.4 string, and/or a projection "well known" shorthand
54  string (one of 'WGS84', 'WGS72', 'NAD27', 'NAD83').
55  """
56  srs_type = 'user'
57 
58  if isinstance(srs_input, six.string_types):
59  # Encoding to ASCII if unicode passed in.
60  if isinstance(srs_input, six.text_type):
61  srs_input = srs_input.encode('ascii')
62  try:
63  # If SRID is a string, e.g., '4326', then make acceptable
64  # as user input.
65  srid = int(srs_input)
66  srs_input = 'EPSG:%d' % srid
67  except ValueError:
68  pass
69  elif isinstance(srs_input, six.integer_types):
70  # EPSG integer code was input.
71  srs_type = 'epsg'
72  elif isinstance(srs_input, self.ptr_type):
73  srs = srs_input
74  srs_type = 'ogr'
75  else:
76  raise TypeError('Invalid SRS type "%s"' % srs_type)
77 
78  if srs_type == 'ogr':
79  # Input is already an SRS pointer.
80  srs = srs_input
81  else:
82  # Creating a new SRS pointer, using the string buffer.
83  buf = c_char_p(b'')
84  srs = capi.new_srs(buf)
85 
86  # If the pointer is NULL, throw an exception.
87  if not srs:
88  raise SRSException('Could not create spatial reference from: %s' % srs_input)
89  else:
90  self.ptr = srs
91 
92  # Importing from either the user input string or an integer SRID.
93  if srs_type == 'user':
94  self.import_user_input(srs_input)
95  elif srs_type == 'epsg':
96  self.import_epsg(srs_input)

102  def __getitem__(self, target):
103  """
104  Returns the value of the given string attribute node, None if the node
105  doesn't exist. Can also take a tuple as a parameter, (target, child),
106  where child is the index of the attribute in the WKT. For example:
107 
108  >>> wkt = 'GEOGCS["WGS 84", DATUM["WGS_1984, ... AUTHORITY["EPSG","4326"]]')
109  >>> srs = SpatialReference(wkt) # could also use 'WGS84', or 4326
110  >>> print(srs['GEOGCS'])
111  WGS 84
112  >>> print(srs['DATUM'])
113  WGS_1984
114  >>> print(srs['AUTHORITY'])
115  EPSG
116  >>> print(srs['AUTHORITY', 1]) # The authority value
117  4326
118  >>> print(srs['TOWGS84', 4]) # the fourth value in this wkt
119  0
120  >>> print(srs['UNIT|AUTHORITY']) # For the units authority, have to use the pipe symbole.
121  EPSG
122  >>> print(srs['UNIT|AUTHORITY', 1]) # The authority value for the untis
123  9122
124  """
125  if isinstance(target, tuple):
126  return self.attr_value(*target)
127  else:
128  return self.attr_value(target)
