483  def check_constraints(self, table_names=None):
484  """
485  Checks each table name in `table_names` for rows with invalid foreign key references. This method is
486  intended to be used in conjunction with `disable_constraint_checking()` and `enable_constraint_checking()`, to
487  determine if rows with invalid references were entered while constraint checks were off.
488 
489  Raises an IntegrityError on the first invalid foreign key reference encountered (if any) and provides
490  detailed information about the invalid reference in the error message.
491 
492  Backends can override this method if they can more directly apply constraint checking (e.g. via "SET CONSTRAINTS
493  ALL IMMEDIATE")
494  """
495  cursor = self.cursor()
496  if table_names is None:
497  table_names = self.introspection.table_names(cursor)
498  for table_name in table_names:
499  primary_key_column_name = self.introspection.get_primary_key_column(cursor, table_name)
500  if not primary_key_column_name:
501  continue
502  key_columns = self.introspection.get_key_columns(cursor, table_name)
503  for column_name, referenced_table_name, referenced_column_name in key_columns:
504  cursor.execute("""
505  SELECT REFERRING.`%s`, REFERRING.`%s` FROM `%s` as REFERRING
506  LEFT JOIN `%s` as REFERRED
507  ON (REFERRING.`%s` = REFERRED.`%s`)
508  WHERE REFERRING.`%s` IS NOT NULL AND REFERRED.`%s` IS NULL"""
509  % (primary_key_column_name, column_name, table_name, referenced_table_name,
510  column_name, referenced_column_name, column_name, referenced_column_name))
511  for bad_row in cursor.fetchall():
512  raise utils.IntegrityError("The row in table '%s' with primary key '%s' has an invalid "
513  "foreign key: %s.%s contains a value '%s' that does not have a corresponding value in %s.%s."
514  % (table_name, bad_row[0],
515  table_name, column_name, bad_row[1],
516  referenced_table_name, referenced_column_name))
