132 def encode_multipart(boundary, data):
133  """
134  Encodes multipart POST data from a dictionary of form values.
135 
136  The key will be used as the form data name; the value will be transmitted
137  as content. If the value is a file, the contents of the file will be sent
138  as an application/octet-stream; otherwise, str(value) will be sent.
139  """
140  lines = []
141  to_bytes = lambda s: force_bytes(s, settings.DEFAULT_CHARSET)
142 
143  # Not by any means perfect, but good enough for our purposes.
144  is_file = lambda thing: hasattr(thing, "read") and callable(thing.read)
145 
146  # Each bit of the multipart form data could be either a form value or a
147  # file, or a *list* of form values and/or files. Remember that HTTP field
148  # names can be duplicated!
149  for (key, value) in data.items():
150  if is_file(value):
151  lines.extend(encode_file(boundary, key, value))
152  elif not isinstance(value, six.string_types) and is_iterable(value):
153  for item in value:
154  if is_file(item):
155  lines.extend(encode_file(boundary, key, item))
156  else:
157  lines.extend([to_bytes(val) for val in [
158  '--%s' % boundary,
159  'Content-Disposition: form-data; name="%s"' % key,
160  '',
161  item
162  ]])
163  else:
164  lines.extend([to_bytes(val) for val in [
165  '--%s' % boundary,
166  'Content-Disposition: form-data; name="%s"' % key,
167  '',
168  value
169  ]])
170 
171  lines.extend([
172  to_bytes('--%s--' % boundary),
173  b'',
174  ])
175  return b'\r\n'.join(lines)

177 def encode_file(boundary, key, file):
178  to_bytes = lambda s: force_bytes(s, settings.DEFAULT_CHARSET)
179  content_type = mimetypes.guess_type(file.name)[0]
180  if content_type is None:
181  content_type = 'application/octet-stream'
182  return [
183  to_bytes('--%s' % boundary),
184  to_bytes('Content-Disposition: form-data; name="%s"; filename="%s"' \
185  % (key, os.path.basename(file.name))),
186  to_bytes('Content-Type: %s' % content_type),
187  b'',
