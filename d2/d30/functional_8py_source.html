405  def total_ordering(cls):
406  """Class decorator that fills in missing ordering methods"""
407  convert = {
408  '__lt__': [('__gt__', lambda self, other: not (self < other or self == other)),
409  ('__le__', lambda self, other: self < other or self == other),
410  ('__ge__', lambda self, other: not self < other)],
411  '__le__': [('__ge__', lambda self, other: not self <= other or self == other),
412  ('__lt__', lambda self, other: self <= other and not self == other),
413  ('__gt__', lambda self, other: not self <= other)],
414  '__gt__': [('__lt__', lambda self, other: not (self > other or self == other)),
415  ('__ge__', lambda self, other: self > other or self == other),
416  ('__le__', lambda self, other: not self > other)],
417  '__ge__': [('__le__', lambda self, other: (not self >= other) or self == other),
418  ('__gt__', lambda self, other: self >= other and not self == other),
419  ('__lt__', lambda self, other: not self >= other)]
420  }
421  roots = set(dir(cls)) & set(convert)
422  if not roots:
423  raise ValueError('must define at least one ordering operation: < > <= >=')
424  root = max(roots) # prefer __lt__ to __le__ to __gt__ to __ge__
425  for opname, opfunc in convert[root]:
426  if opname not in roots:
427  opfunc.__name__ = opname
428  opfunc.__doc__ = getattr(int, opname).__doc__
429  setattr(cls, opname, opfunc)
430  return cls
