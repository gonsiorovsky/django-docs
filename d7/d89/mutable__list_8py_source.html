4 This module contains a base type which provides list-style mutations
5 without specific data storage methods.
6 
7 See also http://www.aryehleib.com/MutableLists.html
8 
9 Author: Aryeh Leib Taurog.
10 """
11 from django.utils.functional import total_ordering
12 from django.utils import six
13 from django.utils.six.moves import xrange

15 @total_ordering
16 class ListMixin(object):
17  """
18  A base class which provides complete list interface.
19  Derived classes must call ListMixin's __init__() function
20  and implement the following:
21 
22  function _get_single_external(self, i):
23  Return single item with index i for general use.
24  The index i will always satisfy 0 <= i < len(self).
25 
26  function _get_single_internal(self, i):
27  Same as above, but for use within the class [Optional]
28  Note that if _get_single_internal and _get_single_internal return
29  different types of objects, _set_list must distinguish
30  between the two and handle each appropriately.
31 
32  function _set_list(self, length, items):
33  Recreate the entire object.
34 
35  NOTE: items may be a generator which calls _get_single_internal.
36  Therefore, it is necessary to cache the values in a temporary:
37  temp = list(items)
38  before clobbering the original storage.
39 
40  function _set_single(self, i, value):
41  Set the single item at index i to value [Optional]
42  If left undefined, all mutations will result in rebuilding
43  the object using _set_list.
44 
45  function __len__(self):
46  Return the length
47 
48  int _minlength:
49  The minimum legal length [Optional]
50 
51  int _maxlength:
52  The maximum legal length [Optional]
53 
54  type or tuple _allowed:
55  A type or tuple of allowed item types [Optional]
56 
57  class _IndexError:
58  The type of exception to be raise on invalid index [Optional]
59  """
60

63  _IndexError = IndexError

236  def _rebuild(self, newLen, newItems):

238  raise ValueError('Must have at least %d items' % self._minlength)

247  def _checkindex(self, index, correct=True):
248  length = len(self)

261  "Assign values to a slice of the object"

280  def _assign_extended_slice_rebuild(self, start, stop, step, valueList):
281  'Assign an extended slice by rebuilding entire list'
282  indexList = range(start, stop, step)
283  # extended slice, only allow assigning slice of same size
284  if len(valueList) != len(indexList):
285  raise ValueError('attempt to assign sequence of size %d '
286  'to extended slice of size %d'
287  % (len(valueList), len(indexList)))
288 
289  # we're not changing the length of the sequence
290  newLen = len(self)
291  newVals = dict(zip(indexList, valueList))
292  def newItems():
293  for i in xrange(newLen):
294  if i in newVals:
295  yield newVals[i]
296  else:
297  yield self._get_single_internal(i)
298 
299  self._rebuild(newLen, newItems())

301  def _assign_extended_slice(self, start, stop, step, valueList):
302  'Assign an extended slice by re-assigning individual items'
303  indexList = range(start, stop, step)
304  # extended slice, only allow assigning slice of same size
305  if len(valueList) != len(indexList):
306  raise ValueError('attempt to assign sequence of size %d '
307  'to extended slice of size %d'
308  % (len(valueList), len(indexList)))
309 
310  for i, val in zip(indexList, valueList):
311  self._set_single(i, val)

313  def _assign_simple_slice(self, start, stop, valueList):
314  'Assign a simple slice; Can assign slice of any length'
315  origLen = len(self)
316  stop = max(start, stop)
317  newLen = origLen - stop + start + len(valueList)
318  def newItems():

320  if i == start:
