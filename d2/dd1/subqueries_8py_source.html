51  def delete_qs(self, query, using):
52  """
53  Delete the queryset in one SQL query (if possible). For simple queries
54  this is done by copying the query.query.where to self.query, for
55  complex queries by using subquery.
56  """
57  innerq = query.query
58  # Make sure the inner query has at least one table in use.
59  innerq.get_initial_alias()
60  # The same for our new query.
61  self.get_initial_alias()
62  innerq_used_tables = [t for t in innerq.tables
63  if innerq.alias_refcount[t]]
64  if ((not innerq_used_tables or innerq_used_tables == self.tables)
65  and not len(innerq.having)):
66  # There is only the base table in use in the query, and there are
67  # no aggregate filtering going on.
68  self.where = innerq.where
69  else:
70  pk = query.model._meta.pk
71  if not connections[using].features.update_can_self_select:
72  # We can't do the delete using subquery.
73  values = list(query.values_list('pk', flat=True))
74  if not values:
75  return
76  self.delete_batch(values, using)
77  return
78  else:
79  innerq.clear_select_clause()
80  innerq.select = [SelectInfo((self.get_initial_alias(), pk.column), None)]
81  values = innerq
82  where = self.where_class()
83  where.add((Constraint(None, pk.column, pk), 'in', values), AND)
84  self.where = where
85  self.get_compiler(using).execute_sql(None)

124  def add_update_values(self, values):
125  """
126  Convert a dictionary of field name to value mappings into an update
127  query. This is the entry point for the public update() method on
128  querysets.
129  """
130  values_seq = []
131  for name, val in six.iteritems(values):
132  field, model, direct, m2m = self.get_meta().get_field_by_name(name)
133  if not direct or m2m:
134  raise FieldError('Cannot update model field %r (only non-relations and foreign keys permitted).' % field)
135  if model:
136  self.add_related_update(model, field, val)
137  continue
138  values_seq.append((field, model, val))
139  return self.add_update_fields(values_seq)

153  def add_related_update(self, model, field, value):
154  """
155  Adds (name, value) to an update query for an ancestor model.
156 
157  Updates are coalesced so that we only run one update query per ancestor.
158  """
159  try:
160  self.related_updates[model].append((field, None, value))
161  except KeyError:
162  self.related_updates[model] = [(field, None, value)]
163 
164  def get_related_updates(self):
165  """
166  Returns a list of query objects: one for each update required to an
167  ancestor model. Each query will have the same filtering conditions as
168  the current query but will only update a single table.
169  """
170  if not self.related_updates:
171  return []
172  result = []
173  for model, values in six.iteritems(self.related_updates):
174  query = UpdateQuery(model)
175  query.values = values
176  if self.related_ids is not None:
177  query.add_filter(('pk__in', self.related_ids))
178  result.append(query)
179  return result
