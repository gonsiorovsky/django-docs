15 def method_decorator(decorator):
16  """
17  Converts a function decorator into a method decorator
18  """
19  # 'func' is a function at the time it is passed to _dec, but will eventually
20  # be a method of the class it is defined it.
21  def _dec(func):
22  def _wrapper(self, *args, **kwargs):
23  @decorator
24  def bound_func(*args2, **kwargs2):
25  return func(self, *args2, **kwargs2)
26  # bound_func has the signature that 'decorator' expects i.e. no
27  # 'self' argument, but it is a closure over self so it can call
28  # 'func' correctly.
29  return bound_func(*args, **kwargs)
30  # In case 'decorator' adds attributes to the function it decorates, we
31  # want to copy those. We don't have access to bound_func in this scope,
32  # but we can cheat by using it on a dummy function.
33  @decorator
34  def dummy(*args, **kwargs):
35  pass
36  update_wrapper(_wrapper, dummy)
37  # Need to preserve any existing attributes of 'func', including the name.
38  update_wrapper(_wrapper, func)
39 
40  return _wrapper
41  update_wrapper(_dec, decorator)
42  # Change the name to aid debugging.
43  _dec.__name__ = 'method_decorator(%s)' % decorator.__name__
44  return _dec
