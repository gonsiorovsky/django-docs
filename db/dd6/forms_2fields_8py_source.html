313  Validates that the input is a decimal number. Returns a Decimal
314  instance. Returns None for empty values. Ensures that there are no more
315  than max_digits in the number, and no more than decimal_places digits
316  after the decimal point.
317  """

321  value = formats.sanitize_separators(value)
322  value = smart_text(value).strip()

329  def validate(self, value):
330  super(DecimalField, self).validate(value)
331  if value in self.empty_values:
332  return
333  # Check for NaN, Inf and -Inf values. We can't compare directly for NaN,
334  # since it is never equal to itself. However, NaN is the only value that
335  # isn't equal to itself, so we can use this to identify NaN
336  if value != value or value == Decimal("Inf") or value == Decimal("-Inf"):
337  raise ValidationError(self.error_messages['invalid'], code='invalid')
338  sign, digittuple, exponent = value.as_tuple()
339  decimals = abs(exponent)
340  # digittuple doesn't include any leading zeros.
341  digits = len(digittuple)
342  if decimals > digits:
343  # We have leading zeros up to or past the decimal point. Count
344  # everything past the decimal point as a digit. We do not count
345  # 0 before the decimal point as a digit since that would mean
346  # we would not allow max_digits = decimal_places.
347  digits = decimals
348  whole_digits = digits - decimals
349 
350  if self.max_digits is not None and digits > self.max_digits:
351  raise ValidationError(
352  self.error_messages['max_digits'],
353  code='max_digits',
354  params={'max': self.max_digits},
355  )
356  if self.decimal_places is not None and decimals > self.decimal_places:
357  raise ValidationError(
358  self.error_messages['max_decimal_places'],
359  code='max_decimal_places',
360  params={'max': self.decimal_places},
361  )
362  if (self.max_digits is not None and self.decimal_places is not None
363  and whole_digits > (self.max_digits - self.decimal_places)):
364  raise ValidationError(
365  self.error_messages['max_whole_digits'],
366  code='max_whole_digits',

936 class MultiValueField(Field):
937  """
938  A Field that aggregates the logic of multiple Fields.
939 
940  Its clean() method takes a "decompressed" list of values, which are then
941  cleaned into a single value according to self.fields. Each value in
942  this list is cleaned by the corresponding field -- the first value is
943  cleaned by the first field, the second value is cleaned by the second
944  field, etc. Once all fields are cleaned, the list of clean values is
945  "compressed" into a single value.
946 
947  Subclasses should not have to implement clean(). Instead, they must
948  implement compress(), which takes a list of valid values and returns a
949  "compressed" version of those values -- a single value.
950 
951  You'll probably want to use this with MultiWidget.
952  """
953  default_error_messages = {
954  'invalid': _('Enter a list of values.'),
955  }
956 
957  def __init__(self, fields=(), *args, **kwargs):
958  super(MultiValueField, self).__init__(*args, **kwargs)
959  # Set 'required' to False on the individual fields, because the
960  # required validation will be handled by MultiValueField, not by those
961  # individual fields.
962  for f in fields:
963  f.required = False
964  self.fields = fields
965 
966  def validate(self, value):
967  pass
968 
969  def clean(self, value):
970  """
971  Validates every value in the given list. A value is validated against
972  the corresponding Field in self.fields.
973 
974  For example, if this MultiValueField was instantiated with
975  fields=(DateField(), TimeField()), clean() would call
976  DateField.clean(value[0]) and TimeField.clean(value[1]).
977  """
978  clean_data = []
979  errors = ErrorList()
980  if not value or isinstance(value, (list, tuple)):
981  if not value or not [v for v in value if v not in self.empty_values]:
982  if self.required:
983  raise ValidationError(self.error_messages['required'], code='required')
984  else:
985  return self.compress([])
986  else:
987  raise ValidationError(self.error_messages['invalid'], code='invalid')
988  for i, field in enumerate(self.fields):
989  try:
990  field_value = value[i]
991  except IndexError:
992  field_value = None
993  if self.required and field_value in self.empty_values:
994  raise ValidationError(self.error_messages['required'], code='required')
995  try:
996  clean_data.append(field.clean(field_value))
997  except ValidationError as e:
998  # Collect all validation errors in a single list, which we'll
999  # raise at the end of clean(), rather than raising a single
1000  # exception for the first error we encounter.
1001  errors.extend(e.error_list)
1002  if errors:
1003  raise ValidationError(errors)
1004 
1005  out = self.compress(clean_data)
1006  self.validate(out)
1007  self.run_validators(out)
1008  return out
