67  def assertURLEqual(self, url, expected, parse_qs=False):
68  """
69  Given two URLs, make sure all their components (the ones given by
70  urlparse) are equal, only comparing components that are present in both
71  URLs.
72  If `parse_qs` is True, then the querystrings are parsed with QueryDict.
73  This is useful if you don't want the order of parameters to matter.
74  Otherwise, the query strings are compared as-is.
75  """
76  fields = ParseResult._fields
77 
78  for attr, x, y in zip(fields, urlparse(url), urlparse(expected)):
79  if parse_qs and attr == 'query':
80  x, y = QueryDict(x), QueryDict(y)
81  if x and y and x != y:
82  self.fail("%r != %r (%s doesn't match)" % (url, expected, attr))
