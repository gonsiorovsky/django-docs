35 def rfc2822_date(date):
36  # We can't use strftime() because it produces locale-dependent results, so
37  # we have to map english month and day names manually
38  months = ('Jan', 'Feb', 'Mar', 'Apr', 'May', 'Jun', 'Jul', 'Aug', 'Sep', 'Oct', 'Nov', 'Dec',)
39  days = ('Mon', 'Tue', 'Wed', 'Thu', 'Fri', 'Sat', 'Sun')
40  # Support datetime objects older than 1900
41  date = datetime_safe.new_datetime(date)
42  # We do this ourselves to be timezone aware, email.Utils is not tz aware.
43  dow = days[date.weekday()]
44  month = months[date.month - 1]
45  time_str = date.strftime('%s, %%d %s %%Y %%H:%%M:%%S ' % (dow, month))
46  if six.PY2: # strftime returns a byte string in Python 2
47  time_str = time_str.decode('utf-8')
48  if is_aware(date):
49  offset = date.tzinfo.utcoffset(date)
50  timezone = (offset.days * 24 * 60) + (offset.seconds // 60)
51  hour, minute = divmod(timezone, 60)
52  return time_str + '%+03d%02d' % (hour, minute)
53  else:
54  return time_str + '-0000'

70 def get_tag_uri(url, date):
71  """
72  Creates a TagURI.
73 
74  See http://web.archive.org/web/20110514113830/http://diveintomark.org/archives/2004/05/28/howto-atom-id
75  """
76  bits = urlparse(url)
77  d = ''
78  if date is not None:
79  d = ',%s' % datetime_safe.new_datetime(date).strftime('%Y-%m-%d')
