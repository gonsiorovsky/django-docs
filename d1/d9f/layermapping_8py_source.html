319  def verify_ogr_field(self, ogr_field, model_field):
320  """
321  Verifies if the OGR Field contents are acceptable to the Django
322  model field. If they are, the verified value is returned,
323  otherwise the proper exception is raised.
324  """
325  if (isinstance(ogr_field, OFTString) and
326  isinstance(model_field, (models.CharField, models.TextField))):
327  if self.encoding:
328  # The encoding for OGR data sources may be specified here
329  # (e.g., 'cp437' for Census Bureau boundary files).
330  val = force_text(ogr_field.value, self.encoding)
331  else:
332  val = ogr_field.value
333  if model_field.max_length and len(val) > model_field.max_length:
334  raise InvalidString('%s model field maximum string length is %s, given %s characters.' %
335  (model_field.name, model_field.max_length, len(val)))
336  elif isinstance(ogr_field, OFTReal) and isinstance(model_field, models.DecimalField):
337  try:
338  # Creating an instance of the Decimal value to use.
339  d = Decimal(str(ogr_field.value))
340  except:
341  raise InvalidDecimal('Could not construct decimal from: %s' % ogr_field.value)
342 
343  # Getting the decimal value as a tuple.
344  dtup = d.as_tuple()
345  digits = dtup[1]
346  d_idx = dtup[2] # index where the decimal is
347 
348  # Maximum amount of precision, or digits to the left of the decimal.
349  max_prec = model_field.max_digits - model_field.decimal_places
350 
351  # Getting the digits to the left of the decimal place for the
352  # given decimal.
353  if d_idx < 0:
354  n_prec = len(digits[:d_idx])
355  else:
356  n_prec = len(digits) + d_idx
357 
358  # If we have more than the maximum digits allowed, then throw an
359  # InvalidDecimal exception.
360  if n_prec > max_prec:
361  raise InvalidDecimal('A DecimalField with max_digits %d, decimal_places %d must round to an absolute value less than 10^%d.' %
362  (model_field.max_digits, model_field.decimal_places, max_prec))
363  val = d
364  elif isinstance(ogr_field, (OFTReal, OFTString)) and isinstance(model_field, models.IntegerField):
365  # Attempt to convert any OFTReal and OFTString value to an OFTInteger.
366  try:
367  val = int(ogr_field.value)
368  except:
369  raise InvalidInteger('Could not construct integer from: %s' % ogr_field.value)
370  else:
371  val = ogr_field.value
372  return val
