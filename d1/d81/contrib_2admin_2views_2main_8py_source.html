282  def get_ordering(self, request, queryset):
283  """
284  Returns the list of ordering fields for the change list.
285  First we check the get_ordering() method in model admin, then we check
286  the object's default ordering. Then, any manually-specified ordering
287  from the query string overrides anything. Finally, a deterministic
288  order is guaranteed by ensuring the primary key is used as the last
289  ordering field.
290  """
291  params = self.params
292  ordering = list(self.model_admin.get_ordering(request)
293  or self._get_default_ordering())
294  if ORDER_VAR in params:
295  # Clear ordering and used params
296  ordering = []
297  order_params = params[ORDER_VAR].split('.')
298  for p in order_params:
299  try:
300  none, pfx, idx = p.rpartition('-')
301  field_name = self.list_display[int(idx)]
302  order_field = self.get_ordering_field(field_name)
303  if not order_field:
304  continue # No 'admin_order_field', skip it
305  ordering.append(pfx + order_field)
306  except (IndexError, ValueError):
307  continue # Invalid ordering specified, skip it.
308 
309  # Add the given query's ordering fields, if any.
310  ordering.extend(queryset.query.order_by)
311 
312  # Ensure that the primary key is systematically present in the list of
313  # ordering fields so we can guarantee a deterministic order across all
314  # database backends.
315  pk_name = self.lookup_opts.pk.name
316  if not (set(ordering) & set(['pk', '-pk', pk_name, '-' + pk_name])):
317  # The two sets do not intersect, meaning the pk isn't present. So
318  # we add it.
319  ordering.append('-pk')
320 
321  return ordering

371  # filters).
372  qs = qs.filter(**remaining_lookup_params)
