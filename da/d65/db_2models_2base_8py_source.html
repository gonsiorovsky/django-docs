610  def _save_table(self, raw=False, cls=None, force_insert=False,
611  force_update=False, using=None, update_fields=None):

613  Does the heavy-lifting involved in saving. Updates or inserts the data
614  for a single table.
615  """

626  raise ValueError("Cannot force an update in save() with no primary key.")
627  updated = False
628  # If possible, try an UPDATE. If that doesn't update anything, do an INSERT.
629  if pk_set and not force_insert:

631  values = [(f, None, (getattr(self, f.attname) if raw else f.pre_save(self, False)))
632  for f in non_pks]

635  forced_update)

637  raise DatabaseError("Forced update did not affect any rows.")
638  if update_fields and not updated:
639  raise DatabaseError("Save with update_fields did not affect any rows.")
640  if not updated:
641  if meta.order_with_respect_to:
642  # If this is a model with an order_with_respect_to
643  # autopopulate the _order field
644  field = meta.order_with_respect_to

646  **{field.name: getattr(self, field.attname)}).count()

659  def _do_update(self, base_qs, using, pk_val, values, update_fields, forced_update):
660  """
661  This method will try to update the model. If the model was updated (in
662  the sense that an update query was done and a matching row was found
663  from the DB) the method will return True.
664  """
665  filtered = base_qs.filter(pk=pk_val)
666  if not values:
667  # We can end up here when saving a model in inheritance chain where
668  # update_fields doesn't target any field in current model. In that
669  # case we just say the update succeeded. Another case ending up here
670  # is a model with just PK - in that case check that the PK still
671  # exists.
672  return update_fields is not None or filtered.exists()
673  if self._meta.select_on_save and not forced_update:
674  if filtered.exists():
675  filtered._update(values)
676  return True
677  else:
678  return False

763  def _get_unique_checks(self, exclude=None):
764  """
765  Gather a list of checks to perform. Since validate_unique could be
766  called from a ModelForm, some fields may have been excluded; we can't
767  perform a unique check on a model that is missing fields involved
768  in that check.
769  Fields that did not validate should also be excluded, but they need
770  to be passed in via the exclude argument.
771  """
772  if exclude is None:
773  exclude = []
774  unique_checks = []
775 
776  unique_togethers = [(self.__class__, self._meta.unique_together)]
777  for parent_class in self._meta.parents.keys():
778  if parent_class._meta.unique_together:
779  unique_togethers.append((parent_class, parent_class._meta.unique_together))
780 
781  for model_class, unique_together in unique_togethers:
782  for check in unique_together:
783  for name in check:
784  # If this is an excluded field, don't add this check.
785  if name in exclude:
786  break
787  else:
788  unique_checks.append((model_class, tuple(check)))
789 
790  # These are checks for the unique_for_<date/year/month>.
791  date_checks = []
792 
793  # Gather a list of checks for fields declared as unique and add them to
794  # the list of checks.
795 
796  fields_with_class = [(self.__class__, self._meta.local_fields)]
797  for parent_class in self._meta.parents.keys():
798  fields_with_class.append((parent_class, parent_class._meta.local_fields))
799 
800  for model_class, fields in fields_with_class:
801  for f in fields:
802  name = f.name
803  if name in exclude:
804  continue
805  if f.unique:
806  unique_checks.append((model_class, (name,)))
807  if f.unique_for_date and f.unique_for_date not in exclude:
808  date_checks.append((model_class, 'date', name, f.unique_for_date))
809  if f.unique_for_year and f.unique_for_year not in exclude:
810  date_checks.append((model_class, 'year', name, f.unique_for_year))
811  if f.unique_for_month and f.unique_for_month not in exclude:
812  date_checks.append((model_class, 'month', name, f.unique_for_month))
813  return unique_checks, date_checks

815  def _perform_unique_checks(self, unique_checks):
816  errors = {}
817 
818  for model_class, unique_check in unique_checks:
819  # Try to look up an existing object with the same values as this
820  # object's values for all the unique field.
821 
822  lookup_kwargs = {}
823  for field_name in unique_check:
824  f = self._meta.get_field(field_name)
825  lookup_value = getattr(self, f.attname)
826  if lookup_value is None:
827  # no value, skip the lookup
828  continue
829  if f.primary_key and not self._state.adding:
830  # no need to check for unique primary key when editing
831  continue
832  lookup_kwargs[str(field_name)] = lookup_value
833 
834  # some fields were skipped, no reason to do the check
835  if len(unique_check) != len(lookup_kwargs):
836  continue
837 
838  qs = model_class._default_manager.filter(**lookup_kwargs)
839 
840  # Exclude the current object from the query if we are editing an
841  # instance (as opposed to creating a new one)
842  # Note that we need to use the pk as defined by model_class, not
843  # self.pk. These can be different fields because model inheritance
844  # allows single model to have effectively multiple primary keys.
845  # Refs #17615.
846  model_class_pk = self._get_pk_val(model_class._meta)
847  if not self._state.adding and model_class_pk is not None:
848  qs = qs.exclude(pk=model_class_pk)
849  if qs.exists():
850  if len(unique_check) == 1:
851  key = unique_check[0]
852  else:
853  key = NON_FIELD_ERRORS
854  errors.setdefault(key, []).append(self.unique_error_message(model_class, unique_check))
855 
856  return errors
857 
858  def _perform_date_checks(self, date_checks):
859  errors = {}
860  for model_class, lookup_type, field, unique_for in date_checks:
861  lookup_kwargs = {}
862  # there's a ticket to add a date lookup, we can remove this special
863  # case if that makes it's way in
864  date = getattr(self, unique_for)
865  if date is None:
866  continue
867  if lookup_type == 'date':
868  lookup_kwargs['%s__day' % unique_for] = date.day
869  lookup_kwargs['%s__month' % unique_for] = date.month
870  lookup_kwargs['%s__year' % unique_for] = date.year
871  else:
872  lookup_kwargs['%s__%s' % (unique_for, lookup_type)] = getattr(date, lookup_type)
873  lookup_kwargs[field] = getattr(self, field)
874 
875  qs = model_class._default_manager.filter(**lookup_kwargs)
876  # Exclude the current object from the query if we are editing an
877  # instance (as opposed to creating a new one)
878  if not self._state.adding and self.pk is not None:
879  qs = qs.exclude(pk=self.pk)
880 
881  if qs.exists():
882  errors.setdefault(field, []).append(
883  self.date_error_message(lookup_type, field, unique_for)
884  )
885  return errors

918  def full_clean(self, exclude=None, validate_unique=True):
919  """
920  Calls clean_fields, clean, and validate_unique, on the model,
921  and raises a ``ValidationError`` for any errors that occurred.
922  """
923  errors = {}
924  if exclude is None:
925  exclude = []
926 
927  try:
928  self.clean_fields(exclude=exclude)
929  except ValidationError as e:
930  errors = e.update_error_dict(errors)
931 
932  # Form.clean() is run even if other validation fails, so do the
933  # same with Model.clean() for consistency.
934  try:
935  self.clean()
936  except ValidationError as e:
937  errors = e.update_error_dict(errors)
938 
939  # Run unique checks, but only for fields that passed validation.
940  if validate_unique:
941  for name in errors.keys():
942  if name != NON_FIELD_ERRORS and name not in exclude:
943  exclude.append(name)
944  try:
945  self.validate_unique(exclude=exclude)
946  except ValidationError as e:
947  errors = e.update_error_dict(errors)
948 
949  if errors:
950  raise ValidationError(errors)

952  def clean_fields(self, exclude=None):
953  """
954  Cleans all fields and raises a ValidationError containing message_dict
955  of all validation errors if any occur.
956  """
957  if exclude is None:
958  exclude = []
959 
960  errors = {}
961  for f in self._meta.fields:
962  if f.name in exclude:
963  continue
964  # Skip validation for empty fields with blank=True. The developer
965  # is responsible for making sure they have a valid value.
966  raw_value = getattr(self, f.attname)
967  if f.blank and raw_value in f.empty_values:
968  continue
969  try:
970  setattr(self, f.attname, f.clean(raw_value, self))
971  except ValidationError as e:
972  errors[f.name] = e.error_list
973 
974  if errors:
975  raise ValidationError(errors)
