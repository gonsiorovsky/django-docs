34 def find_management_module(app_name):
35  """
36  Determines the path to the management module for the given app_name,
37  without actually importing the application or the management module.
38 
39  Raises ImportError if the management module cannot be found for any reason.
40  """
41  parts = app_name.split('.')
42  parts.append('management')
43  parts.reverse()
44  part = parts.pop()
45  path = None
46 
47  # When using manage.py, the project module is added to the path,
48  # loaded, then removed from the path. This means that
49  # testproject.testapp.models can be loaded in future, even if
50  # testproject isn't in the path. When looking for the management
51  # module, we need look for the case where the project name is part
52  # of the app_name but the project directory itself isn't on the path.
53  try:
54  f, path, descr = imp.find_module(part, path)
55  except ImportError as e:
56  if os.path.basename(os.getcwd()) != part:
57  raise e
58  else:
59  if f:
60  f.close()
61 
62  while parts:
63  part = parts.pop()
64  f, path, descr = imp.find_module(part, [path] if path else None)
65  if f:
66  f.close()
67  return path

78 def get_commands():
79  """
80  Returns a dictionary mapping command names to their callback applications.
81 
82  This works by looking for a management.commands package in django.core, and
83  in each installed application -- if a commands package exists, all commands
84  in that package are registered.
85 
86  Core commands are always included. If a settings module has been
87  specified, user-defined commands will also be included.
88 
89  The dictionary is in the format {command_name: app_name}. Key-value
90  pairs from this dictionary can then be used in calls to
91  load_command_class(app_name, command_name)
92 
93  If a specific version of a command must be loaded (e.g., with the
94  startapp command), the instantiated module can be placed in the
95  dictionary in place of the application name.
96 
97  The dictionary is cached on the first call and reused on subsequent
98  calls.
99  """
100  global _commands
101  if _commands is None:
102  _commands = dict([(name, 'django.core') for name in find_commands(__path__[0])])
103 
104  # Find the installed apps
105  from django.conf import settings
106  try:
107  apps = settings.INSTALLED_APPS
108  except ImproperlyConfigured:
109  # Still useful for commands that do not require functional settings,
110  # like startproject or help
111  apps = []
112 
113  # Find and load the management module for each installed app.
114  for app_name in apps:
115  try:
116  path = find_management_module(app_name)
117  _commands.update(dict([(name, app_name)
118  for name in find_commands(path)]))
119  except ImportError:
120  pass # No management module - ignore this app
121 
122  return _commands

186  def _process_args(self, largs, rargs, values):
187  """
188  Overrides OptionParser._process_args to exclusively handle default
189  options and ignore args and other options.
190 
191  This overrides the behavior of the super class, which stop parsing
192  at the first unrecognized option.
193  """
194  while rargs:
195  arg = rargs[0]
196  try:
197  if arg[0:2] == "--" and len(arg) > 2:
198  # process a single long option (possibly with value(s))
199  # the superclass code pops the arg off rargs
200  self._process_long_opt(rargs, values)
201  elif arg[:1] == "-" and len(arg) > 1:
202  # process a cluster of short options (possibly with
203  # value(s) for the last one only)
204  # the superclass code pops the arg off rargs
205  self._process_short_opts(rargs, values)
206  else:
207  # it's either a non-default option or an arg
208  # either way, add it to the args list so we can keep
209  # dealing with options
210  del rargs[0]
211  raise Exception
212  except:
213  largs.append(arg)

276  """
277  Output completion suggestions for BASH.
278 
279  The output of this function is passed to BASH's `COMREPLY` variable and
280  treated as completion suggestions. `COMREPLY` expects a space
281  separated string as the result.
282 
283  The `COMP_WORDS` and `COMP_CWORD` BASH environment variables are used
284  to get information about the cli input. Please refer to the BASH
285  man-page for more information about this variables.
286 
287  Subcommand options are saved as pairs. A pair consists of
288  the long option string (e.g. '--exclude') and a boolean
289  value indicating if the option requires arguments. When printing to
290  stdout, a equal sign is appended to options which require arguments.
291 
292  Note: If debugging this function, it is recommended to write the debug
293  output in a separate file. Otherwise the debug output will be treated
294  and formatted as potential completion suggestions.
295  """

297  if 'DJANGO_AUTO_COMPLETE' not in os.environ:
298  return

309  options = [('--help', None)]

312  if cword == 1:

314  # subcommand options

353  being run, creates a parser appropriate to that command, and runs it.
354  """

356  # These options could affect the commands that are available, so they

359  version=get_version(),

367

369  subcommand = self.argv[1]

372

384  # 'django-admin.py --help' to work, for backwards compatibility.
385  elif self.argv[1:] == ['--version']:
386  # LaxOptionParser already takes care of printing the version.
387  pass
388  elif self.argv[1:] in (['--help'], ['-h']):
389  parser.print_lax_help()
390  sys.stdout.write(self.main_help_text() + '\n')
