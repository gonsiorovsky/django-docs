32 def get_declared_fields(bases, attrs, with_base_fields=True):
33  """
34  Create a list of form field instances from the passed in 'attrs', plus any
35  similar fields on the base classes (in 'bases'). This is used by both the
36  Form and ModelForm metaclasses.
37 
38  If 'with_base_fields' is True, all fields from the bases are used.
39  Otherwise, only fields in the 'declared_fields' attribute on the bases are
40  used. The distinction is useful in ModelForm subclassing.
41  Also integrates any additional media definitions.
42  """
43  fields = [(field_name, attrs.pop(field_name)) for field_name, obj in list(six.iteritems(attrs)) if isinstance(obj, Field)]
44  fields.sort(key=lambda x: x[1].creation_counter)
45 
46  # If this class is subclassing another Form, add that Form's fields.
47  # Note that we loop over the bases in *reverse*. This is necessary in
48  # order to preserve the correct order of fields.
49  if with_base_fields:
50  for base in bases[::-1]:
51  if hasattr(base, 'base_fields'):
52  fields = list(six.iteritems(base.base_fields)) + fields
53  else:
54  for base in bases[::-1]:
55  if hasattr(base, 'declared_fields'):
56  fields = list(six.iteritems(base.declared_fields)) + fields
57 
58  return SortedDict(fields)

146  def _html_output(self, normal_row, error_row, row_ender, help_text_html, errors_on_separate_row):
147  "Helper function for outputting HTML. Used by as_table(), as_ul(), as_p()."
148  top_errors = self.non_field_errors() # Errors that should be displayed above all fields.
149  output, hidden_fields = [], []
150 
151  for name, field in self.fields.items():
152  html_class_attr = ''
153  bf = self[name]
154  # Escape and cache in local variable.
155  bf_errors = self.error_class([conditional_escape(error) for error in bf.errors])
156  if bf.is_hidden:
157  if bf_errors:
158  top_errors.extend(
159  [_('(Hidden field %(name)s) %(error)s') % {'name': name, 'error': force_text(e)}
160  for e in bf_errors])
161  hidden_fields.append(six.text_type(bf))
162  else:
163  # Create a 'class="..."' atribute if the row should have any
164  # CSS classes applied.
165  css_classes = bf.css_classes()
166  if css_classes:
167  html_class_attr = ' class="%s"' % css_classes
168 
169  if errors_on_separate_row and bf_errors:
170  output.append(error_row % force_text(bf_errors))
171 
172  if bf.label:
173  label = conditional_escape(force_text(bf.label))
174  label = bf.label_tag(label) or ''
175  else:
176  label = ''
177 
178  if field.help_text:
179  help_text = help_text_html % force_text(field.help_text)
180  else:
181  help_text = ''
182 
183  output.append(normal_row % {
184  'errors': force_text(bf_errors),
185  'label': force_text(label),
186  'field': six.text_type(bf),
187  'help_text': help_text,
188  'html_class_attr': html_class_attr
189  })
190 
191  if top_errors:
192  output.insert(0, error_row % force_text(top_errors))
193 
194  if hidden_fields: # Insert any hidden fields in the last row.
195  str_hidden = ''.join(hidden_fields)
196  if output:
197  last_row = output[-1]
198  # Chop off the trailing row_ender (e.g. '</td></tr>') and
199  # insert the hidden fields.
200  if not last_row.endswith(row_ender):
201  # This can happen in the as_p() case (and possibly others
202  # that users write): if there are only top errors, we may
203  # not be able to conscript the last row for our purposes,
204  # so insert a new, empty row.
205  last_row = (normal_row % {'errors': '', 'label': '',
206  'field': '', 'help_text':'',
207  'html_class_attr': html_class_attr})
208  output.append(last_row)
209  output[-1] = last_row[:-len(row_ender)] + str_hidden + row_ender
210  else:
211  # If there aren't any rows in the output, just append the
212  # hidden fields.
213  output.append(str_hidden)
214  return mark_safe('\n'.join(output))

277  def _clean_fields(self):
278  for name, field in self.fields.items():
279  # value_from_datadict() gets the data from the data dictionaries.
280  # Each widget type knows how to retrieve its own data, because some
281  # widgets split data over several HTML fields.
282  value = field.widget.value_from_datadict(self.data, self.files, self.add_prefix(name))
283  try:
284  if isinstance(field, FileField):
285  initial = self.initial.get(name, field.initial)
286  value = field.clean(value, initial)
287  else:
288  value = field.clean(value)
289  self.cleaned_data[name] = value
290  if hasattr(self, 'clean_%s' % name):
291  value = getattr(self, 'clean_%s' % name)()
292  self.cleaned_data[name] = value
293  except ValidationError as e:
294  self._errors[name] = self.error_class(e.messages)
295  if name in self.cleaned_data:
296  del self.cleaned_data[name]

515  def label_tag(self, contents=None, attrs=None, label_suffix=None):
516  """
517  Wraps the given contents in a <label>, if the field has an ID attribute.
518  contents should be 'mark_safe'd to avoid HTML escaping. If contents
519  aren't given, uses the field's HTML-escaped label.
520 
521  If attrs are given, they're used as HTML attributes on the <label> tag.
522 
523  label_suffix allows overriding the form's label_suffix.
524  """
525  contents = contents or self.label
526  # Only add the suffix if the label does not end in punctuation.
527  label_suffix = label_suffix if label_suffix is not None else self.form.label_suffix
528  # Translators: If found as last label character, these punctuation
529  # characters will prevent the default label_suffix to be appended to the label
530  if label_suffix and contents and contents[-1] not in _(':?.!'):
531  contents = format_html('{0}{1}', contents, label_suffix)

533  id_ = widget.attrs.get('id') or self.auto_id
534  if id_:

536  if id_for_label:
537  attrs = dict(attrs or {}, **{'for': id_for_label})
538  attrs = flatatt(attrs) if attrs else ''
539  contents = format_html('<label{0}>{1}</label>', attrs, contents)
