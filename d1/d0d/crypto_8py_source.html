31 def salted_hmac(key_salt, value, secret=None):
32  """
33  Returns the HMAC-SHA1 of 'value', using a key generated from key_salt and a
34  secret (which defaults to settings.SECRET_KEY).
35 
36  A different key_salt should be passed in for every application of HMAC.
37  """
38  if secret is None:
39  secret = settings.SECRET_KEY
40 
41  # We need to generate a derived key from our base key. We can do this by
42  # passing the key_salt and our base key through a pseudo-random function and
43  # SHA1 works nicely.
44  key = hashlib.sha1((key_salt + secret).encode('utf-8')).digest()
45 
46  # If len(key_salt + secret) > sha_constructor().block_size, the above
47  # line is redundant and could be replaced by key = key_salt + secret, since
48  # the hmac module does the same thing for keys longer than the block size.
49  # However, we need to ensure that we *always* do this.
50  return hmac.new(key, msg=force_bytes(value), digestmod=hashlib.sha1)

60  a 71-bit value. log_2((26+26+10)^12) =~ 71 bits
61  """

79 def constant_time_compare(val1, val2):
80  """
81  Returns True if the two strings are equal, False otherwise.
82 
83  The time taken is independent of the number of characters that match.
84 
85  For the sake of simplicity, this function executes in constant time only
86  when the two strings have the same length. It short-circuits when they
87  have different lengths. Since Django only uses it to compare hashes of
88  known expected length, this is acceptable.
89  """
