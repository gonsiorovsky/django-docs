52  def register(self, model_or_iterable, admin_class=None, **options):
53  """
54  Registers the given model(s) with the given admin class.
55 
56  The model(s) should be Model classes, not instances.
57 
58  If an admin class isn't given, it will use ModelAdmin (the default
59  admin options). If keyword arguments are given -- e.g., list_display --
60  they'll be applied as options to the admin class.
61 
62  If a model is already registered, this will raise AlreadyRegistered.
63 
64  If a model is abstract, this will raise ImproperlyConfigured.
65  """
66  if not admin_class:
67  admin_class = ModelAdmin
68 
69  if isinstance(model_or_iterable, ModelBase):
70  model_or_iterable = [model_or_iterable]
71  for model in model_or_iterable:
72  if model._meta.abstract:
73  raise ImproperlyConfigured('The model %s is abstract, so it '
74  'cannot be registered with admin.' % model.__name__)
75 
76  if model in self._registry:
77  raise AlreadyRegistered('The model %s is already registered' % model.__name__)
78 
79  # Ignore the registration if the model has been
80  # swapped out.
81  if not model._meta.swapped:
82  # If we got **options then dynamically construct a subclass of
83  # admin_class with those **options.
84  if options:
85  # For reasons I don't quite understand, without a __module__
86  # the created class appears to "live" in the wrong place,
87  # which causes issues later on.
88  options['__module__'] = __name__
89  admin_class = type("%sAdmin" % model.__name__, (admin_class,), options)
90 
91  if admin_class is not ModelAdmin and settings.DEBUG:
92  admin_class.validate(model)
93 
94  # Instantiate the admin class to save in the registry
95  self._registry[model] = admin_class(model, self)

166  def admin_view(self, view, cacheable=False):
167  """
168  Decorator to create an admin view attached to this ``AdminSite``. This
169  wraps the view and provides permission checking by calling
170  ``self.has_permission``.
171 
172  You'll want to use this from within ``AdminSite.get_urls()``:
173 
174  class MyAdminSite(AdminSite):
175 
176  def get_urls(self):
177  from django.conf.urls import patterns, url
178 
179  urls = super(MyAdminSite, self).get_urls()
180  urls += patterns('',
181  url(r'^my_view/$', self.admin_view(some_view))
182  )
183  return urls
184 
185  By default, admin_views are marked non-cacheable using the
186  ``never_cache`` decorator. If the view can be safely cached, set
187  cacheable=True.
188  """
189  def inner(request, *args, **kwargs):
190  if LOGIN_FORM_KEY in request.POST and request.user.is_authenticated():
191  auth_logout(request)
192  if not self.has_permission(request):
193  if request.path == reverse('admin:logout',
194  current_app=self.name):
195  index_path = reverse('admin:index', current_app=self.name)
196  return HttpResponseRedirect(index_path)
197  return self.login(request)
198  return view(request, *args, **kwargs)
199  if not cacheable:
200  inner = never_cache(inner)
201  # We add csrf_protect here so this function can be used as a utility
202  # function for any view, without having to repeat 'csrf_protect'.
203  if not getattr(view, 'csrf_exempt', False):
204  inner = csrf_protect(inner)
205  return update_wrapper(inner, view)
