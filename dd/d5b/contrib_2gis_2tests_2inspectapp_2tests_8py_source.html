101 def get_ogr_db_string():
102  """
103  Construct the DB string that GDAL will use to inspect the database.
104  GDAL will create its own connection to the database, so we re-use the
105  connection settings from the Django test.
106  """
107  db = connections.databases['default']
108 
109  # Map from the django backend into the OGR driver name and database identifier
110  # http://www.gdal.org/ogr/ogr_formats.html
111  #
112  # TODO: Support Oracle (OCI).
113  drivers = {
114  'django.contrib.gis.db.backends.postgis': ('PostgreSQL', "PG:dbname='%(db_name)s'", ' '),
115  'django.contrib.gis.db.backends.mysql': ('MySQL', 'MYSQL:"%(db_name)s"', ','),
116  'django.contrib.gis.db.backends.spatialite': ('SQLite', '%(db_name)s', '')
117  }
118 
119  drv_name, db_str, param_sep = drivers[db['ENGINE']]
120 
121  # Ensure that GDAL library has driver support for the database.
122  try:
123  Driver(drv_name)
124  except:
125  return None
126 
127  # SQLite/Spatialite in-memory databases
128  if db['NAME'] == ":memory:":
129  return None
130 
131  # Build the params of the OGR database connection string
132  params = [db_str % {'db_name': db['NAME']}]
133  def add(key, template):
134  value = db.get(key, None)
135  # Don't add the parameter if it is not in django's settings
136  if value:
137  params.append(template % value)
138  add('HOST', "host='%s'")
139  add('PORT', "port='%s'")
140  add('USER', "user='%s'")
141  add('PASSWORD', "password='%s'")
142 
143  return param_sep.join(params)
