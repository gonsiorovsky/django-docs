194 def iri_to_uri(iri):
195  """
196  Convert an Internationalized Resource Identifier (IRI) portion to a URI
197  portion that is suitable for inclusion in a URL.
198 
199  This is the algorithm from section 3.1 of RFC 3987. However, since we are
200  assuming input is either UTF-8 or unicode already, we can simplify things a
201  little from the full method.
202 
203  Returns an ASCII string containing the encoded result.
204  """
205  # The list of safe characters here is constructed from the "reserved" and
206  # "unreserved" characters specified in sections 2.2 and 2.3 of RFC 3986:
207  # reserved = gen-delims / sub-delims
208  # gen-delims = ":" / "/" / "?" / "#" / "[" / "]" / "@"
209  # sub-delims = "!" / "$" / "&" / "'" / "(" / ")"
210  # / "*" / "+" / "," / ";" / "="
211  # unreserved = ALPHA / DIGIT / "-" / "." / "_" / "~"
212  # Of the unreserved characters, urllib.quote already considers all but
213  # the ~ safe.
214  # The % character is also added to the list of safe characters here, as the
215  # end of section 3.1 of RFC 3987 specifically mentions that % must not be
216  # converted.
217  if iri is None:
218  return iri
219  return quote(force_bytes(iri), safe=b"/#%[]=:;$&()+,!?*@'~")
