21  def get_object(self, queryset=None):
22  """
23  Returns the object the view is displaying.
24 
25  By default this requires `self.queryset` and a `pk` or `slug` argument
26  in the URLconf, but subclasses can override this to return any object.
27  """
28  # Use a custom queryset if provided; this is required for subclasses
29  # like DateDetailView
30  if queryset is None:
31  queryset = self.get_queryset()
32 
33  # Next, try looking up by primary key.
34  pk = self.kwargs.get(self.pk_url_kwarg, None)
35  slug = self.kwargs.get(self.slug_url_kwarg, None)
36  if pk is not None:
37  queryset = queryset.filter(pk=pk)
38 
39  # Next, try looking up by slug.
40  elif slug is not None:
41  slug_field = self.get_slug_field()
42  queryset = queryset.filter(**{slug_field: slug})
43 
44  # If none of those are defined, it's an error.
45  else:
46  raise AttributeError("Generic detail view %s must be called with "
47  "either an object pk or a slug."
48  % self.__class__.__name__)
49

52  obj = queryset.get()
53  except ObjectDoesNotExist:
