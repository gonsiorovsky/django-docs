746 class MultiWidget(Widget):
747  """
748  A widget that is composed of multiple widgets.
749 
750  Its render() method is different than other widgets', because it has to
751  figure out how to split a single value for display in multiple widgets.
752  The ``value`` argument can be one of two things:
753 
754  * A list.
755  * A normal value (e.g., a string) that has been "compressed" from
756  a list of values.
757 
758  In the second case -- i.e., if the value is NOT a list -- render() will
759  first "decompress" the value into a list before rendering it. It does so by
760  calling the decompress() method, which MultiWidget subclasses must
761  implement. This method takes a single "compressed" value and returns a
762  list.
763 
764  When render() does its HTML rendering, each value in the list is rendered
765  with the corresponding widget -- the first value is rendered in the first
766  widget, the second value is rendered in the second widget, etc.
767 
768  Subclasses may implement format_output(), which takes the list of rendered
769  widgets and returns a string of HTML that formats them any way you'd like.
770 
771  You'll probably want to use this class with MultiValueField.
772  """
773  def __init__(self, widgets, attrs=None):
774  self.widgets = [w() if isinstance(w, type) else w for w in widgets]
775  super(MultiWidget, self).__init__(attrs)
776 
777  def render(self, name, value, attrs=None):
778  if self.is_localized:
779  for widget in self.widgets:
780  widget.is_localized = self.is_localized
781  # value is a list of values, each corresponding to a widget
782  # in self.widgets.
783  if not isinstance(value, list):
784  value = self.decompress(value)
785  output = []
786  final_attrs = self.build_attrs(attrs)
787  id_ = final_attrs.get('id', None)
788  for i, widget in enumerate(self.widgets):
789  try:
790  widget_value = value[i]
791  except IndexError:
792  widget_value = None
793  if id_:
794  final_attrs = dict(final_attrs, id='%s_%s' % (id_, i))
795  output.append(widget.render(name + '_%s' % i, widget_value, final_attrs))
796  return mark_safe(self.format_output(output))
797 
798  def id_for_label(self, id_):
799  # See the comment for RadioSelect.id_for_label()
800  if id_:
801  id_ += '_0'
802  return id_
803 
804  def value_from_datadict(self, data, files, name):
805  return [widget.value_from_datadict(data, files, name + '_%s' % i) for i, widget in enumerate(self.widgets)]

808  """
809  Given a list of rendered widgets (as strings), returns a Unicode string
810  representing the HTML for the whole lot.
811 
812  This hook allows you to format the HTML design of the widgets, if
813  needed.
814  """
815  return ''.join(rendered_widgets)
