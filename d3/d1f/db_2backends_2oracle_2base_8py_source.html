184  raise ValueError("Invalid time zone name: %s" % tzname)
185  # Convert from UTC to local time, returning TIMESTAMP WITH TIME ZONE.
186  result = "(FROM_TZ(%s, '0:00') AT TIME ZONE '%s')" % (field_name, tzname)

223  def convert_values(self, value, field):
224  if isinstance(value, Database.LOB):
225  value = value.read()
226  if field and field.get_internal_type() == 'TextField':
227  value = force_text(value)
228 
229  # Oracle stores empty strings as null. We need to undo this in
230  # order to adhere to the Django convention of using the empty
231  # string instead of null, but only if the field accepts the
232  # empty string.
233  if value is None and field and field.empty_strings_allowed:
234  value = ''
235  # Convert 1 or 0 to True or False
236  elif value in (1, 0) and field and field.get_internal_type() in ('BooleanField', 'NullBooleanField'):
237  value = bool(value)
238  # Force floats to the correct type
239  elif value is not None and field and field.get_internal_type() == 'FloatField':
240  value = float(value)
241  # Convert floats to decimals
242  elif value is not None and field and field.get_internal_type() == 'DecimalField':
243  value = util.typecast_decimal(field.format_number(value))
244  # cx_Oracle always returns datetime.datetime objects for
245  # DATE and TIMESTAMP columns, but Django wants to see a
246  # python datetime.date, .time, or .datetime. We use the type
247  # of the Field to determine which to cast to, but it's not
248  # always available.
249  # As a workaround, we cast to date if all the time-related
250  # values are 0, or to time if the date is 1/1/1900.
251  # This could be cleaned a bit by adding a method to the Field
252  # classes to normalize values from the database (the to_python
253  # method is used for validation and isn't what we want here).
254  elif isinstance(value, Database.Timestamp):
255  if field and field.get_internal_type() == 'DateTimeField':
256  pass
257  elif field and field.get_internal_type() == 'DateField':
258  value = value.date()
259  elif field and field.get_internal_type() == 'TimeField' or (value.year == 1900 and value.month == value.day == 1):

261  elif value.hour == value.minute == value.second == value.microsecond == 0:
262  value = value.date()

314  def quote_name(self, name):
315  # SQL92 requires delimited (quoted) names to be case-sensitive. When
316  # not quoted, Oracle has case-insensitive behavior for identifiers, but
317  # always defaults to uppercase.
318  # We simplify things by making Oracle identifiers always uppercase.
319  if not name.startswith('"') and not name.endswith('"'):
320  name = '"%s"' % util.truncate_name(name.upper(),
321  self.max_name_length())
322  # Oracle puts the query text into a (query % args) construct, so % signs
323  # in names need to be escaped. The '%%' will be collapsed back to '%' at
324  # that stage so we aren't really making the name longer here.
325  name = name.replace('%','%%')
326  return name.upper()

790  def _format_params(self, params):
791  try:
792  return dict((k,OracleParam(v, self, True)) for k,v in params.items())
793  except AttributeError:

796  def _guess_input_sizes(self, params_list):
797  # Try dict handling; if that fails, treat as sequence
798  if hasattr(params_list[0], 'keys'):
799  sizes = {}

801  for k, value in params.items():
802  if value.input_size:

821  def _fix_for_params(self, query, params):
822  # cx_Oracle wants no trailing ';' for SQL statements. For PL/SQL, it
823  # it does want a trailing ';' but not a trailing '/'. However, these
824  # characters must be included in the original query in case the query
825  # is being passed to SQL*Plus.
826  if query.endswith(';') or query.endswith('/'):
827  query = query[:-1]
828  if params is None:
829  params = []

831  elif hasattr(params, 'keys'):

834  query = convert_unicode(query % args, self.charset)
835  else:
836  # Handle params as sequence
837  args = [(':arg%d' % i) for i in range(len(params))]
838  query = convert_unicode(query % tuple(args), self.charset)
839  return query, self._format_params(params)

919 def _rowfactory(row, cursor):
920  # Cast numeric values as the appropriate Python type based upon the
921  # cursor description, and convert strings to unicode.
922  casted = []
923  for value, desc in zip(row, cursor.description):
924  if value is not None and desc[1] is Database.NUMBER:
925  precision, scale = desc[4:6]
926  if scale == -127:
927  if precision == 0:
928  # NUMBER column: decimal-precision floating point
929  # This will normally be an integer from a sequence,
930  # but it could be a decimal value.
931  if '.' in value:
932  value = decimal.Decimal(value)
933  else:
934  value = int(value)
935  else:
936  # FLOAT column: binary-precision floating point.
937  # This comes from FloatField columns.
938  value = float(value)
939  elif precision > 0:
940  # NUMBER(p,s) column: decimal-precision fixed point.
941  # This comes from IntField and DecimalField columns.
942  if scale == 0:
943  value = int(value)
944  else:
945  value = decimal.Decimal(value)
946  elif '.' in value:
947  # No type information. This normally comes from a
948  # mathematical expression in the SELECT list. Guess int
949  # or Decimal based on whether it has a decimal point.
950  value = decimal.Decimal(value)
951  else:
952  value = int(value)
953  # datetimes are returned as TIMESTAMP, except the results
954  # of "dates" queries, which are returned as DATETIME.
955  elif desc[1] in (Database.TIMESTAMP, Database.DATETIME):
956  # Confirm that dt is naive before overwriting its tzinfo.
957  if settings.USE_TZ and value is not None and timezone.is_naive(value):
958  value = value.replace(tzinfo=timezone.utc)
959  elif desc[1] in (Database.STRING, Database.FIXED_CHAR,
960  Database.LONG_STRING):
961  value = to_unicode(value)
962  casted.append(value)
963  return tuple(casted)
