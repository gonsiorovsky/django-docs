213 def urlize(text, trim_url_limit=None, nofollow=False, autoescape=False):
214  """
215  Converts any URLs in text into clickable links.
216 
217  Works on http://, https://, www. links, and also on links ending in one of
218  the original seven gTLDs (.com, .edu, .gov, .int, .mil, .net, and .org).
219  Links can have trailing punctuation (periods, commas, close-parens) and
220  leading punctuation (opening parens) and it'll still do the right thing.
221 
222  If trim_url_limit is not None, the URLs in link text longer than this limit
223  will truncated to trim_url_limit-3 characters and appended with an elipsis.
224 
225  If nofollow is True, the URLs in link text will get a rel="nofollow"
226  attribute.
227 
228  If autoescape is True, the link text and URLs will get autoescaped.
229  """
230  def trim_url(x, limit=trim_url_limit):
231  if limit is None or len(x) <= limit:
232  return x
233  return '%s...' % x[:max(0, limit - 3)]
234  safe_input = isinstance(text, SafeData)
235  words = word_split_re.split(force_text(text))
236  for i, word in enumerate(words):
237  match = None
238  if '.' in word or '@' in word or ':' in word:
239  # Deal with punctuation.
240  lead, middle, trail = '', word, ''
241  for punctuation in TRAILING_PUNCTUATION:
242  if middle.endswith(punctuation):
243  middle = middle[:-len(punctuation)]
244  trail = punctuation + trail
245  for opening, closing in WRAPPING_PUNCTUATION:
246  if middle.startswith(opening):
247  middle = middle[len(opening):]
248  lead = lead + opening
249  # Keep parentheses at the end only if they're balanced.
250  if (middle.endswith(closing)
251  and middle.count(closing) == middle.count(opening) + 1):
252  middle = middle[:-len(closing)]
253  trail = closing + trail
254 
255  # Make URL we want to point to.
256  url = None
257  nofollow_attr = ' rel="nofollow"' if nofollow else ''
258  if simple_url_re.match(middle):
259  url = smart_urlquote(middle)
260  elif simple_url_2_re.match(middle):
261  url = smart_urlquote('http://%s' % middle)
262  elif not ':' in middle and simple_email_re.match(middle):
263  local, domain = middle.rsplit('@', 1)
264  try:
265  domain = domain.encode('idna').decode('ascii')
266  except UnicodeError:
267  continue
268  url = 'mailto:%s@%s' % (local, domain)
269  nofollow_attr = ''
270 
271  # Make link.
272  if url:
273  trimmed = trim_url(middle)
274  if autoescape and not safe_input:
275  lead, trail = escape(lead), escape(trail)
276  url, trimmed = escape(url), escape(trimmed)
277  middle = '<a href="%s"%s>%s</a>' % (url, nofollow_attr, trimmed)
278  words[i] = mark_safe('%s%s%s' % (lead, middle, trail))
279  else:
280  if safe_input:
281  words[i] = mark_safe(word)
282  elif autoescape:
283  words[i] = escape(word)
284  elif safe_input:
285  words[i] = mark_safe(word)
286  elif autoescape:
287  words[i] = escape(word)
288  return ''.join(words)

291 def clean_html(text):
292  """
293  Clean the given HTML. Specifically, do the following:
294  * Convert <b> and <i> to <strong> and <em>.
295  * Encode all ampersands correctly.
296  * Remove all "target" attributes from <a> tags.
297  * Remove extraneous HTML, such as presentational tags that open and
298  immediately close and <br clear="all">.
299  * Convert hard-coded bullets into HTML unordered lists.
300  * Remove stuff like "<p>&nbsp;&nbsp;</p>", but only if it's at the
301  bottom of the text.
302  """
303  from django.utils.text import normalize_newlines
304  text = normalize_newlines(force_text(text))
305  text = re.sub(r'<(/?)\s*b\s*>', '<\\1strong>', text)
306  text = re.sub(r'<(/?)\s*i\s*>', '<\\1em>', text)
307  text = fix_ampersands(text)
308  # Remove all target="" attributes from <a> tags.
309  text = link_target_attribute_re.sub('\\1', text)
310  # Trim stupid HTML such as <br clear="all">.
311  text = html_gunk_re.sub('', text)
312  # Convert hard-coded bullets into HTML unordered lists.
313  def replace_p_tags(match):
314  s = match.group().replace('</p>', '</li>')
315  for d in DOTS:
316  s = s.replace('<p>%s' % d, '<li>')
317  return '<ul>\n%s\n</ul>' % s
318  text = hard_coded_bullets_re.sub(replace_p_tags, text)
319  # Remove stuff like "<p>&nbsp;&nbsp;</p>", but only if it's at the bottom
320  # of the text.
321  text = trailing_empty_content_re.sub('', text)
322  return text
323 clean_html = allow_lazy(clean_html, six.text_type)
